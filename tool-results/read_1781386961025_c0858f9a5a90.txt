   661тЖТ   661тЖТ    def _clean_json_leak(self, text):
   662тЖТ   662тЖТ        """Limpia texto que tiene restos de formato JSON para mostrar al usuario.
   663тЖТ   663тЖТ        v3: Mas agresivo - tambien limpia JSON parcial al inicio/final del texto.
   664тЖТ   664тЖТ        """
   665тЖТ   665тЖТ        # Si el texto es JSON completo, extraer solo el contenido util
   666тЖТ   666тЖТ        parsed = self._parse_json(text)
   667тЖТ   667тЖТ        if parsed:
   668тЖТ   668тЖТ            # Prioridad: respuesta_final > pensamiento > texto original
   669тЖТ   669тЖТ            if parsed.get("respuesta_final", "").strip():
   670тЖТ   670тЖТ                return parsed["respuesta_final"].strip()
   671тЖТ   671тЖТ            if parsed.get("pensamiento", "").strip():
   672тЖТ   672тЖТ                return parsed["pensamiento"].strip()
   673тЖТ   673тЖТ        # Si no es JSON completo, intentar limpiar restos
   674тЖТ   674тЖТ        # Caso: texto que empieza con JSON parcial
   675тЖТ   675тЖТ        import re as _re
   676тЖТ   676тЖТ        # Remover JSON parcial al inicio: {"pensamiento": "..." ... restos
   677тЖТ   677тЖТ        cleaned = _re.sub(r'^\s*\{[^}]*$', '', text).strip()
   678тЖТ   678тЖТ        # Remover JSON parcial al final
   679тЖТ   679тЖТ        cleaned = _re.sub(r'\{[^}]*$\s*$', '', cleaned).strip()
   680тЖТ   680тЖТ        # Si despues de limpiar quedo vacio, devolver texto original
   681тЖТ   681тЖТ        return cleaned if cleaned else text

   686тЖТ   686тЖТ    def _execute_tool_calls(self, tool_calls, messages):
   687тЖТ   687тЖТ        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   688тЖТ   688тЖТ        from concurrent.futures import ThreadPoolExecutor, as_completed
   689тЖТ   689тЖТ
   690тЖТ   690тЖТ        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   691тЖТ   691тЖТ        filtered_calls = []
   692тЖТ   692тЖТ        for tc in tool_calls:
   693тЖТ   693тЖТ            tool_name = tc["name"]
   694тЖТ   694тЖТ            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   695тЖТ   695тЖТ            self._total_tool_calls += 1
   696тЖТ   696тЖТ
   697тЖТ   697тЖТ            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   698тЖТ   698тЖТ                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   699тЖТ   699тЖТ                break
   700тЖТ   700тЖТ
   701тЖТ   701тЖТ            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   702тЖТ   702тЖТ                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   703тЖТ   703тЖТ                continue
   704тЖТ   704тЖТ
   705тЖТ   705тЖТ            filtered_calls.append(tc)
   706тЖТ   706тЖТ
   707тЖТ   707тЖТ        tool_calls = filtered_calls
   708тЖТ   708тЖТ
   709тЖТ   709тЖТ        # Validar parametros de todos los tool calls primero
   710тЖТ   710тЖТ        for tc in tool_calls:
   711тЖТ   711тЖТ            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   712тЖТ   712тЖТ
   713тЖТ   713тЖТ        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   714тЖТ   714тЖТ        if len(tool_calls) == 1:
   715тЖТ   715тЖТ            return [self._execute_single_tool(tool_calls[0], messages)]
   716тЖТ   716тЖТ
   717тЖТ   717тЖТ        # Si hay multiples tools, ejecutar en paralelo
   718тЖТ   718тЖТ        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   719тЖТ   719тЖТ        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   720тЖТ   720тЖТ        
   721тЖТ   721тЖТ        if has_sequential:
   722тЖТ   722тЖТ            # Ejecutar secuencialmente para evitar race conditions
   723тЖТ   723тЖТ            results = []
   724тЖТ   724тЖТ            for tc in tool_calls:
   725тЖТ   725тЖТ                results.append(self._execute_single_tool(tc, messages))
   726тЖТ   726тЖТ            return results
   727тЖТ   727тЖТ
   728тЖТ   728тЖТ        # Ejecucion paralela para tools de solo lectura
   729тЖТ   729тЖТ        results = [None] * len(tool_calls)
   730тЖТ   730тЖТ        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   731тЖТ   731тЖТ            futures = {}
   732тЖТ   732тЖТ            for i, tc in enumerate(tool_calls):
   733тЖТ   733тЖТ                future = executor.submit(self._execute_single_tool, tc, messages)
   734тЖТ   734тЖТ                futures[future] = i
   735тЖТ   735тЖТ            
   736тЖТ   736тЖТ            for future in as_completed(futures):
   737тЖТ   737тЖТ                idx = futures[future]
   738тЖТ   738тЖТ                try:
   739тЖТ   739тЖТ                    results[idx] = future.result(timeout=30)
   740тЖТ   740тЖТ                except Exception as e:
   741тЖТ   741тЖТ                    results[idx] = f"ERROR: {e}"
   742тЖТ   742тЖТ
   743тЖТ   743тЖТ        return results

   745тЖТ   745тЖТ    def _execute_single_tool(self, tc, messages, max_retries=1):
   746тЖТ   746тЖТ        """Ejecuta un solo tool call con retry automatico."""
   747тЖТ   747тЖТ        import time as _time
   748тЖТ   748тЖТ        tool_name = tc["name"]
   749тЖТ   749тЖТ        tool_params = tc["params"]
   750тЖТ   750тЖТ
   751тЖТ   751тЖТ        self._log(f"Ejecutando: {tool_name}({tool_params})", "execution")
   752тЖТ   752тЖТ
   753тЖТ   753тЖТ        _tool_start = _time.monotonic()
   754тЖТ   754тЖТ        tool_result = self._execute_tool(tool_name, tool_params)
   755тЖТ   755тЖТ
   756тЖТ   756тЖТ        # Retry si fallo y el tool es reintentable
   757тЖТ   757тЖТ        if "ERROR" in tool_result and max_retries > 0:
   758тЖТ   758тЖТ            retryable_errors = ["TIMEOUT", "ConnectionError", "empty", "vacia"]
   759тЖТ   759тЖТ            if any(err in tool_result for err in retryable_errors):
   760тЖТ   760тЖТ                self._log(f"Reintentando {tool_name} (error transitorio)...", "execution")
   761тЖТ   761тЖТ                tool_result = self._execute_tool(tool_name, tool_params)
   762тЖТ   762тЖТ
   763тЖТ   763тЖТ        _tool_elapsed_ms = (_time.monotonic() - _tool_start) * 1000.0
   764тЖТ   764тЖТ        get_metrics().record_tool_call(tool_name, _tool_elapsed_ms)
   765тЖТ   765тЖТ
   766тЖТ   766тЖТ        if "ERROR" in tool_result:
   767тЖТ   767тЖТ            get_metrics().record_error("tool:" + tool_name)
   768тЖТ   768тЖТ
   769тЖТ   769тЖТ        self._log(f"Resultado: {tool_result[:150]}...", "observation")
   770тЖТ   770тЖТ
   771тЖТ   771тЖТ        # Alimentar memoria de trabajo
   772тЖТ   772тЖТ        self.memory.add_step(f"{tool_name}({tool_params})", tool_result[:200])
   773тЖТ   773тЖТ        if "ERROR" in tool_result:
   774тЖТ   774тЖТ            self.memory.set_error(f"{tool_name}: {tool_result[:100]}")
   775тЖТ   775тЖТ        if "PELIGROSO" in tool_result:
   776тЖТ   776тЖТ            self._log(f"Comando peligroso bloqueado: {tool_name}", "warning")
   777тЖТ   777тЖТ        if len(tool_result) > 50 and "ERROR" not in tool_result:
   778тЖТ   778тЖТ            self.memory.remember(
   779тЖТ   779тЖТ                f"Resultado de {tool_name}: {tool_result[:300]}",
   780тЖТ   780тЖТ                metadata={"type": "tool_result", "tool": tool_name}
   781тЖТ   781тЖТ            )
   782тЖТ   782тЖТ
   783тЖТ   783тЖТ        return tool_result
