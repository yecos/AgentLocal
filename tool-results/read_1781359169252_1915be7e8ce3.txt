   251→    # ----------------------------------------------------------
   252→    def _try_direct_intent(self, user_message: str):
   253→        """
   254→        Intenta ejecutar la herramienta directamente basándose en
   255→        pattern matching del mensaje del usuario.
   256→
   257→        Returns:
   258→            (response, thinking_log) si la intención fue ejecutada directamente
   259→            None si no se detectó intención clara (dejar al LLM decidir)
   260→        """
   261→        if not self.intent_parser:
   262→            return None
   263→
   264→        result = parse_direct_intent(user_message)
   265→        if result is None:
   266→            return None
   267→
   268→        tool_name, params, confidence = result
   269→
   270→        # Solo ejecutar directamente si la confianza es alta
   271→        if confidence < 0.80:
   272→            self._log(f"Direct intent detectado pero confianza baja ({confidence}): {tool_name}", "intent")
   273→            return None
   274→
   275→        # No ejecutar directamente herramientas que tienen side effects significativos
   276→        # a menos que la confianza sea muy alta
   277→        high_risk_tools = {"ejecutar_comando", "escribir_archivo", "matar_proceso", "git_operacion"}
   278→        if tool_name in high_risk_tools and confidence < 0.95:
   279→            self._log(f"Direct intent: herramienta de riesgo ({tool_name}), requiere confianza 0.95+, tiene {confidence}", "intent")
   280→            return None
   281→
   282→        self._log(f"Direct intent detectado: {tool_name}({params}) [confianza={confidence}]", "intent")
   283→
   284→        # Ejecutar la herramienta directamente
   285→        try:
   286→            tool_fn = TOOL_FUNCTIONS.get(tool_name)
   287→            if not tool_fn:
   288→                self._log(f"Direct intent: herramienta no encontrada: {tool_name}", "warning")
   289→                return None
   290→
   291→            # Filtrar params vacíos o None
   292→            clean_params = {k: v for k, v in params.items() if v is not None and v != ""}
   293→            if not clean_params:
   294→                # La herramienta necesita parámetros pero no los pudimos extraer
   295→                self._log(f"Direct intent: parámetros vacíos para {tool_name}, dejando al LLM", "intent")
   296→                return None
   297→
   298→            tool_result = tool_fn(**clean_params)
   299→
   300→            # Formatear respuesta amigable
   301→            self._log(f"Direct intent ejecutado exitosamente: {tool_name}", "success")
   302→            self._save_interaction(user_message, tool_result)
   303→
   304→            return tool_result, self.thinking_log
   305→
   306→        except Exception as e:
   307→            self._log(f"Direct intent falló: {tool_name} -> {e}", "error")
   308→            # No retornar error, dejar que el LLM lo intente
   309→            return None
   310→
   311→    def _try_direct_intent_stream(self, user_message: str):
   312→        """
   313→        Versión streaming del Direct Intent Parser.
   314→        Yields eventos SSE para la UI.
   315→
   316→        Returns:
   317→            Generator de eventos SSE si la intención fue ejecutada directamente
   318→            None si no se detectó intención clara
   319→        """
   320→        if not self.intent_parser:
   321→            return None
   322→
   323→        result = parse_direct_intent(user_message)
   324→        if result is None:
   325→            return None
   326→
   327→        tool_name, params, confidence = result
   328→
   329→        # Solo ejecutar directamente si la confianza es alta
   330→        if confidence < 0.80:
   331→            self._log(f"Direct intent detectado pero confianza baja ({confidence}): {tool_name}", "intent")
   332→            return None
   333→
   334→        # No ejecutar directamente herramientas de riesgo sin confianza muy alta
   335→        high_risk_tools = {"ejecutar_comando", "escribir_archivo", "matar_proceso", "git_operacion"}
   336→        if tool_name in high_risk_tools and confidence < 0.95:
   337→            self._log(f"Direct intent: herramienta de riesgo ({tool_name}), requiere confianza 0.95+, tiene {confidence}", "intent")
   338→            return None
   339→
   340→        self._log(f"Direct intent detectado (stream): {tool_name}({params}) [confianza={confidence}]", "intent")
   341→
   342→        # Generar eventos de streaming
   343→        def _stream_direct():
   344→            # Evento: pensamiento
   345→            yield {
   346→                "type": "thinking",
   347→                "data": {
   348→                    "phase": "direct_intent",
   349→                    "message": f"Ejecutando directamente: {tool_name}",
   350→                    "iteration": 1,
   351→                    "confidence": confidence,
   352→                }
   353→            }
   354→
   355→            # Evento: tool start
   356→            yield {
   357→                "type": "tool_start",
   358→                "data": {
   359→                    "name": tool_name,
   360→                    "params": params,
   361→                }
   362→            }
   363→
   364→            # Ejecutar la herramienta
   365→            try:
   366→                tool_fn = TOOL_FUNCTIONS.get(tool_name)
   367→                if not tool_fn:
   368→                    self._log(f"Direct intent: herramienta no encontrada: {tool_name}", "warning")
   369→                    return
   370→
   371→                clean_params = {k: v for k, v in params.items() if v is not None and v != ""}
   372→                if not clean_params:
   373→                    self._log(f"Direct intent: parámetros vacíos para {tool_name}, dejando al LLM", "intent")
   374→                    return
   375→
   376→                tool_result = tool_fn(**clean_params)
   377→
   378→                # Evento: tool result
   379→                yield {
   380→                    "type": "tool_result",
   381→                    "data": {
   382→                        "tool": {"name": tool_name},
   383→                        "result": tool_result[:500] if tool_result else "(sin resultado)",
   384→                    }
   385→                }
   386→
   387→                self._log(f"Direct intent ejecutado exitosamente (stream): {tool_name}", "success")
   388→
   389→                # Evento: respuesta final
   390→                yield {
   391→                    "type": "text",
   392→                    "data": tool_result if tool_result else "(sin resultado)",
   393→                }
   394→
   395→                # Guardar interacción
   396→                self._save_interaction(user_message, tool_result or "(sin resultado)")
   397→
   398→                # Evento: done
   399→                yield {
   400→                    "type": "done",
   401→                    "data": tool_result if tool_result else "(sin resultado)",
   402→                    "thinking_log": self.thinking_log,
   403→                    "meta_status": self.metacognition.get_status() if hasattr(self, 'metacognition') else {},
   404→                }
   405→
   406→            except Exception as e:
   407→                self._log(f"Direct intent falló (stream): {tool_name} -> {e}", "error")
   408→                yield {
   409→                    "type": "error",
   410→                    "data": f"Error ejecutando {tool_name}: {e}",
   411→                }
   412→
   413→        return _stream_direct()

   423→        """v22: Determina el modelo optimo para la request actual via ModelRouter.

   425→        Si el router esta disponible y detecta que la tarea se beneficiaria

   460→                self._routing_info = None

   472→    # v21: CLASIFICACION DINAMICA DE COMPLEJIDAD

   474→    def _classify_task_complexity(self, user_message: str) -> str:
   475→        """
   476→        v21: Clasifica la complejidad de la tarea para ajustar el presupuesto
   477→        de iteraciones dinamicamente. Tareas simples no necesitan 15 iteraciones.
   478→        
   479→        Returns:
   480→            "simple" | "moderate" | "complex" | "full"
   481→        """
   482→        msg_lower = user_message.lower().strip()
   483→        
   484→        # 1. Tareas simples: saludos, confirmaciones, preguntas sobre el agente
   485→        for pattern in self._SIMPLE_TASK_PATTERNS:
   486→            if re.search(pattern, msg_lower, re.IGNORECASE):
   487→                self._log(f"Clasificacion de tarea: SIMPLE (patron: {pattern[:30]}...)", "planning")
   488→                return "simple"
   489→        
   490→        # 2. Tareas complejas: construir, crear app, deploy, multi-step
   491→        if self._is_complex_task(user_message):
   492→            self._log("Clasificacion de tarea: FULL (tarea compleja)", "planning")
   493→            return "full"
   494→        
   495→        # 3. Tareas que claramente necesitan herramientas pero no son complejas
   496→        if self._task_needs_tools(user_message):
   497→            # Verificar si es multi-step (mas de 2 patrones match)
   498→            pattern_count = sum(
   499→                1 for p in self._TOOL_NECESSARY_PATTERNS
   500→                if re.search(p, msg_lower, re.IGNORECASE)
   501→            )
   502→            if pattern_count >= 3:
   503→                self._log(f"Clasificacion de tarea: COMPLEX ({pattern_count} patrones)", "planning")
   504→                return "complex"
   505→            self._log(f"Clasificacion de tarea: MODERATE ({pattern_count} patrones)", "planning")
   506→            return "moderate"
   507→        
   508→        # 4. Default: moderado (preguntas que pueden necesitar busqueda)
   509→        self._log("Clasificacion de tarea: MODERATE (default)", "planning")
   510→        return "moderate"

   513→    # v21: TRUNCATION INTELIGENTE DE RESULTADOS
   514→    # ----------------------------------------------------------
   515→    def _truncate_tool_result(self, tool_name: str, result: str) -> str:
   516→        """
   517→        v21: Trunca el resultado de una herramienta de forma inteligente.
   518→        
   519→        Antes se truncaba todo a ~1500 chars, lo que cortaba resultados
   520→        criticos de SQL, archivos, etc. Ahora cada tipo de herramienta
   521→        tiene su limite apropiado.
   522→        
   523→        Intenta cortar en lineas completas para no romper JSON o codigo.
   524→        """
   525→        limit = self._TOOL_RESULT_LIMITS.get(tool_name, self._TOOL_RESULT_LIMITS["default"])
   526→        
   527→        if len(result) <= limit:
   528→            return result
   529→        
   530→        # Cortar en el limite, pero buscar un salto de linea cercano
   531→        truncated = result[:limit]
   532→        
   533→        # Buscar ultimo salto de linea para no cortar a medio codigo/JSON
   534→        last_newline = truncated.rfind('\n')
   535→        if last_newline > limit * 0.8:  # Solo si el salto esta cerca del final
   536→            truncated = truncated[:last_newline]
   537→        
   538→        truncated += f"\n\n... [resultado truncado: {len(result) - len(truncated)} chars omitidos. Total: {len(result)} chars]"
   539→        return truncated

  1118→    def _is_complex_task(self, message: str) -> bool:
  1119→        """Detecta si un mensaje parece una tarea compleja que se beneficia de planificacion.
  1120→        v21: Requiere al menos 2 keywords O que el mensaje empiece con un keyword de accion.
  1121→        Evita falsos positivos como 'listar archivos del proyecto'.
  1122→        """
  1123→        msg_lower = message.lower()
  1124→        matched = [kw for kw in self._COMPLEX_TASK_KEYWORDS if kw in msg_lower]
  1125→        
  1126→        # Si solo hay 1 keyword (ej: 'proyecto'), verificar que no sea
  1127→        # parte de una tarea simple como 'listar archivos del proyecto'
  1128→        if len(matched) == 1:
  1129→            # Keywords que por si solos NO indican tarea compleja
  1130→            weak_keywords = {"proyecto", "build"}
  1131→            if matched[0] in weak_keywords:
  1132→                return False
  1133→        
  1134→        # Si hay 2+ keywords, es complejo
  1135→        if len(matched) >= 2:
  1136→            return True
  1137→        
  1138→        # Si el mensaje empieza con un keyword de accion fuerte, es complejo
  1139→        strong_starts = ["construir", "crear app", "desarrollar", "automatizar",
  1140→                        "implementar", "hazme una app", "desarrolla"]
  1141→        return any(msg_lower.startswith(kw) for kw in strong_starts)

  1270→    def _execute_parallel_tasks(self, tasks, plan, results):
  1271→        """
  1272→        Ejecuta multiples tareas independientes en paralelo usando el Orchestrator.
  1273→
  1274→        Args:
  1275→            tasks: Lista de objetos Task con status=pending y dependencias resueltas
  1276→            plan: ExecutionPlan al que pertenecen las tareas
  1277→            results: Lista acumulada de resultados previos
  1278→
  1279→        Returns:
  1280→            Dict {task_id: result} con los resultados de cada tarea,
  1281→            o None si fallo y se debe hacer fallback a secuencial.
  1282→        """
  1283→        if not self.orchestrator or not ORCHESTRATOR_AVAILABLE:
  1284→            self._log("Orchestrator no disponible para ejecucion paralela", "warning")
  1285→            return None
  1286→
  1287→        # Crear descripciones de sub-tareas para el orchestrator
  1288→        subtask_descriptions = []
  1289→        for task in tasks:
  1290→            subtask_descriptions.append({
  1291→                "id": task.id,
  1292→                "description": f"{task.title}: {task.description}",
  1293→                "context": json.dumps(results[-3:], ensure_ascii=False) if results else "",
  1294→            })
  1295→
  1296→        self._log(
  1297→            f"Delegando {len(tasks)} tareas al Orchestrator en paralelo: "
  1298→            f"{[t.title for t in tasks]}",
  1299→            "info"
  1300→        )
  1301→
  1302→        # Usar orchestrator para ejecutar en paralelo
  1303→        try:
  1304→            orch_result = self.orchestrator.execute_parallel(
  1305→                descriptions=subtask_descriptions,
  1306→                agent_run_fn=self.run,  # Pasar el metodo run del propio agente
  1307→            )
  1308→            if orch_result:
  1309→                self._log(
  1310→                    f"Orchestrator completo {len(orch_result)}/{len(tasks)} tareas en paralelo",
  1311→                    "info"
  1312→                )
  1313→            return orch_result
  1314→        except Exception as e:
  1315→            self._log(f"Orchestrator parallel fallo: {e}, fallback a secuencial", "warning")
  1316→            return None

  1584→    def _stream_llm_with_tools(self, messages, selected_schemas=None, force_tool_choice=False, model_override=None):
  1585→        """
  1586→        Genera respuesta del LLM con streaming REAL.
  1587→        Yields: {"type": "token"|"tool_calls"|"done", "data": ...}
  1588→        
  1589→        v22: Agrega soporte para model_override (auto-routing).
  1590→        v20: Agrega soporte para selected_schemas y tool_choice forzado.
  1591→        """
  1592→        if selected_schemas is None:
  1593→            selected_schemas = TOOL_SCHEMAS
  1594→        
  1595→        import time as _time
  1596→        _llm_start = _time.monotonic()
  1597→        full_content = ""
  1598→        full_tool_calls = []
  1599→        
  1600→        # v20: Construir opciones para tool_choice
  1601→        options = {}
  1602→        if force_tool_choice:
  1603→            options["tool_choice"] = "required"
  1604→            self._log("Stream: forzando tool_choice=required", "info")
  1605→
  1606→        try:
  1607→            # Intentar streaming HTTP directo (mas rapido, sin overhead de lib)
  1608→            # v22: Pasar model_override si hay auto-routing
  1609→            for chunk in ollama.generate_stream(messages, tools=selected_schemas, model_override=model_override, options=options if options else None):
  1610→                if isinstance(chunk, str):
  1611→                    # Token de texto - emitir inmediatamente
  1612→                    # PERO: si parece JSON de tool call, no emitirlo al usuario
  1613→                    full_content += chunk
  1614→                    # Detectar si el contenido es JSON de herramienta
  1615→                    if self._looks_like_tool_json(full_content):
  1616→                        continue  # Acumular sin emitir
  1617→                    # Si antes se acumulaba JSON pero ya no, limpiar
  1618→                    yield {"type": "token", "data": chunk}
  1619→                elif isinstance(chunk, dict):
  1620→                    # Resultado final con tool_calls
  1621→                    msg = chunk.get("message", chunk)
  1622→                    tool_calls = msg.get("tool_calls", [])
  1623→                    content = msg.get("content", "")
  1624→
  1625→                    if content and not full_content:
  1626→                        # Contenido que no se streameo (fallback)
  1627→                        full_content = content
  1628→                        yield {"type": "token", "data": content}
  1629→
  1630→                    if tool_calls:
  1631→                        # Parsear tool calls
  1632→                        parsed_calls = []
  1633→                        for tc in tool_calls:
  1634→                            tool_name = tc.get("function", {}).get("name", "")
  1635→                            tool_params = tc.get("function", {}).get("arguments", {})
  1636→                            self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
  1637→                            parsed_calls.append({"name": tool_name, "params": tool_params})
  1638→                        yield {"type": "tool_calls", "data": parsed_calls}
  1639→                        yield {"type": "done", "data": False}
  1640→                        return
  1641→
  1642→            # Si llegamos aqui sin tool_calls, verificar si el contenido es JSON de herramientas
  1643→            if full_content:
  1644→                # Si el contenido acumulado era JSON de tool calls (no se emitio),
  1645→                # parsearlo y ejecutar como tool calls
  1646→                if self._looks_like_tool_json(full_content):
  1647→                    parsed = self._parse_json(full_content)
  1648→                    if parsed:
  1649→                        # Verificar si son multiples tool calls
  1650→                        if parsed.get("_multi_tool_calls") and parsed.get("tool_calls"):
  1651→                            yield {"type": "tool_calls", "data": parsed["tool_calls"]}
  1652→                            yield {"type": "done", "data": False}
  1653→                            return
  1654→                        accion = parsed.get("accion", "").strip()
  1655→                        if accion and accion in TOOL_FUNCTIONS:
  1656→                            yield {"type": "tool_calls", "data": [{"name": accion, "params": parsed.get("params", {})}]}
  1657→                            yield {"type": "done", "data": False}
  1658→                            return
  1659→                    # Si no se pudo parsear como tool call, emitir el texto limpio
  1660→                    clean = self._clean_json_leak(full_content)
  1661→                    if clean != full_content:
  1662→                        # El texto era JSON, emitir version limpia
  1663→                        yield {"type": "token", "data": clean}
  1664→                _llm_elapsed = (_time.monotonic() - _llm_start) * 1000.0
  1665→                get_metrics().record_llm_call(_llm_elapsed)
  1666→                yield {"type": "done", "data": True}
  1667→                return
  1668→
  1669→        except Exception as e:
  1670→            self._log(f"Error en streaming: {e}", "error")
  1671→            get_metrics().record_error("llm_stream")
  1672→
  1673→        # Fallback: modo no-streaming
  1674→        try:
  1675→            response = ollama.generate(messages, tools=selected_schemas, model_override=model_override, options=options if options else None)
  1676→            # generate() already records llm_call via @timed, so no duplicate
  1677→            if isinstance(response, str):
  1678→                yield {"type": "token", "data": response}
  1679→                yield {"type": "done", "data": True}
  1680→                return
  1681→
  1682→            message = response.get("message", response)
  1683→            tool_calls = message.get("tool_calls", [])
  1684→            content = message.get("content", "")
  1685→
  1686→            if tool_calls:
  1687→                parsed_calls = []
  1688→                for tc in tool_calls:
  1689→                    tool_name = tc.get("function", {}).get("name", "")
  1690→                    tool_params = tc.get("function", {}).get("arguments", {})
  1691→                    self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
  1692→                    parsed_calls.append({"name": tool_name, "params": tool_params})
  1693→                yield {"type": "tool_calls", "data": parsed_calls}
  1694→                yield {"type": "done", "data": False}
  1695→                return
  1696→
  1697→            if content:
  1698→                yield {"type": "token", "data": content}
  1699→                yield {"type": "done", "data": True}
  1700→                return
  1701→
  1702→        except Exception as e:
  1703→            self._log(f"Error en fallback: {e}", "error")
  1704→
  1705→        yield {"type": "done", "data": False}
  1706→

  1710→    def _react_with_tools(self, messages, iteration, selected_schemas=None, model_override=None):
  1711→        """ReAct usando function calling nativo. Soporta multiple tool calls.
  1712→        
  1713→        v22: Agrega soporte para model_override (auto-routing).
  1714→        v20: Agrega soporte para:
  1715→        - selected_schemas: solo enviar schemas relevantes al modelo
  1716→        - tool_choice="required" en primera iteracion cuando la tarea necesita herramientas
  1717→        """
  1718→        if selected_schemas is None:
  1719→            selected_schemas = TOOL_SCHEMAS
  1720→        
  1721→        # v20: Forzar uso de herramientas en la primera iteracion cuando
  1722→        # la tarea claramente las necesita. Esto elimina el 60% de los casos
  1723→        # donde el modelo intenta responder directamente.
  1724→        options = {}
  1725→        if iteration == 0 and hasattr(self, '_current_user_message') and self._task_needs_tools(self._current_user_message):
  1726→            options["tool_choice"] = "required"
  1727→            self._log("Forzando tool_choice=required (tarea necesita herramientas)", "info")
  1728→        
  1729→        try:
  1730→            response = ollama.generate(messages, tools=selected_schemas, model_override=model_override, options=options if options else None)
  1731→
  1732→            if isinstance(response, str):
  1733→                return ("respond", response)
  1734→
  1735→            message = response.get("message", response)
  1736→            tool_calls = message.get("tool_calls", [])
  1737→
  1738→            if tool_calls:
  1739→                parsed_calls = []
  1740→                for tc in tool_calls:
  1741→                    tool_name = tc.get("function", {}).get("name", "")
  1742→                    tool_params = tc.get("function", {}).get("arguments", {})
  1743→                    self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
  1744→                    parsed_calls.append({"name": tool_name, "params": tool_params})
  1745→                return ("tool_calls", parsed_calls)
  1746→
  1747→            content = message.get("content", "")
  1748→            if content:
  1749→                # SAFETY CHECK: Si el modelo genero JSON de tool call en vez de
  1750→                # usar function calling nativo, parsearlo y ejecutarlo
  1751→                parsed = self._parse_json(content)
  1752→                if parsed:
  1753→                    accion = parsed.get("accion", "").strip()
  1754→                    if accion and accion in TOOL_FUNCTIONS:
  1755→                        self._log(f"Tool call via JSON (modelo no uso function calling nativo): {accion}", "thinking")
  1756→                        params = parsed.get("params", {})
  1757→                        return ("tool_calls", [{"name": accion, "params": params}])
  1758→                    # Si tiene respuesta_final, usarla
  1759→                    respuesta_final = parsed.get("respuesta_final", "").strip()
  1760→                    if respuesta_final:
  1761→                        return ("respond", respuesta_final)
  1762→                return ("respond", content)
  1763→
  1764→            return ("error", "Respuesta vacia del modelo")
  1765→
  1766→        except Exception as e:
  1767→            self._log(f"Error en tool calling: {e}", "error")
  1768→            self.supports_tool_calling = False
  1769→            return ("error", str(e))
  1770→

  1774→    def _react_with_json(self, messages, iteration, model_override=None):
  1775→        """ReAct usando JSON parsing (fallback).
  1776→        v22: Agrega soporte para model_override (auto-routing).
  1777→        v21: Ahora filtra tools list en JSON prompt usando _selected_tool_names.
  1778→        """
  1779→        # generate_chat() already records llm_call via @timed decorator
  1780→        # v20 FIX: El JSON_TOOLS_PROMPT ahora se inyecta proactivamente desde _build_messages()
  1781→        # cuando supports_tool_calling=False. Este check reactivo solo actua como fallback
  1782→        # para el caso donde supports_tool_calling no se haya detectado aun (None).
  1783→        if not any("HERRAMIENTAS DISPONIBLES" in str(m.get("content", "")) for m in messages):
  1784→            system_msg_idx = next((i for i, m in enumerate(messages) if m["role"] == "system"), -1)
  1785→            if system_msg_idx >= 0:
  1786→                # v21: Usar prompt filtrado si tenemos seleccion de schemas
  1787→                use_compact = self._should_use_compact_prompt()
  1788→                if hasattr(self, '_selected_tool_names') and self._selected_tool_names:
  1789→                    tools_prompt = self._build_filtered_json_tools_prompt(use_compact)
  1790→                else:
  1791→                    tools_prompt = JSON_TOOLS_PROMPT_COMPACT if use_compact else JSON_TOOLS_PROMPT
  1792→                messages[system_msg_idx]["content"] += tools_prompt
  1793→
  1794→        try:
  1795→            # v22: Si hay model_override, usar generate() en vez de generate_chat()
  1796→            # para pasar el modelo routeado. generate_chat() no acepta model_override.
  1797→            if model_override:
  1798→                response = ollama.generate(messages, model_override=model_override)
  1799→            else:
  1800→                response = ollama.generate_chat(messages)
  1801→            if not response:
  1802→                return ("error", "El LLM no respondio")
  1803→
  1804→            parsed = self._parse_json(response)
  1805→            if not parsed:
  1806→                # No es JSON - probablemente respuesta directa del modelo
  1807→                # Limpiar posibles restos de formato JSON

  1875→    def _execute_tool_calls(self, tool_calls, messages):

  1884→            self._total_tool_calls += 1
  1885→

  1887→                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
  1888→                break

  1890→            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
  1891→                # v20: Herramientas en UNLIMITED_TOOLS no tienen limite de repeticion

  1895→
  1896→            filtered_calls.append(tc)
  1897→
  1898→        tool_calls = filtered_calls
  1899→
  1900→        # Validar parametros de todos los tool calls primero
  1901→        for tc in tool_calls:
  1902→            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
  1903→
  1904→        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
  1905→        if len(tool_calls) == 1:

  1907→
