   342тЖТ   342тЖТ        # Determinar nivel de profundidad

   409тЖТ   409тЖТ            f"Deep thinking completado: {result.thinking_tokens} tokens, "
   410тЖТ   410тЖТ            f"{len(result.plan)} pasos, profundidad={depth}, {result.duration_ms}ms"

   412тЖТ   412тЖТ
   413тЖТ   413тЖТ        # ---- Persistir pensamiento en memoria ----

   644тЖТ   644тЖТ        return []
   645тЖТ   645тЖТ
   646тЖТ   646тЖТ    def _parse_plan(self, plan_text):
   647тЖТ   647тЖТ        """Parsea texto de plan a lista de pasos, con fallback robusto."""
   648тЖТ   648тЖТ        if not plan_text:
   649тЖТ   649тЖТ            return []
   650тЖТ   650тЖТ
   651тЖТ   651тЖТ        # Buscar JSON en la respuesta (puede estar en code block)
   652тЖТ   652тЖТ        json_str = plan_text
   653тЖТ   653тЖТ        if "```json" in json_str:
   654тЖТ   654тЖТ            json_str = json_str.split("```json")[1].split("```")[0]
   655тЖТ   655тЖТ        elif "```" in json_str:
   656тЖТ   656тЖТ            json_str = json_str.split("```")[1].split("```")[0]
   657тЖТ   657тЖТ
   658тЖТ   658тЖТ        try:
   659тЖТ   659тЖТ            plan_data = json.loads(json_str.strip())
   660тЖТ   660тЖТ            steps = plan_data.get("steps", [])
   661тЖТ   661тЖТ            # Normalizar campos de cada paso
   662тЖТ   662тЖТ            for step in steps:
   663тЖТ   663тЖТ                if "confidence" not in step:
   664тЖТ   664тЖТ                    step["confidence"] = None
   665тЖТ   665тЖТ                if "purpose" not in step:
   666тЖТ   666тЖТ                    step["purpose"] = step.get("proposito", "auto")
   667тЖТ   667тЖТ            return steps[:5]
   668тЖТ   668тЖТ        except json.JSONDecodeError:
   669тЖТ   669тЖТ            # Fallback: parsear como lista numerada
   670тЖТ   670тЖТ            steps = []
   671тЖТ   671тЖТ            for line in plan_text.strip().split("\n"):
   672тЖТ   672тЖТ                line = line.strip()
   673тЖТ   673тЖТ                if line and (line[0].isdigit() or line.startswith("-")):
   674тЖТ   674тЖТ                    action = line.lstrip("0123456789.-) ").strip()
   675тЖТ   675тЖТ                    if action:
   676тЖТ   676тЖТ                        steps.append({
   677тЖТ   677тЖТ                            "action": action,
   678тЖТ   678тЖТ                            "tool": "infer",
   679тЖТ   679тЖТ                            "purpose": "auto",
   680тЖТ   680тЖТ                            "confidence": None,
   681тЖТ   681тЖТ                        })
   682тЖТ   682тЖТ            return steps[:5]
   683тЖТ   683тЖТ
   684тЖТ   684тЖТ    # ----------------------------------------------------------
   685тЖТ   685тЖТ    # ESTIMACION DE CONFIANZA BASADA EN CONTENIDO (Mejora 8)
   686тЖТ   686тЖТ    # ----------------------------------------------------------
   687тЖТ   687тЖТ
   688тЖТ   688тЖТ    def _estimate_plan_confidence(self, plan, user_message=""):
   689тЖТ   689тЖТ        """
   690тЖТ   690тЖТ        Estima la confianza en el plan generado basandose en contenido,
   691тЖТ   691тЖТ        no solo en numero de pasos.
   692тЖТ   692тЖТ
   693тЖТ   693тЖТ        Criterios:
   694тЖТ   694тЖТ        - Herramientas especificas mencionadas (+)
   695тЖТ   695тЖТ        - Propositos concretos (+)
   696тЖТ   696тЖТ        - Plan tiene fallback (+)
   697тЖТ   697тЖТ        - Pasos con confianza explicita (+)
   698тЖТ   698тЖТ        - Herramienta 'infer' o 'auto' (-)
   699тЖТ   699тЖТ        - Sin proposito (-)
   700тЖТ   700тЖТ        - Demasiados pasos (>5) (-)
   701тЖТ   701тЖТ        """
   702тЖТ   702тЖТ        if not plan:
   703тЖТ   703тЖТ            return 0.3
   704тЖТ   704тЖТ
   705тЖТ   705тЖТ        score = 0.5  # Base
   706тЖТ   706тЖТ        n_steps = len(plan)
   707тЖТ   707тЖТ
   708тЖТ   708тЖТ        # Bonus por herramientas especificas mencionadas
   709тЖТ   709тЖТ        specific_tools = sum(
   710тЖТ   710тЖТ            1 for s in plan
   711тЖТ   711тЖТ            if s.get("tool", "infer") not in ("infer", "auto", "responder", "")
   712тЖТ   712тЖТ        )
   713тЖТ   713тЖТ        if specific_tools >= 2:
   714тЖТ   714тЖТ            score += 0.15
   715тЖТ   715тЖТ        elif specific_tools >= 1:
   716тЖТ   716тЖТ            score += 0.08
   717тЖТ   717тЖТ
   718тЖТ   718тЖТ        # Bonus por propositos concretos (no 'auto')
   719тЖТ   719тЖТ        concrete_purposes = sum(
   720тЖТ   720тЖТ            1 for s in plan
   721тЖТ   721тЖТ            if s.get("purpose", "auto") not in ("auto", "") and len(s.get("purpose", "")) > 10
   722тЖТ   722тЖТ        )
   723тЖТ   723тЖТ        if concrete_purposes >= 2:
   724тЖТ   724тЖТ            score += 0.1
   725тЖТ   725тЖТ
   726тЖТ   726тЖТ        # Bonus por confianza explicita en pasos
   727тЖТ   727тЖТ        with_confidence = sum(1 for s in plan if s.get("confidence") is not None)
   728тЖТ   728тЖТ        if with_confidence >= n_steps * 0.5:
   729тЖТ   729тЖТ            avg_step_conf = sum(
   730тЖТ   730тЖТ                s["confidence"] for s in plan if s.get("confidence") is not None
   731тЖТ   731тЖТ            ) / max(with_confidence, 1)
   732тЖТ   732тЖТ            score += 0.1 * avg_step_conf
   733тЖТ   733тЖТ
   734тЖТ   734тЖТ        # Penalizacion por herramientas inferidas
   735тЖТ   735тЖТ        inferred_tools = sum(1 for s in plan if s.get("tool", "") == "infer")
   736тЖТ   736тЖТ        if inferred_tools > n_steps * 0.5:
   737тЖТ   737тЖТ            score -= 0.1
   738тЖТ   738тЖТ
   739тЖТ   739тЖТ        # Penalizacion por demasiados pasos
   740тЖТ   740тЖТ        if n_steps > 5:
   741тЖТ   741тЖТ            score -= 0.15
   742тЖТ   742тЖТ        elif n_steps == 1:
   743тЖТ   743тЖТ            score -= 0.05
   744тЖТ   744тЖТ
   745тЖТ   745тЖТ        return max(0.1, min(1.0, score))
   746тЖТ   746тЖТ
   747тЖТ   747тЖТ    # ----------------------------------------------------------
   748тЖТ   748тЖТ    # POST-REFLEXION (Mejora 6: con contexto de tools)

   750тЖТ   750тЖТ
   751тЖТ   751тЖТ    def reflect(self, user_message, response, had_errors=False, tool_results=None):
   752тЖТ   752тЖТ        """
   753тЖТ   753тЖТ        Post-reflexion: evalua y potencialmente mejora la respuesta final.
