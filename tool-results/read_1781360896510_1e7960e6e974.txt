   181тЖТ   181тЖТ    def _compute_decay(self, created_at, now=None):
   182тЖТ   182тЖТ        """
   183тЖТ   183тЖТ        Computa el factor de decaimiento temporal.
   184тЖТ   184тЖТ        Los recuerdos recientes pesan mas que los viejos.
   185тЖТ   185тЖТ        Retorna un valor entre 0 y 1.
   186тЖТ   186тЖТ        """
   187тЖТ   187тЖТ        if not created_at:
   188тЖТ   188тЖТ            return 0.5
   189тЖТ   189тЖТ        try:
   190тЖТ   190тЖТ            created = datetime.fromisoformat(created_at)
   191тЖТ   191тЖТ            if now is None:
   192тЖТ   192тЖТ                now = datetime.now()
   193тЖТ   193тЖТ            days_old = (now - created).total_seconds() / 86400
   194тЖТ   194тЖТ            # Decaimiento exponencial
   195тЖТ   195тЖТ            import math
   196тЖТ   196тЖТ            decay = math.exp(-0.693 * days_old / self.DECAY_HALF_LIFE_DAYS)
   197тЖТ   197тЖТ            return max(decay, 0.1)  # Minimo 10% de relevancia
   198тЖТ   198тЖТ        except Exception as e:
   199тЖТ   199тЖТ            logger.debug(f"Error computando decaimiento: {e}")
   200тЖТ   200тЖТ            return 0.5

   202тЖТ   202тЖТ    def _is_duplicate(self, text, threshold=0.95):
   203тЖТ   203тЖТ        """
   204тЖТ   204тЖТ        Verifica si un texto es duplicado semantico de uno existente.
   205тЖТ   205тЖТ        Usa embeddings para comparar.
   206тЖТ   206тЖТ        v3: Catch de errores de dimension para no crashear.
   207тЖТ   207тЖТ        """
   208тЖТ   208тЖТ        embedding = ollama.get_embedding(text)
   209тЖТ   209тЖТ        if not embedding:
   210тЖТ   210тЖТ            return False
   211тЖТ   211тЖТ
   212тЖТ   212тЖТ        # Verificar dimension antes de consultar
   213тЖТ   213тЖТ        if self._embedding_dim is not None and len(embedding) != self._embedding_dim:
   214тЖТ   214тЖТ            logger.debug(f"Dimension mismatch en _is_duplicate, saltando verificacion")
   215тЖТ   215тЖТ            return False
   216тЖТ   216тЖТ
   217тЖТ   217тЖТ        try:
   218тЖТ   218тЖТ            results = self._collection.query(
   219тЖТ   219тЖТ                query_embeddings=[embedding],
   220тЖТ   220тЖТ                n_results=1,
   221тЖТ   221тЖТ                include=["distances"]
   222тЖТ   222тЖТ            )
   223тЖТ   223тЖТ            if results and results["distances"] and results["distances"][0]:
   224тЖТ   224тЖТ                # ChromaDB con cosine space: distance < (1 - threshold) = duplicado
   225тЖТ   225тЖТ                distance = results["distances"][0][0]
   226тЖТ   226тЖТ                similarity = 1 - distance
   227тЖТ   227тЖТ                return similarity >= threshold
   228тЖТ   228тЖТ        except Exception as e:
   229тЖТ   229тЖТ            err_msg = str(e).lower()
   230тЖТ   230тЖТ            if "dimension" in err_msg or "dim" in err_msg:
   231тЖТ   231тЖТ                logger.warning(f"Dimension mismatch en _is_duplicate, recreando coleccion...")
   232тЖТ   232тЖТ                try:
   233тЖТ   233тЖТ                    self._handle_dimension_error(embedding)
   234тЖТ   234тЖТ                except Exception:
   235тЖТ   235тЖТ                    pass
   236тЖТ   236тЖТ            else:
   237тЖТ   237тЖТ                logger.debug(f"Error verificando duplicado semantico: {e}")
   238тЖТ   238тЖТ        return False

   264тЖТ   264тЖТ                    return entry_id
   265тЖТ   265тЖТ            except Exception as e:

   269тЖТ   269тЖТ

   273тЖТ   273тЖТ        meta["text_preview"] = text[:100]
   274тЖТ   274тЖТ
   275тЖТ   275тЖТ        # Si skip_embedding, guardar solo con metadatos (sin calculo de embedding)

   280тЖТ   280тЖТ                    ids=[entry_id],

   283тЖТ   283тЖТ                )
   284тЖТ   284тЖТ            except Exception as e:
   285тЖТ   285тЖТ                logger.debug(f"Error agregando sin embedding a ChromaDB: {e}")
   286тЖТ   286тЖТ            return entry_id
   287тЖТ   287тЖТ
   288тЖТ   288тЖТ        # Obtener embedding
   289тЖТ   289тЖТ        embedding = ollama.get_embedding(text)
   290тЖТ   290тЖТ
   291тЖТ   291тЖТ        if embedding:
   292тЖТ   292тЖТ            # Actualizar dimension si es la primera vez que obtenemos un embedding valido

   294тЖТ   294тЖТ                self._embedding_dim = len(embedding)
   295тЖТ   295тЖТ                logger.info(f"Dimension de embedding detectada dinamicamente: {self._embedding_dim}")

   300тЖТ   300тЖТ                    f"Embedding con dimension incorrecta ({len(embedding)} vs {self._embedding_dim}). "
   301тЖТ   301тЖТ                    f"Actualizando dimension y recreando coleccion..."
   302тЖТ   302тЖТ                )
   303тЖТ   303тЖТ                self._handle_dimension_error(embedding)
   304тЖТ   304тЖТ
   305тЖТ   305тЖТ            # Intentar insertar con catch-all de errores de dimension
   306тЖТ   306тЖТ            for attempt in range(self.MAX_RECREATE_RETRIES + 1):
   307тЖТ   307тЖТ                try:
   308тЖТ   308тЖТ                    self._collection.add(
   309тЖТ   309тЖТ                        ids=[entry_id],
   310тЖТ   310тЖТ                        embeddings=[embedding],
   311тЖТ   311тЖТ                        documents=[text[:500]],
   312тЖТ   312тЖТ                        metadatas=[meta]
   313тЖТ   313тЖТ                    )
   314тЖТ   314тЖТ                    return entry_id
   315тЖТ   315тЖТ                except Exception as e:
   316тЖТ   316тЖТ                    err_msg = str(e).lower()
   317тЖТ   317тЖТ                    if "dimension" in err_msg or "dim" in err_msg:
   318тЖТ   318тЖТ                        if attempt < self.MAX_RECREATE_RETRIES:
   319тЖТ   319тЖТ                            logger.warning(
   320тЖТ   320тЖТ                                f"Error de dimension al insertar (intento {attempt + 1}). "
   321тЖТ   321тЖТ                                f"Recreando coleccion..."
   322тЖТ   322тЖТ                            )
   323тЖТ   323тЖТ                            self._handle_dimension_error(embedding)
   324тЖТ   324тЖТ                        else:
   325тЖТ   325тЖТ                            logger.error(
   326тЖТ   326тЖТ                                f"No se pudo resolver error de dimension tras {attempt + 1} intentos. "
   327тЖТ   327тЖТ                                f"Guardando sin embedding."
   328тЖТ   328тЖТ                            )
   329тЖТ   329тЖТ                            # Fallback: guardar sin embedding para no perder el dato
   330тЖТ   330тЖТ                            meta["no_embedding"] = True
   331тЖТ   331тЖТ                            try:
   332тЖТ   332тЖТ                                self._collection.add(
   333тЖТ   333тЖТ                                    ids=[entry_id],
   334тЖТ   334тЖТ                                    documents=[text[:500]],
   335тЖТ   335тЖТ                                    metadatas=[meta]
   336тЖТ   336тЖТ                                )
   337тЖТ   337тЖТ                            except Exception:
   338тЖТ   338тЖТ                                pass

   400тЖТ   400тЖТ        # Actualizar dimension si es la primera vez que obtenemos un embedding valido

   402тЖТ   402тЖТ            self._embedding_dim = len(query_embedding)

   410тЖТ   410тЖТ            )

   414тЖТ   414тЖТ                logger.warning(f"Error recreando coleccion: {e}")

   416тЖТ   416тЖТ
   417тЖТ   417тЖТ        try:
   418тЖТ   418тЖТ            # Buscar mas candidatos de los necesarios para re-rankear con decaimiento
   419тЖТ   419тЖТ            n_candidates = min(limit * 3, 20)
   420тЖТ   420тЖТ            results = self._collection.query(
   421тЖТ   421тЖТ                query_embeddings=[query_embedding],
   422тЖТ   422тЖТ                n_results=n_candidates,
   423тЖТ   423тЖТ                include=["documents", "metadatas", "distances"]
   424тЖТ   424тЖТ            )
   425тЖТ   425тЖТ
   426тЖТ   426тЖТ            if not results or not results["ids"] or not results["ids"][0]:
   427тЖТ   427тЖТ                return self._text_search(query, limit)
   428тЖТ   428тЖТ
   429тЖТ   429тЖТ            scored = []
   430тЖТ   430тЖТ            now = datetime.now()
   431тЖТ   431тЖТ            for i, doc_id in enumerate(results["ids"][0]):
   432тЖТ   432тЖТ                distance = results["distances"][0][i]

   434тЖТ   434тЖТ
   435тЖТ   435тЖТ                # Aplicar decaimiento temporal
   436тЖТ   436тЖТ                meta = results["metadatas"][0][i] if results["metadatas"] else {}
   437тЖТ   437тЖТ                created_at = meta.get("created", "")
   438тЖТ   438тЖТ                decay = self._compute_decay(created_at, now)
   439тЖТ   439тЖТ
   440тЖТ   440тЖТ                # Score final = similitud * decaimiento
   441тЖТ   441тЖТ                final_score = similarity * decay
   442тЖТ   442тЖТ
   443тЖТ   443тЖТ                if final_score >= min_similarity * 0.5:  # Umbral mas bajo tras decaimiento
   444тЖТ   444тЖТ                    scored.append({
   445тЖТ   445тЖТ                        "id": doc_id,
   446тЖТ   446тЖТ                        "text": results["documents"][0][i] if results["documents"] else "",
   447тЖТ   447тЖТ                        "metadata": meta,
   448тЖТ   448тЖТ                        "has_vector": True,
   449тЖТ   449тЖТ                        "score": round(final_score, 3),
   450тЖТ   450тЖТ                        "raw_similarity": round(similarity, 3),
   451тЖТ   451тЖТ                        "decay": round(decay, 3),
   452тЖТ   452тЖТ                    })
   453тЖТ   453тЖТ
   454тЖТ   454тЖТ            scored.sort(key=lambda x: x["score"], reverse=True)
   455тЖТ   455тЖТ            return scored[:limit]
   456тЖТ   456тЖТ
   457тЖТ   457тЖТ        except Exception as e:
   458тЖТ   458тЖТ            err_msg = str(e).lower()
   459тЖТ   459тЖТ            if "dimension" in err_msg or "dim" in err_msg:
   460тЖТ   460тЖТ                logger.warning(f"Dimension mismatch en query, recreando coleccion...")
   461тЖТ   461тЖТ                try:
   462тЖТ   462тЖТ                    self._handle_dimension_error(query_embedding)
   463тЖТ   463тЖТ                except Exception:
   464тЖТ   464тЖТ                    pass
   465тЖТ   465тЖТ            else:
   466тЖТ   466тЖТ                logger.warning(f"Error en busqueda ChromaDB: {e}")
   467тЖТ   467тЖТ            return self._text_search(query, limit)

   469тЖТ   469тЖТ    def _text_search(self, query, limit=5):
   470тЖТ   470тЖТ        """Busqueda por texto cuando no hay embeddings."""
   471тЖТ   471тЖТ        query_lower = query.lower()
   472тЖТ   472тЖТ        query_words = [w for w in query_lower.split() if len(w) > 3]
   473тЖТ   473тЖТ
   474тЖТ   474тЖТ        if not query_words:
   475тЖТ   475тЖТ            return []
   476тЖТ   476тЖТ
   477тЖТ   477тЖТ        try:

   481тЖТ   481тЖТ

   483тЖТ   483тЖТ                doc_lower = doc.lower()

   485тЖТ   485тЖТ                if matches > 0:

   487тЖТ   487тЖТ                    meta = all_docs["metadatas"][i] if all_docs["metadatas"] else {}

   539тЖТ   539тЖТclass SimpleVectorStore(VectorStore):
   540тЖТ   540тЖТ    """
   541тЖТ   541тЖТ    Vector store casero (fallback cuando ChromaDB no esta instalado).
   542тЖТ   542тЖТ    Hereda de VectorStore y agrega decaimiento temporal + deduplicacion por texto.
   543тЖТ   543тЖТ    Solo sobreescribe los metodos que difieren del base.
   544тЖТ   544тЖТ    Elimina ~150 lineas de codigo duplicado vs la version anterior.
   545тЖТ   545тЖТ    """
   546тЖТ   546тЖТ
   547тЖТ   547тЖТ    DECAY_HALF_LIFE_DAYS = 30
   548тЖТ   548тЖТ
   549тЖТ   549тЖТ    def _compute_decay(self, created_at):
   550тЖТ   550тЖТ        """Computa el factor de decaimiento temporal (0.1 - 1.0)."""
   551тЖТ   551тЖТ        if not created_at:
   552тЖТ   552тЖТ            return 0.5
   553тЖТ   553тЖТ        try:
   554тЖТ   554тЖТ            created = datetime.fromisoformat(created_at)
   555тЖТ   555тЖТ            days_old = (datetime.now() - created).total_seconds() / 86400
   556тЖТ   556тЖТ            import math
   557тЖТ   557тЖТ            decay = math.exp(-0.693 * days_old / self.DECAY_HALF_LIFE_DAYS)
   558тЖТ   558тЖТ            return max(decay, 0.1)
   559тЖТ   559тЖТ        except Exception as e:
   560тЖТ   560тЖТ            logger.debug(f"Error computando decaimiento: {e}")
   561тЖТ   561тЖТ            return 0.5

   565тЖТ   565тЖТ        if not entry_id:
   566тЖТ   566тЖТ            entry_id = hashlib.md5(text.encode()).hexdigest()[:12]
   567тЖТ   567тЖТ
   568тЖТ   568тЖТ        # Verificar si ya existe

   571тЖТ   571тЖТ                return entry_id
   572тЖТ   572тЖТ
   573тЖТ   573тЖТ        # Deduplicacion basica por texto similar (100 chars lowercase)

   575тЖТ   575тЖТ        for entry in self.index:
   576тЖТ   576тЖТ            if entry["text"][:100].lower() == text_lower:
   577тЖТ   577тЖТ                return entry["id"]
   578тЖТ   578тЖТ

   582тЖТ   582тЖТ    def search(self, query, limit=5, min_similarity=0.3):
   583тЖТ   583тЖТ        """Busca con decaimiento temporal. Sobreescribe para agregar decay al scoring."""
   584тЖТ   584тЖТ        if not self.index:
   585тЖТ   585тЖТ            return []
   586тЖТ   586тЖТ

   608тЖТ   608тЖТ
   609тЖТ   609тЖТ            if final_score >= min_similarity * 0.5:
   610тЖТ   610тЖТ                scored.append({
   611тЖТ   611тЖТ                    **entry,
   612тЖТ   612тЖТ                    "score": round(final_score, 3),
   613тЖТ   613тЖТ                    "raw_similarity": round(raw_sim, 3),
   614тЖТ   614тЖТ                    "decay": round(decay, 3),
   615тЖТ   615тЖТ                })
   616тЖТ   616тЖТ
   617тЖТ   617тЖТ        scored.sort(key=lambda x: x["score"], reverse=True)
