    63→function formatBytes(bytes: number): string {
    64→  if (bytes === 0) return "0 B";
    65→  const k = 1024;
    66→  const sizes = ["B", "KB", "MB", "GB", "TB"];
    67→  const i = Math.floor(Math.log(bytes) / Math.log(k));
    68→  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + " " + sizes[i];
    69→}
    70→
    71→function formatUptime(seconds: number): string {
    72→  const h = Math.floor(seconds / 3600);
    73→  const m = Math.floor((seconds % 3600) / 60);
    74→  const s = Math.floor(seconds % 60);
    75→  return `${h.toString().padStart(2, "0")}:${m.toString().padStart(2, "0")}:${s.toString().padStart(2, "0")}`;
    76→}

   172→// ─── Chat Message Component ──────────────────────────────────────────────────
   173→
   174→function ChatMessage({
   175→  msg,
   176→  isLastAssistant,
   177→  isLoading,
   178→  expandedThinking,
   179→  toggleThinking,
   180→}: {
   181→  msg: Message;
   182→  isLastAssistant: boolean;
   183→  isLoading: boolean;
   184→  expandedThinking: Record<string, boolean>;
   185→  toggleThinking: (id: string) => void;
   186→}) {
   187→  if (msg.role === "user") {
   188→    return (
   189→      <div className="flex justify-end animate-fade-in">
   190→        <div className="max-w-[70%] px-3 py-2 border border-[rgba(255,255,255,0.06)] text-[13px] leading-[1.7] text-[#e0e0e0]">
   191→          {msg.content}
   192→        </div>
   193→      </div>
   194→    );
   195→  }
   196→
   197→  const isStreaming = isLastAssistant && isLoading;
   198→  const cleanedThinking = msg.thinking ? cleanThinking(msg.thinking) : "";
   199→
   200→  return (
   201→    <div className="animate-fade-in">
   202→      <div className="max-w-[85%]">
   203→        <div className="border-l border-[rgba(0,212,255,0.2)] pl-3">
   204→          {/* Thinking section */}
   205→          {cleanedThinking && (
   206→            <div className="mb-2.5">
   207→              <button
   208→                onClick={() => toggleThinking(msg.id)}
   209→                className="flex items-center gap-1 text-[9px] text-[#555555] hover:text-[#888888] transition-colors duration-150"
   210→              >
   211→                {expandedThinking[msg.id] ? (
   212→                  <ChevronDown size={10} />
   213→                ) : (
   214→                  <ChevronRight size={10} />
   215→                )}
   216→                <span>thinking</span>
   217→                <span className="text-[#333333] ml-1">
   218→                  ({cleanedThinking.length} chars)
   219→                </span>
   220→              </button>
   221→              {expandedThinking[msg.id] && (
   222→                <div className="mt-1.5 text-[10px] leading-[1.6] text-[#3a3a3a] max-h-40 overflow-y-auto pr-2 whitespace-pre-wrap border-l border-[rgba(255,255,255,0.04)] pl-2">
   223→                  {cleanedThinking}
   224→                </div>
   225→              )}
   226→            </div>
   227→          )}
   228→
   229→          {/* Tool calls */}
   230→          {msg.toolCalls && msg.toolCalls.length > 0 && (
   231→            <div className="flex flex-wrap gap-1.5 mb-2">
   232→              {msg.toolCalls.map((tool, i) => (
   233→                <span
   234→                  key={i}
   235→                  className="text-[9px] px-1.5 py-0.5 border border-[rgba(0,212,255,0.15)] text-[rgba(0,212,255,0.5)]"
   236→                >
   237→                  [{tool}]
   238→                </span>
   239→              ))}
   240→            </div>
   241→          )}
   242→
   243→          {/* Content */}
   244→          <div className="text-[13px] leading-[1.7] text-[#e0e0e0] whitespace-pre-wrap">
   245→            {msg.content}
   246→            {/* Streaming cursor */}
   247→            {isStreaming && !msg.content && (
   248→              <span className="inline-block w-[6px] h-[14px] bg-[rgba(0,212,255,0.5)] cursor-blink ml-0.5" />
   249→            )}
   250→            {isStreaming && msg.content && (
   251→              <span className="inline-block w-[5px] h-[13px] bg-[rgba(0,212,255,0.5)] cursor-blink ml-0.5 align-middle" />
   252→            )}
   253→          </div>
   254→
   255→          {/* Response metadata */}
   256→          {msg.responseTime && !isStreaming && (
   257→            <div className="mt-2 text-[9px] text-[#333333] flex items-center gap-2">
   258→              <span>{msg.responseTime}ms</span>
   259→              {msg.tokenCount ? (
   260→                <>
   261→                  <span className="text-[#222222]">·</span>
   262→                  <span>~{msg.tokenCount} tokens</span>
   263→                </>
   264→              ) : null}
   265→            </div>
   266→          )}
   267→        </div>
   268→      </div>
   269→    </div>
   270→  );
   271→}

   275→export default function ZAIInterface() {
   276→  // State
   277→  const [messages, setMessages] = useState<Message[]>([]);
   278→  const [input, setInput] = useState("");
   279→  const [isLoading, setIsLoading] = useState(false);
   280→  const [selectedModel, setSelectedModel] = useState<string>("qwen3:4b");
   281→  const [status, setStatus] = useState<SystemStatus>({
   282→    connected: false,
   283→    models: [],
   284→    modelCount: 0,
   285→  });
   286→  const [uptime, setUptime] = useState(0);
   287→  const [sessionStats, setSessionStats] = useState<SessionStats>({
   288→    messageCount: 0,
   289→    toolsUsed: 0,
   290→    avgResponseTime: 0,
   291→    totalTokens: 0,
   292→  });
   293→  const [sidebarOpen, setSidebarOpen] = useState(true);
   294→  const [expandedThinking, setExpandedThinking] = useState<Record<string, boolean>>({});
   295→  const [isMobile, setIsMobile] = useState(false);
   296→  const [useAgent, setUseAgent] = useState(true);
   297→
   298→  // Refs
   299→  const chatEndRef = useRef<HTMLDivElement>(null);
   300→  const inputRef = useRef<HTMLTextAreaElement>(null);
   301→  const responseTimesRef = useRef<number[]>([]);
   302→
   303→  // ─── Detect Mobile ──────────────────────────────────────────────────────
   304→
   305→  useEffect(() => {
   306→    const checkMobile = () => setIsMobile(window.innerWidth < 768);
   307→    checkMobile();
   308→    window.addEventListener("resize", checkMobile);
   309→    return () => window.removeEventListener("resize", checkMobile);
   310→  }, []);
   311→
   312→  // Close sidebar by default on mobile
   313→  useEffect(() => {
   314→    if (isMobile) setSidebarOpen(false);
   315→  }, [isMobile]);
   316→
   317→  // ─── Fetch Status ────────────────────────────────────────────────────────
   318→
   319→  const fetchStatus = useCallback(async () => {
   320→    try {
   321→      const res = await fetch("/api/status");
   322→      const data = await res.json();
   323→      setStatus(data);
   324→      if (data.connected && data.models.length > 0) {
   325→        setSelectedModel((prev) => {
   326→          if (data.models.find((m: OllamaModel) => m.name === prev)) return prev;
   327→          return data.models[0].name;
   328→        });
   329→      }
   330→    } catch {
   331→      setStatus({ connected: false, agentAvailable: false, models: [], modelCount: 0, error: "Failed to fetch" });
   332→    }
   333→  }, []);
   334→
   335→  // ─── Effects ─────────────────────────────────────────────────────────────
   336→
   337→  useEffect(() => {
   338→    fetchStatus();
   339→    const interval = setInterval(fetchStatus, 10000);
   340→    return () => clearInterval(interval);
   341→  }, [fetchStatus]);
   342→
   343→  useEffect(() => {
   344→    const timer = setInterval(() => setUptime((prev) => prev + 1), 1000);
   345→    return () => clearInterval(timer);
   346→  }, []);
   347→
   348→  useEffect(() => {
   349→    chatEndRef.current?.scrollIntoView({ behavior: "smooth" });
   350→  }, [messages]);
   351→
   352→  // ─── Auto-select first model ────────────────────────────────────────────
   353→
   354→  const lastAssistantId = useMemo(() => {
   355→    for (let i = messages.length - 1; i >= 0; i--) {
   356→      if (messages[i].role === "assistant") return messages[i].id;
   357→    }
   358→    return null;
   359→  }, [messages]);
   360→
   361→  // ─── Send Message ───────────────────────────────────────────────────────
   362→
   363→  const sendMessage = async () => {
   364→    if (!input.trim() || isLoading) return;
   365→
   366→    const userMessage: Message = {
   367→      id: generateId(),
   368→      role: "user",
   369→      content: input.trim(),
   370→      timestamp: Date.now(),
   371→    };
   372→
   373→    const assistantId = generateId();
   374→    const assistantMessage: Message = {
   375→      id: assistantId,
   376→      role: "assistant",
   377→      content: "",
   378→      timestamp: Date.now(),
   379→      isStreaming: true,
   380→    };
   381→
   382→    setMessages((prev) => [...prev, userMessage, assistantMessage]);
   383→    setInput("");
   384→    setIsLoading(true);
   385→
   386→    // Reset textarea height
   387→    if (inputRef.current) {
   388→      inputRef.current.style.height = "auto";
   389→    }
   390→
   391→    const startTime = Date.now();
   392→
   393→    const ollamaMessages = [...messages, userMessage].map((m) => ({
   394→      role: m.role,
   395→      content: m.content,
   396→    }));
   397→
   398→    try {
   399→      const res = await fetch("/api/chat", {
   400→        method: "POST",
   401→        headers: { "Content-Type": "application/json" },
   402→        body: JSON.stringify({ messages: ollamaMessages, model: selectedModel, useAgent }),
   403→      });
   404→
   405→      if (!res.ok) {
   406→        const errorData = await res.json().catch(() => ({ error: "Unknown error" }));
   407→        setMessages((prev) =>
   408→          prev.map((m) =>
   409→            m.id === assistantId
   410→              ? { ...m, content: `Error: ${errorData.error || "Unknown error"}`, isStreaming: false, responseTime: Date.now() - startTime }
   411→              : m
   412→          )
   413→        );
   414→        setIsLoading(false);
   415→        return;
   416→      }
   417→
   418→      const reader = res.body?.getReader();
   419→      if (!reader) {
   420→        setIsLoading(false);
   421→        return;
   422→      }
   423→
   424→      const decoder = new TextDecoder();
   425→      let fullContent = "";
   426→      let thinkingContent = "";
   427→      let isThinking = false;
   428→      let toolCalls: string[] = [];
   429→      let tokenCount = 0;
   430→
   431→      while (true) {
   432→        const { done, value } = await reader.read();
   433→        if (done) break;
   434→
   435→        const chunk = decoder.decode(value, { stream: true });
   436→        const lines = chunk.split("\n").filter((line) => line.startsWith("data: "));
   437→
   438→        for (const line of lines) {
   439→          const data = line.slice(6);
   440→          if (data === "[DONE]") continue;
   441→
   442→          try {
   443→            const parsed = JSON.parse(data);
   444→            if (parsed.error) {
   445→              fullContent += `\nError: ${parsed.error}`;
   446→              break;
   447→            }
   448→
   449→            // Bridge agent events (type field present)
   450→            if (parsed.type) {
   451→              if (parsed.type === "text") {
   452→                const content = parsed.data || "";
   453→                if (content.includes("<think")) {
   454→                  isThinking = true;
   455→                }
   456→                if (isThinking) {
   457→                  thinkingContent += content;
   458→                  if (content.includes("</think")) {
   459→                    isThinking = false;
   460→                  }
   461→                } else {
   462→                  fullContent += content;
   463→                }
   464→                tokenCount++;
   465→              } else if (parsed.type === "tool_start") {
   466→                const toolName = parsed.data?.name || "unknown";
   467→                toolCalls = [...new Set([...toolCalls, toolName])];
   468→              } else if (parsed.type === "tool_result") {
   469→                // Tool completed - already tracked in toolCalls
   470→              } else if (parsed.type === "meta") {
   471→                // Metacognition event - we could display this
   472→              } else if (parsed.type === "done") {
   473→                // Agent finished
   474→              } else if (parsed.type === "error") {
   475→                fullContent += `\nError: ${parsed.data}`;
   476→              }
   477→            } else {
   478→              // Direct Ollama format (no type field)
   479→              const content = parsed.message?.content || "";
   480→              tokenCount++;
   481→
   482→              if (content.includes("<think")) {
   483→                isThinking = true;
   484→              }
   485→
   486→              if (isThinking) {
   487→                thinkingContent += content;
   488→                if (content.includes("</think")) {
   489→                  isThinking = false;
   490→                }
   491→              } else {
   492→                fullContent += content;
   493→              }
   494→
   495→              // Detect tool calls in text
   496→              const toolMatches = content.match(/\[([a-z_]+)\]/g);
   497→              if (toolMatches) {
   498→                toolCalls = [...new Set([...toolCalls, ...toolMatches.map((t) => t.slice(1, -1))])];
   499→              }
   500→            }
   501→          } catch {
   502→            // Skip malformed JSON
   503→          }
   504→        }
   505→
   506→        setMessages((prev) =>
   507→          prev.map((m) =>
   508→            m.id === assistantId
   509→              ? {
   510→                  ...m,
   511→                  content: fullContent,
   512→                  thinking: thinkingContent || undefined,
   513→                  toolCalls: toolCalls.length > 0 ? toolCalls : undefined,
   514→                  tokenCount,
   515→                }
   516→              : m
   517→          )
   518→        );
   519→      }
   520→
   521→      const responseTime = Date.now() - startTime;
   522→      responseTimesRef.current.push(responseTime);
   523→
   524→      setMessages((prev) =>
   525→        prev.map((m) =>
   526→          m.id === assistantId ? { ...m, responseTime, isStreaming: false } : m
   527→        )
   528→      );
   529→
   530→      setSessionStats((prev) => ({
   531→        messageCount: prev.messageCount + 1,
   532→        toolsUsed: prev.toolsUsed + toolCalls.length,
   533→        avgResponseTime:
   534→          responseTimesRef.current.length > 0
   535→            ? Math.round(responseTimesRef.current.reduce((a, b) => a + b, 0) / responseTimesRef.current.length)
   536→            : 0,
   537→        totalTokens: prev.totalTokens + tokenCount,
   538→      }));
   539→    } catch (error) {
   540→      const errMsg = error instanceof Error ? error.message : "Unknown error";
   541→      setMessages((prev) =>
   542→        prev.map((m) =>
   543→          m.id === assistantId
   544→            ? { ...m, content: `Connection error: ${errMsg}`, isStreaming: false, responseTime: Date.now() - startTime }
   545→            : m
   546→        )
   547→      );
   548→    } finally {
   549→      setIsLoading(false);
   550→      inputRef.current?.focus();
   551→    }
   552→  };
   553→
   554→  // ─── Handle Key Press ───────────────────────────────────────────────────
   555→
   556→  const handleKeyDown = (e: React.KeyboardEvent) => {
   557→    if (e.key === "Enter" && !e.shiftKey) {
   558→      e.preventDefault();
   559→      sendMessage();
   560→    }
   561→  };
   562→
   563→  // ─── Toggle Thinking ────────────────────────────────────────────────────
   564→
   565→  const toggleThinking = (id: string) => {
   566→    setExpandedThinking((prev) => ({ ...prev, [id]: !prev[id] }));
   567→  };
   568→
   569→  // ─── Clear Chat ─────────────────────────────────────────────────────────
   570→
   571→  const clearChat = () => {
   572→    setMessages([]);
   573→    setSessionStats({ messageCount: 0, toolsUsed: 0, avgResponseTime: 0, totalTokens: 0 });
   574→    responseTimesRef.current = [];
   575→  };
   576→
   577→  // ─── Render ─────────────────────────────────────────────────────────────
   578→
   579→  return (
   580→    <div className="h-screen flex flex-col bg-[#000000] text-[#e0e0e0] overflow-hidden" style={{ fontFamily: "var(--font-geist-mono), 'JetBrains Mono', 'Fira Code', ui-monospace, monospace" }}>
   581→      {/* ─── Top Bar ─────────────────────────────────────────────────────── */}
   582→      <header className="flex items-center justify-between h-9 px-4 border-b border-[rgba(255,255,255,0.06)] shrink-0 select-none">
   583→        <div className="flex items-center gap-3">
