    55→def crear_grafico_avanzado(
    56→    ruta: str,
    57→    tipo: str = "bar",
    58→    datos: str = "",
    59→    titulo: str = "",
    60→    xlabel: str = "",
    61→    ylabel: str = "",
    62→    opciones: str = "{}",
    63→) -> str:
    64→    """Crea un grafico avanzado y lo guarda como imagen PNG/SVG.
    65→    Soporta 15+ tipos: bar, line, pie, scatter, histogram, area,
    66→    heatmap, radar, candlestick, boxplot, waterfall, regression,
    67→    distribution, violin, stem.
    68→
    69→    Args:
    70→        ruta: Ruta donde guardar la imagen (PNG o SVG)
    71→        tipo: Tipo de grafico (ver lista arriba)
    72→        datos: Datos en formato JSON o CSV
    73→        titulo: Titulo del grafico
    74→        xlabel: Etiqueta del eje X
    75→        ylabel: Etiqueta del eje Y
    76→        opciones: Opciones extra en JSON (colores, leyenda, grid, etc.)
    77→    """
    78→    validation = validate_path(ruta)
    79→    if validation != ruta:
    80→        return validation
    81→
    82→    tipo = tipo.lower().strip()
    83→    if tipo not in SUPPORTED_CHART_TYPES:
    84→        return (f"ERROR: Tipo '{tipo}' no soportado. "
    85→                f"Usar: {', '.join(SUPPORTED_CHART_TYPES)}")
    86→
    87→    try:
    88→        import matplotlib
    89→        matplotlib.use('Agg')
    90→        import matplotlib.pyplot as plt
    91→        import matplotlib.font_manager as fm
    92→        import numpy as np
    93→
    94→        # Configurar fuentes
    95→        _setup_fonts()
    96→
    97→        # Parsear datos y opciones
    98→        parsed_data = _parse_chart_data(datos)
    99→        opts = _parse_options(opciones)
   100→
   101→        # Crear grafico
   102→        fig, ax = plt.subplots(figsize=opts.get("figsize", (10, 6)))
   103→        fig.patch.set_facecolor(opts.get("bg_color", "white"))
   104→
   105→        # Dispatch por tipo
   106→        handler = _CHART_HANDLERS.get(tipo)
   107→        if handler:
   108→            handler(ax, parsed_data, opts, title=titulo, xlabel=xlabel, ylabel=ylabel)
   109→        else:
   110→            plt.close()
   111→            return f"ERROR: Handler no implementado para tipo '{tipo}'"
   112→
   113→        # Aplicar estilo comun
   114→        _apply_common_style(ax, titulo, xlabel, ylabel, opts)
   115→
   116→        # Guardar
   117→        dir_name = os.path.dirname(ruta)
   118→        if dir_name:
   119→            os.makedirs(dir_name, exist_ok=True)
   120→
   121→        dpi = opts.get("dpi", 150)
   122→        bbox = opts.get("bbox_inches", "tight")
   123→        plt.tight_layout()
   124→        plt.savefig(ruta, dpi=dpi, bbox_inches=bbox,
   125→                    facecolor=fig.get_facecolor(), edgecolor='none')
   126→        plt.close()
   127→
   128→        size_kb = os.path.getsize(ruta) / 1024
   129→        return f"Grafico [{tipo}] creado: {ruta} ({size_kb:.0f} KB)"
   130→
   131→    except ImportError as e:
   132→        return (f"ERROR: Faltan dependencias para visualizacion: {e}\n"
   133→                "Instala: pip install matplotlib numpy")
   134→    except Exception as e:
   135→        logger.error(f"Error creando grafico: {e}")
   136→        return f"ERROR creando grafico: {e}"

   139→def crear_dashboard(
   140→    ruta: str,
   141→    graficos: str = "[]",
   142→    titulo: str = "Dashboard",
   143→    layout: str = "auto",
   144→    opciones: str = "{}",
   145→) -> str:
   146→    """Crea un dashboard con multiples graficos en una sola imagen.
   147→    Cada grafico se define como un objeto JSON con tipo, datos y opciones.
   148→
   149→    Args:
   150→        ruta: Ruta donde guardar el dashboard (PNG o SVG)
   151→        graficos: Lista JSON de graficos, cada uno con: tipo, datos, titulo, opciones
   152→        titulo: Titulo del dashboard
   153→        layout: Layout del grid: 'auto', '2x2', '3x2', '2x3', '1x3', '3x1'
   154→        opciones: Opciones globales en JSON
   155→    """
   156→    validation = validate_path(ruta)
   157→    if validation != ruta:
   158→        return validation
   159→
   160→    try:
   161→        import matplotlib
   162→        matplotlib.use('Agg')
   163→        import matplotlib.pyplot as plt
   164→        import matplotlib.font_manager as fm
   165→        import numpy as np
   166→
   167→        _setup_fonts()
   168→        opts = _parse_options(opciones)
   169→
   170→        # Parsear lista de graficos
   171→        charts = _parse_dashboard_charts(graficos)
   172→        if not charts:
   173→            return "ERROR: No se proporcionaron graficos. Formato: [{tipo, datos, titulo}, ...]"
   174→
   175→        n = len(charts)
   176→
   177→        # Calcular layout
   178→        rows, cols = _calc_layout(n, layout)
   179→
   180→        # Crear figura
   181→        fig_width = cols * 5
   182→        fig_height = rows * 4 + 1.5  # Extra para titulo
   183→        fig, axes = plt.subplots(rows, cols, figsize=(fig_width, fig_height))
   184→        fig.suptitle(titulo, fontsize=16, fontweight='bold', y=0.98)
   185→
   186→        # Aplanar axes para iteracion
   187→        if rows == 1 and cols == 1:
   188→            axes = [[axes]]
   189→        elif rows == 1:
   190→            axes = [axes]
   191→        elif cols == 1:
   192→            axes = [[ax] for ax in axes]
   193→
   194→        # Renderizar cada grafico
   195→        for idx, chart in enumerate(charts):
   196→            row = idx // cols
   197→            col = idx % cols
   198→            if row < rows and col < cols:
   199→                ax = axes[row][col]
   200→                chart_type = chart.get("tipo", "bar")
   201→                chart_data = _parse_chart_data(chart.get("datos", ""))
   202→                chart_opts = _parse_options(chart.get("opciones", "{}"))
   203→                chart_title = chart.get("titulo", "")
   204→
   205→                handler = _CHART_HANDLERS.get(chart_type)
   206→                if handler:
   207→                    handler(ax, chart_data, chart_opts)
   208→                    _apply_common_style(ax, chart_title, "", "", chart_opts)
   209→                else:
   210→                    ax.text(0.5, 0.5, f"Tipo '{chart_type}' no soportado",
   211→                            ha='center', va='center', transform=ax.transAxes)
   212→                    ax.set_title(chart_title)
   213→
   214→        # Ocultar axes vacios
   215→        for idx in range(n, rows * cols):
   216→            row = idx // cols
   217→            col = idx % cols
   218→            if row < rows and col < cols:
   219→                axes[row][col].set_visible(False)
   220→
   221→        # Guardar
   222→        dir_name = os.path.dirname(ruta)
   223→        if dir_name:
   224→            os.makedirs(dir_name, exist_ok=True)
   225→
   226→        plt.tight_layout(rect=[0, 0, 1, 0.95])
   227→        plt.savefig(ruta, dpi=150, bbox_inches="tight",
   228→                    facecolor='white', edgecolor='none')
   229→        plt.close()
   230→
   231→        size_kb = os.path.getsize(ruta) / 1024
   232→        return f"Dashboard creado: {ruta} ({size_kb:.0f} KB, {n} graficos, layout {rows}x{cols})"
   233→
   234→    except ImportError:
   235→        return "ERROR: Instala matplotlib y numpy: pip install matplotlib numpy"
   236→    except Exception as e:
   237→        logger.error(f"Error creando dashboard: {e}")
   238→        return f"ERROR creando dashboard: {e}"

   245→def _chart_bar(ax, data, opts, **kwargs):
   246→    """Grafico de barras (horizontal/vertical, agrupado, apilado)."""
   247→    labels = data.get("labels", [])
   248→    values = data.get("values", [])
   249→    if not values:
   250→        return
   251→
   252→    horizontal = opts.get("horizontal", False)
   253→    stacked = opts.get("stacked", False)
   254→
   255→    # Multiple series?
   256→    if isinstance(values[0], list):
   257→        # Datos multi-serie
   258→        x = np.arange(len(labels))
   259→        width = 0.8 / len(values)
   260→        colors = opts.get("colors", CHART_COLORS)
   261→        series_names = opts.get("series_names", [f"Serie {i+1}" for i in range(len(values))])
   262→
   263→        for i, serie in enumerate(values):
   264→            offset = (i - len(values)/2 + 0.5) * width
   265→            if stacked:
   266→                if i == 0:
   267→                    bottom = np.zeros(len(labels))
   268→                else:
   269→                    bottom = np.array(values[i-1]) if i == 1 else np.add.reduce(values[:i])
   270→                ax.bar(x, serie, width, label=series_names[i],
   271→                       color=colors[i % len(colors)], bottom=bottom)
   272→            else:
   273→                if horizontal:
   274→                    ax.barh(x + offset, serie, width, label=series_names[i],
   275→                            color=colors[i % len(colors)])
   276→                else:
   277→                    ax.bar(x + offset, serie, width, label=series_names[i],
   278→                           color=colors[i % len(colors)])
   279→        ax.set_xticks(x)
   280→        ax.set_xticklabels(labels, rotation=45, ha='right')
   281→        ax.legend(loc='best', fontsize=8)
   282→    else:
   283→        color = opts.get("color", PALETTE["primary"])
   284→        if horizontal:
   285→            ax.barh(labels, values, color=color)
   286→        else:
   287→            ax.bar(labels, values, color=color)
   288→            plt.xticks(rotation=45, ha='right')

   291→def _chart_line(ax, data, opts, **kwargs):
   292→    """Grafico de lineas (simple, multi-serie, con marcadores)."""
   293→    labels = data.get("labels", [])
   294→    values = data.get("values", [])

   311→        color = opts.get("color", PALETTE["primary"])

   315→            ax.fill_between(range(len(labels)), values, alpha=0.1, color=color)

   420→def _chart_heatmap(ax, data, opts, **kwargs):
   421→    """Mapa de calor (heatmap) con anotaciones."""
   422→    matrix = data.get("matrix", data.get("values", []))
   423→    row_labels = data.get("row_labels", data.get("labels", []))
   424→    col_labels = data.get("col_labels", [])
   425→
   426→    if not matrix:
   427→        return
   428→
   429→    import numpy as np
   430→    mat = np.array(matrix, dtype=float)
   431→
   432→    cmap = opts.get("cmap", "YlOrRd")
   433→    annot = opts.get("annot", True)
   434→    fmt = opts.get("fmt", ".1f")
   435→
   436→    im = ax.imshow(mat, cmap=cmap, aspect='auto')
   437→
   438→    if annot:
   439→        for i in range(mat.shape[0]):
   440→            for j in range(mat.shape[1]):
   441→                val = mat[i, j]
   442→                text_color = "white" if val > np.max(mat) * 0.7 else "black"
   443→                ax.text(j, i, f"{val:{fmt}}", ha="center", va="center",
   444→                        color=text_color, fontsize=8)
   445→
   446→    if col_labels:
   447→        ax.set_xticks(range(len(col_labels)))
   448→        ax.set_xticklabels(col_labels, rotation=45, ha='right')
   449→    if row_labels:
   450→        ax.set_yticks(range(len(row_labels)))
   451→        ax.set_yticklabels(row_labels)
   452→
   453→    plt.colorbar(im, ax=ax, shrink=0.8)

   457→    """Grafico radar/spider."""
   458→    labels = data.get("labels", [])
   459→    values = data.get("values", [])
   460→
   461→    if not labels or not values:
   462→        return
   463→
   464→    import numpy as np
   465→
   466→    # Multi-serie?
   467→    if isinstance(values[0], list):
   468→        series = values
   469→        series_names = opts.get("series_names", [f"Serie {i+1}" for i in range(len(series))])
   470→        colors = opts.get("colors", CHART_COLORS)
   471→    else:
   472→        series = [values]

   474→        colors = [opts.get("color", PALETTE["primary"])]
   475→
   476→    N = len(labels)

   478→    angles += angles[:1]  # Cerrar
   479→
   480→    ax.set_theta_offset(np.pi / 2)
   481→    ax.set_theta_direction(-1)
   482→    ax.set_rlabel_position(0)
   483→
   484→    ax.set_xticks(angles[:-1])
   485→    ax.set_xticklabels(labels)
   486→
   487→    for i, serie in enumerate(series):
   488→        values_closed = serie + serie[:1]
   489→        ax.plot(angles, values_closed, 'o-', linewidth=2,
   490→                color=colors[i % len(colors)], label=series_names[i], markersize=5)
   491→        ax.fill(angles, values_closed, alpha=0.15, color=colors[i % len(colors)])
   492→
   493→    ax.legend(loc='upper right', bbox_to_anchor=(0.1, 0.1), fontsize=8)

   496→def _chart_candlestick(ax, data, opts, **kwargs):
   497→    """Grafico de velas (candlestick) para datos financieros."""
   498→    import numpy as np
   499→
   500→    candles = data.get("candles", data.get("values", []))
   501→    labels = data.get("labels", [])
   502→
   503→    if not candles:
   504→        return
   505→
   506→    # Parsear: cada vela = [open, high, low, close] o dict
   507→    parsed = []
   508→    for c in candles:
   509→        if isinstance(c, dict):
   510→            parsed.append([c.get("open", 0), c.get("high", 0),
   511→                           c.get("low", 0), c.get("close", 0)])
   512→        elif isinstance(c, (list, tuple)) and len(c) >= 4:
   513→            parsed.append(list(c[:4]))
   514→
   515→    if not parsed:
   516→        return
   517→
   518→    width = opts.get("width", 0.6)
   519→    x_range = range(len(parsed))
   520→
   521→    for i, (o, h, l, cl) in enumerate(parsed):
   522→        # Color: verde si cierra por encima de apertura, rojo si no
   523→        color = PALETTE["success"] if cl >= o else PALETTE["danger"]
   524→        # Sombra (mecha)
   525→        ax.plot([i, i], [l, h], color=color, linewidth=1)
   526→        # Cuerpo
   527→        body_bottom = min(o, cl)
   528→        body_height = abs(cl - o) or 0.01  # Minimo para velas doji
   529→        ax.bar(i, body_height, bottom=body_bottom, width=width,

   562→def _chart_waterfall(ax, data, opts, **kwargs):
   563→    """Grafico de cascada (waterfall) para analisis incremental."""
   564→    import numpy as np
   565→
   566→    labels = data.get("labels", [])
   567→    values = data.get("values", [])
   568→
   569→    if not labels or not values:
   570→        return
   571→
   572→    cumulative = [0]
   573→    for v in values[:-1] if len(values) > 1 else values:
   574→        cumulative.append(cumulative[-1] + v)
   575→
   576→    # Agregar total al final si no existe
   577→    has_total = opts.get("show_total", True)
   578→    if has_total and len(values) > 1:
   579→        labels.append("Total")
   580→        values.append(cumulative[-1] + values[-1] if len(values) > 1 else values[0])
   581→        cumulative.append(0)
   582→
   583→    bottoms = []
   584→    heights = []
   585→    colors = []
   586→
   587→    for i, v in enumerate(values):
   588→        if labels[i] == "Total":
   589→            bottoms.append(0)
   590→            heights.append(v)
   591→            colors.append(PALETTE["primary"])
   592→        elif v >= 0:
   593→            bottoms.append(cumulative[i])
   594→            heights.append(v)
   595→            colors.append(PALETTE["success"])
   596→        else:
   597→            bottoms.append(cumulative[i] + v)
   598→            heights.append(abs(v))
   599→            colors.append(PALETTE["danger"])
   600→
   601→    ax.bar(labels, heights, bottom=bottoms, color=colors)
   602→
   603→    # Anotaciones
   604→    for i, (b, h, v) in enumerate(zip(bottoms, heights, values)):
   605→        y_pos = b + h if v >= 0 or labels[i] == "Total" else b
   606→        prefix = "+" if v > 0 and labels[i] != "Total" else ""
   607→        ax.text(i, y_pos + 0.02 * max(abs(v) for v in values),
   608→                f"{prefix}{v:.1f}", ha='center', va='bottom', fontsize=9)
   609→
   610→    plt.xticks(rotation=45, ha='right')

   613→def _chart_regression(ax, data, opts, **kwargs):
   614→    """Grafico de dispersion con linea de regresion."""
   615→    import numpy as np
   616→
   617→    x_data = data.get("x", data.get("values", []))
   618→    y_data = data.get("y", [])
   619→
   620→    if not y_data:
   621→        if isinstance(x_data[0], (list, tuple)) and len(x_data[0]) >= 2:
   622→            y_data = [p[1] for p in x_data]
   623→            x_data = [p[0] for p in x_data]
   624→        else:
   625→            y_data = x_data
   626→            x_data = list(range(len(y_data)))
   627→
   628→    x = np.array(x_data, dtype=float)
   629→    y = np.array(y_data, dtype=float)
   630→
   631→    # Scatter
   632→    color = opts.get("color", PALETTE["primary"])
   633→    ax.scatter(x, y, s=40, c=color, alpha=0.6, edgecolors='white')
   634→
   635→    # Regresion lineal
   636→    if len(x) > 1:
   637→        z = np.polyfit(x, y, 1)
   638→        p = np.poly1d(z)
   639→        x_line = np.linspace(x.min(), x.max(), 100)
   640→        ax.plot(x_line, p(x_line), color=PALETTE["danger"], linewidth=2,
   641→                linestyle='--', label=f'y = {z[0]:.2f}x + {z[1]:.2f}')
   642→
   643→        # R-squared
   644→        y_pred = p(x)
   645→        ss_res = np.sum((y - y_pred) ** 2)
   646→        ss_tot = np.sum((y - np.mean(y)) ** 2)
   647→        r_squared = 1 - (ss_res / ss_tot) if ss_tot > 0 else 0
   648→        ax.text(0.05, 0.95, f'R² = {r_squared:.3f}',
   649→                transform=ax.transAxes, fontsize=10, va='top',
   650→                bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5))
   651→
   652→        ax.legend(loc='best', fontsize=9)

   655→def _chart_distribution(ax, data, opts, **kwargs):
   656→    """Grafico de distribucion con histograma + curva normal."""
   657→    import numpy as np
   658→
   659→    values = data.get("values", [])
   660→    if not values:
   661→        return
   662→
   663→    values = np.array(values, dtype=float)
   664→    bins = opts.get("bins", min(30, max(5, len(values) // 5)))
   665→    color = opts.get("color", PALETTE["primary"])
   666→
   667→    # Histograma normalizado
   668→    n, bins_edges, patches = ax.hist(values, bins=bins, density=True,
   669→                                      color=color, alpha=0.6, edgecolor='white')
   670→
   671→    # Curva normal
   672→    mu = np.mean(values)
   673→    sigma = np.std(values)
   674→    x = np.linspace(mu - 4*sigma, mu + 4*sigma, 200)
   675→
   676→    try:
   677→        from scipy.stats import norm
   678→        ax.plot(x, norm.pdf(x, mu, sigma), 'r-', linewidth=2, label='Normal')
   679→    except ImportError:
   680→        # Aproximacion sin scipy
   681→        pdf = (1 / (sigma * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x - mu) / sigma) ** 2)
   682→        ax.plot(x, pdf, 'r-', linewidth=2, label='Normal')
   683→
   684→    # Estadisticas
   685→    stats_text = f'Media: {mu:.2f}\nDesv.Est: {sigma:.2f}\nN: {len(values)}'
   686→    ax.text(0.95, 0.95, stats_text, transform=ax.transAxes, fontsize=9,
   687→            va='top', ha='right', bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5))
   688→    ax.legend(loc='best')

   788→def _parse_chart_data(datos_str: str) -> dict:
   789→    """Parsea datos de grafico desde JSON o CSV a estructura estandar.
   790→
   791→    Formatos soportados:
   792→    - JSON con keys: labels, values, x, y, matrix, candles, sizes, colors, row_labels, col_labels
   793→    - CSV simple: etiqueta,valor por linea
   794→    - Lista de puntos: x,y por linea
   795→    """
   796→    if not datos_str:
   797→        return {"labels": [], "values": []}
   798→
   799→    datos_str = datos_str.strip()
   800→
   801→    # Intentar JSON
   802→    if datos_str.startswith('{') or datos_str.startswith('['):
   803→        try:
   804→            parsed = json.loads(datos_str)
   805→            if isinstance(parsed, dict):
   806→                return parsed
   807→            elif isinstance(parsed, list):
   808→                if len(parsed) > 0 and isinstance(parsed[0], dict):
   809→                    # Lista de dicts -> extraer labels y values
   810→                    keys = list(parsed[0].keys())
   811→                    return {
   812→                        "labels": [str(d.get(keys[0], "")) for d in parsed],
   813→                        "values": [d.get(keys[1], 0) if len(keys) > 1 else 0 for d in parsed],
   814→                    }
   815→                elif len(parsed) > 0 and isinstance(parsed[0], (list, tuple)):
   816→                    return {"labels": [str(r[0]) for r in parsed],
   817→                            "values": [r[1] if len(r) > 1 else 0 for r in parsed]}
   818→                else:
   819→                    return {"labels": [str(i) for i in range(len(parsed))],
   820→                            "values": parsed}
   821→        except json.JSONDecodeError:
   822→            pass
   823→
   824→    # Fallback: CSV
   825→    result = {"labels": [], "values": []}
   826→    for line in datos_str.split("\n"):
   827→        line = line.strip()
   828→        if not line or line.startswith("#"):
   829→            continue
   830→
   831→        # Intentar punto x,y
   832→        if "," in line:
   833→            parts = line.rsplit(",", 1)
   834→            if len(parts) == 2:
   835→                result["labels"].append(parts[0].strip())
   836→                try:
   837→                    result["values"].append(float(parts[1].strip()))
   838→                except ValueError:
   839→                    result["values"].append(0)
   840→        else:
   841→            try:
   842→                result["values"].append(float(line))
   843→            except ValueError:
   844→                result["labels"].append(line)
   845→
   846→    return result

   869→def _calc_layout(n: int, layout: str = "auto") -> tuple:
   870→    """Calcula filas y columnas para el dashboard."""
   871→    if layout == "auto":
   872→        if n <= 1: return (1, 1)
   873→        if n <= 2: return (1, 2)
   874→        if n <= 4: return (2, 2)
   875→        if n <= 6: return (2, 3)
   876→        if n <= 9: return (3, 3)
   877→        return ((n + 3) // 4, 4)
   878→
   879→    parts = layout.lower().split('x')
   880→    if len(parts) == 2:
   881→        try:
   882→            return (int(parts[0]), int(parts[1]))
   883→        except ValueError:
   884→            pass
   885→
   886→    return (2, 2)
