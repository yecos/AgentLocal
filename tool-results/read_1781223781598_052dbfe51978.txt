    64тЖТ    64тЖТ  if (bytes === 0) return "0 B";
    65тЖТ    65тЖТ  const k = 1024;
    66тЖТ    66тЖТ  const sizes = ["B", "KB", "MB", "GB", "TB"];
    67тЖТ    67тЖТ  const i = Math.floor(Math.log(bytes) / Math.log(k));
    68тЖТ    68тЖТ  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + " " + sizes[i];
    69тЖТ    69тЖТ}
