    63→function formatBytes(bytes: number): string {
    64→  if (bytes === 0) return "0 B";
    65→  const k = 1024;
    66→  const sizes = ["B", "KB", "MB", "GB", "TB"];
    67→  const i = Math.floor(Math.log(bytes) / Math.log(k));
    68→  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + " " + sizes[i];
    69→}
    70→
    71→function formatUptime(seconds: number): string {
    72→  const h = Math.floor(seconds / 3600);
    73→  const m = Math.floor((seconds % 3600) / 60);
    74→  const s = Math.floor(seconds % 60);
    75→  return `${h.toString().padStart(2, "0")}:${m.toString().padStart(2, "0")}:${s.toString().padStart(2, "0")}`;
    76→}

   172→// ─── Chat Message Component ──────────────────────────────────────────────────
   173→
   174→function ChatMessage({
   175→  msg,
   176→  isLastAssistant,
   177→  isLoading,
   178→  expandedThinking,
   179→  toggleThinking,
   180→}: {
   181→  msg: Message;
   182→  isLastAssistant: boolean;
   183→  isLoading: boolean;
   184→  expandedThinking: Record<string, boolean>;
   185→  toggleThinking: (id: string) => void;
   186→}) {
   187→  if (msg.role === "user") {
   188→    return (
   189→      <div className="flex justify-end animate-fade-in">
   190→        <div className="max-w-[70%] px-3.5 py-2.5 border border-[rgba(255,255,255,0.08)] bg-[rgba(255,255,255,0.02)] text-[15px] leading-[1.7] tracking-[0.01em] text-[#f0f0f0]">
   191→          {msg.content}
   192→        </div>
   193→      </div>
   194→    );
   195→  }
   196→
   197→  const isStreaming = isLastAssistant && isLoading;
   198→  const cleanedThinking = msg.thinking ? cleanThinking(msg.thinking) : "";
   199→
   200→  return (
   201→    <div className="animate-fade-in">
   202→      <div className="max-w-[85%]">
   203→        <div className="border-l border-[rgba(0,212,255,0.2)] pl-3">
   204→          {/* Thinking section */}
   205→          {cleanedThinking && (
   206→            <div className="mb-2.5">
   207→              <button
   208→                onClick={() => toggleThinking(msg.id)}
   209→                className="flex items-center gap-1 text-[10px] text-[#666666] hover:text-[#999999] transition-colors duration-150"
   210→              >
   211→                {expandedThinking[msg.id] ? (
   212→                  <ChevronDown size={10} />
   213→                ) : (
   214→                  <ChevronRight size={10} />
   215→                )}
   216→                <span>thinking</span>
   217→                <span className="text-[#555555] ml-1">
   218→                  ({cleanedThinking.length} chars)
   219→                </span>
   220→              </button>
   221→              {expandedThinking[msg.id] && (
   222→                <div className="mt-1.5 text-[11px] leading-[1.6] text-[#555555] max-h-40 overflow-y-auto pr-2 whitespace-pre-wrap border-l border-[rgba(255,255,255,0.06)] pl-2">
   223→                  {cleanedThinking}
   224→                </div>
   225→              )}
   226→            </div>
   227→          )}
   228→
   229→          {/* Tool calls */}
   230→          {msg.toolCalls && msg.toolCalls.length > 0 && (
   231→            <div className="flex flex-wrap gap-1.5 mb-2">
   232→              {msg.toolCalls.map((tool, i) => (
   233→                <span
   234→                  key={i}
   235→                  className="text-[10px] px-1.5 py-0.5 border border-[rgba(0,212,255,0.2)] text-[rgba(0,212,255,0.7)]"
   236→                >
   237→                  [{tool}]
   238→                </span>
   239→              ))}
   240→            </div>
   241→          )}
   242→
   243→          {/* Content */}
   244→          <div className="text-[15px] leading-[1.7] tracking-[0.01em] text-[#e0e0e0] whitespace-pre-wrap">
   245→            {msg.content}
   246→            {/* Streaming cursor */}
   247→            {isStreaming && !msg.content && (
   248→              <span className="inline-block w-[7px] h-[16px] bg-[rgba(0,212,255,0.5)] cursor-blink ml-0.5" />
   249→            )}
   250→            {isStreaming && msg.content && (
   251→              <span className="inline-block w-[6px] h-[15px] bg-[rgba(0,212,255,0.5)] cursor-blink ml-0.5 align-middle" />
   252→            )}
   253→          </div>
   254→
   255→          {/* Response metadata */}
   256→          {msg.responseTime && !isStreaming && (
   257→            <div className="mt-2 text-[10px] text-[#555555] flex items-center gap-2">
   258→              <span>{msg.responseTime}ms</span>
   259→              {msg.tokenCount ? (
   260→                <>
   261→                  <span className="text-[#444444]">·</span>
   262→                  <span>~{msg.tokenCount} tokens</span>
   263→                </>
   264→              ) : null}
   265→            </div>
   266→          )}
   267→        </div>
   268→      </div>
   269→    </div>
   270→  );
   271→}

   275→export default function ZAIInterface() {
   276→  // State
   277→  const [messages, setMessages] = useState<Message[]>([]);
   278→  const [input, setInput] = useState("");
   279→  const [isLoading, setIsLoading] = useState(false);
   280→  const [selectedModel, setSelectedModel] = useState<string>("qwen3:4b");
   281→  const [status, setStatus] = useState<SystemStatus>({
   282→    connected: false,
   283→    models: [],
   284→    modelCount: 0,
   285→  });
   286→  const [uptime, setUptime] = useState(0);
   287→  const [sessionStats, setSessionStats] = useState<SessionStats>({
   288→    messageCount: 0,
   289→    toolsUsed: 0,
   290→    avgResponseTime: 0,
   291→    totalTokens: 0,
   292→  });
   293→  const [sidebarOpen, setSidebarOpen] = useState(true);
   294→  const [expandedThinking, setExpandedThinking] = useState<Record<string, boolean>>({});
   295→  const [isMobile, setIsMobile] = useState(false);
   296→  const [useAgent, setUseAgent] = useState(true);
   297→
   298→  // Refs
   299→  const chatEndRef = useRef<HTMLDivElement>(null);
   300→  const inputRef = useRef<HTMLTextAreaElement>(null);
   301→  const responseTimesRef = useRef<number[]>([]);
   302→
   303→  // ─── Detect Mobile ──────────────────────────────────────────────────────
   304→
   305→  useEffect(() => {
   306→    const checkMobile = () => setIsMobile(window.innerWidth < 768);
   307→    checkMobile();
   308→    window.addEventListener("resize", checkMobile);
   309→    return () => window.removeEventListener("resize", checkMobile);
   310→  }, []);
   311→
   312→  // Close sidebar by default on mobile
   313→  useEffect(() => {
   314→    if (isMobile) setSidebarOpen(false);
   315→  }, [isMobile]);
   316→
   317→  // ─── Fetch Status ────────────────────────────────────────────────────────
   318→
   319→  const fetchStatus = useCallback(async () => {
   320→    try {
   321→      const res = await fetch("/api/status");
   322→      const data = await res.json();
   323→      setStatus(data);
   324→      if (data.connected && data.models.length > 0) {
   325→        setSelectedModel((prev) => {
   326→          if (data.models.find((m: OllamaModel) => m.name === prev)) return prev;
   327→          return data.models[0].name;
   328→        });
   329→      }
   330→    } catch {
   331→      setStatus({ connected: false, agentAvailable: false, models: [], modelCount: 0, error: "Failed to fetch" });
   332→    }
   333→  }, []);
   334→
   335→  // ─── Effects ─────────────────────────────────────────────────────────────
   336→
   337→  useEffect(() => {
   338→    fetchStatus();
   339→    const interval = setInterval(fetchStatus, 10000);
   340→    return () => clearInterval(interval);
   341→  }, [fetchStatus]);
   342→
   343→  useEffect(() => {
   344→    const timer = setInterval(() => setUptime((prev) => prev + 1), 1000);
   345→    return () => clearInterval(timer);
   346→  }, []);
   347→
   348→  useEffect(() => {
   349→    chatEndRef.current?.scrollIntoView({ behavior: "smooth" });
   350→  }, [messages]);
   351→
   352→  // ─── Auto-select first model ────────────────────────────────────────────
   353→
   354→  const lastAssistantId = useMemo(() => {
   355→    for (let i = messages.length - 1; i >= 0; i--) {
   356→      if (messages[i].role === "assistant") return messages[i].id;
   357→    }
   358→    return null;
   359→  }, [messages]);
   360→
   361→  // ─── Send Message ───────────────────────────────────────────────────────
   362→
   363→  const sendMessage = async () => {
   364→    if (!input.trim() || isLoading) return;
   365→
   366→    const userMessage: Message = {
   367→      id: generateId(),
   368→      role: "user",
   369→      content: input.trim(),
   370→      timestamp: Date.now(),
   371→    };
   372→
   373→    const assistantId = generateId();
   374→    const assistantMessage: Message = {
   375→      id: assistantId,
   376→      role: "assistant",
   377→      content: "",
   378→      timestamp: Date.now(),
   379→      isStreaming: true,
   380→    };
   381→
   382→    setMessages((prev) => [...prev, userMessage, assistantMessage]);
   383→    setInput("");
   384→    setIsLoading(true);
   385→
   386→    // Reset textarea height
   387→    if (inputRef.current) {
   388→      inputRef.current.style.height = "auto";
   389→    }
   390→
   391→    const startTime = Date.now();
   392→
   393→    const ollamaMessages = [...messages, userMessage].map((m) => ({
   394→      role: m.role,
   395→      content: m.content,
   396→    }));
   397→
   398→    try {
   399→      const res = await fetch("/api/chat", {
   400→        method: "POST",
   401→        headers: { "Content-Type": "application/json" },
   402→        body: JSON.stringify({ messages: ollamaMessages, model: selectedModel, useAgent }),
   403→      });
   404→
   405→      if (!res.ok) {
   406→        const errorData = await res.json().catch(() => ({ error: "Unknown error" }));
   407→        let errorMsg = errorData.error || "Unknown error";
   408→        if (errorData.bridgeRequired) {
   409→          errorMsg = `AGENT mode requires the bridge. Run: python bridge_api.py`;
   410→        }
   411→        setMessages((prev) =>
   412→          prev.map((m) =>
   413→            m.id === assistantId
   414→              ? { ...m, content: errorMsg, isStreaming: false, responseTime: Date.now() - startTime }
   415→              : m
   416→          )
   417→        );
   418→        setIsLoading(false);
   419→        return;
   420→      }
   421→
   422→      const reader = res.body?.getReader();
   423→      if (!reader) {
   424→        setIsLoading(false);
   425→        return;
   426→      }
   427→
   428→      const decoder = new TextDecoder();
   429→      let fullContent = "";
   430→      let thinkingContent = "";
   431→      let isThinking = false;
   432→      let toolCalls: string[] = [];
   433→      let tokenCount = 0;
   434→
   435→      while (true) {
   436→        const { done, value } = await reader.read();
   437→        if (done) break;
   438→
   439→        const chunk = decoder.decode(value, { stream: true });
   440→        const lines = chunk.split("\n").filter((line) => line.startsWith("data: "));
   441→
   442→        for (const line of lines) {
   443→          const data = line.slice(6);
   444→          if (data === "[DONE]") continue;
   445→
   446→          try {
   447→            const parsed = JSON.parse(data);
   448→            if (parsed.error) {
   449→              fullContent += `\nError: ${parsed.error}`;
   450→              break;
   451→            }
   452→
   453→            // Bridge agent events (type field present)
   454→            if (parsed.type) {
   455→              if (parsed.type === "text") {
   456→                const content = parsed.data || "";
   457→                if (content.includes("<think")) {
   458→                  isThinking = true;
   459→                }
   460→                if (isThinking) {
   461→                  thinkingContent += content;
   462→                  if (content.includes("</think")) {
   463→                    isThinking = false;
   464→                  }
   465→                } else {
   466→                  fullContent += content;
   467→                }
   468→                tokenCount++;
   469→              } else if (parsed.type === "tool_start") {
   470→                const toolName = parsed.data?.name || "unknown";
   471→                toolCalls = [...new Set([...toolCalls, toolName])];
   472→              } else if (parsed.type === "tool_result") {
   473→                // Tool completed - already tracked in toolCalls
   474→              } else if (parsed.type === "meta") {
   475→                // Metacognition event - we could display this
   476→              } else if (parsed.type === "done") {
   477→                // Agent finished
   478→              } else if (parsed.type === "error") {
   479→                fullContent += `\nError: ${parsed.data}`;
   480→              }
   481→            } else {
   482→              // Direct Ollama format (no type field)
   483→              const content = parsed.message?.content || "";
   484→              tokenCount++;
   485→
   486→              if (content.includes("<think")) {
   487→                isThinking = true;
   488→              }
   489→
   490→              if (isThinking) {
   491→                thinkingContent += content;
   492→                if (content.includes("</think")) {
   493→                  isThinking = false;
   494→                }
   495→              } else {
   496→                fullContent += content;
   497→              }
   498→
   499→              // Detect tool calls in text
   500→              const toolMatches = content.match(/\[([a-z_]+)\]/g);
   501→              if (toolMatches) {
   502→                toolCalls = [...new Set([...toolCalls, ...toolMatches.map((t) => t.slice(1, -1))])];
   503→              }
   504→            }
   505→          } catch {
   506→            // Skip malformed JSON
   507→          }
   508→        }
   509→
   510→        setMessages((prev) =>
   511→          prev.map((m) =>
   512→            m.id === assistantId
   513→              ? {
   514→                  ...m,
   515→                  content: fullContent,
   516→                  thinking: thinkingContent || undefined,
   517→                  toolCalls: toolCalls.length > 0 ? toolCalls : undefined,
   518→                  tokenCount,
   519→                }
   520→              : m
   521→          )
   522→        );
   523→      }
   524→
   525→      const responseTime = Date.now() - startTime;
   526→      responseTimesRef.current.push(responseTime);
   527→
   528→      setMessages((prev) =>
   529→        prev.map((m) =>
   530→          m.id === assistantId ? { ...m, responseTime, isStreaming: false } : m
   531→        )
   532→      );
   533→
   534→      setSessionStats((prev) => ({
   535→        messageCount: prev.messageCount + 1,
   536→        toolsUsed: prev.toolsUsed + toolCalls.length,
   537→        avgResponseTime:
   538→          responseTimesRef.current.length > 0
   539→            ? Math.round(responseTimesRef.current.reduce((a, b) => a + b, 0) / responseTimesRef.current.length)
   540→            : 0,
   541→        totalTokens: prev.totalTokens + tokenCount,
   542→      }));
   543→    } catch (error) {
   544→      const errMsg = error instanceof Error ? error.message : "Unknown error";
   545→      setMessages((prev) =>
   546→        prev.map((m) =>
   547→          m.id === assistantId
   548→            ? { ...m, content: `Connection error: ${errMsg}`, isStreaming: false, responseTime: Date.now() - startTime }
   549→            : m
   550→        )
   551→      );
   552→    } finally {
   553→      setIsLoading(false);
   554→      inputRef.current?.focus();
   555→    }
   556→  };
   557→
   558→  // ─── Handle Key Press ───────────────────────────────────────────────────
   559→
   560→  const handleKeyDown = (e: React.KeyboardEvent) => {
   561→    if (e.key === "Enter" && !e.shiftKey) {
   562→      e.preventDefault();
   563→      sendMessage();
   564→    }
   565→  };
   566→
   567→  // ─── Toggle Thinking ────────────────────────────────────────────────────
   568→
   569→  const toggleThinking = (id: string) => {
   570→    setExpandedThinking((prev) => ({ ...prev, [id]: !prev[id] }));
   571→  };
   572→
   573→  // ─── Clear Chat ─────────────────────────────────────────────────────────
   574→
   575→  const clearChat = () => {
   576→    setMessages([]);
   577→    setSessionStats({ messageCount: 0, toolsUsed: 0, avgResponseTime: 0, totalTokens: 0 });
   578→    responseTimesRef.current = [];
   579→  };
   580→
   581→  // ─── Render ─────────────────────────────────────────────────────────────
   582→
   583→  return (
   584→    <div className="h-screen flex flex-col bg-[#000000] text-[#e0e0e0] overflow-hidden" style={{ fontFamily: "var(--font-geist-mono), 'JetBrains Mono', 'Fira Code', ui-monospace, monospace" }}>
