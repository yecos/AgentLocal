   449→    # ----------------------------------------------------------
   450→    # STREAMING LLM CON TOOL CALLING (token a token)
   451→    # ----------------------------------------------------------
   452→    def _stream_llm_with_tools(self, messages):
   453→        """
   454→        Genera respuesta del LLM con streaming REAL.
   455→        Yields: {"type": "token"|"tool_calls"|"done", "data": ...}
   456→        """
   457→        import time as _time
   458→        _llm_start = _time.monotonic()
   459→        full_content = ""
   460→        full_tool_calls = []
   461→
   462→        try:
   463→            # Intentar streaming HTTP directo (mas rapido, sin overhead de lib)
   464→            for chunk in ollama.generate_stream(messages, tools=TOOL_SCHEMAS):
   465→                if isinstance(chunk, str):
   466→                    # Token de texto - emitir inmediatamente
   467→                    # PERO: si parece JSON de tool call, no emitirlo al usuario
   468→                    full_content += chunk
   469→                    # Detectar si el contenido es JSON de herramienta
   470→                    if self._looks_like_tool_json(full_content):
   471→                        continue  # Acumular sin emitir
   472→                    # Si antes se acumulaba JSON pero ya no, limpiar
   473→                    yield {"type": "token", "data": chunk}
   474→                elif isinstance(chunk, dict):
   475→                    # Resultado final con tool_calls
   476→                    msg = chunk.get("message", chunk)
   477→                    tool_calls = msg.get("tool_calls", [])
   478→                    content = msg.get("content", "")
   479→
   480→                    if content and not full_content:
   481→                        # Contenido que no se streameo (fallback)
   482→                        full_content = content
   483→                        yield {"type": "token", "data": content}
   484→
   485→                    if tool_calls:
   486→                        # Parsear tool calls
   487→                        parsed_calls = []
   488→                        for tc in tool_calls:

   490→                            tool_params = tc.get("function", {}).get("arguments", {})

   492→                            parsed_calls.append({"name": tool_name, "params": tool_params})
   493→                        yield {"type": "tool_calls", "data": parsed_calls}
   494→                        yield {"type": "done", "data": False}
   495→                        return
   496→
   497→            # Si llegamos aqui sin tool_calls, verificar si el contenido es JSON de herramientas
   498→            if full_content:
   499→                # Si el contenido acumulado era JSON de tool calls (no se emitio),
   500→                # parsearlo y ejecutar como tool calls
   501→                if self._looks_like_tool_json(full_content):
   502→                    parsed = self._parse_json(full_content)

   512→                            yield {"type": "done", "data": False}
   513→                            return
   514→                    # Si no se pudo parsear como tool call, emitir el texto limpio
   515→                    clean = self._clean_json_leak(full_content)
   516→                    if clean != full_content:
   517→                        # El texto era JSON, emitir version limpia
   518→                        yield {"type": "token", "data": clean}
   519→                _llm_elapsed = (_time.monotonic() - _llm_start) * 1000.0
   520→                get_metrics().record_llm_call(_llm_elapsed)
   521→                yield {"type": "done", "data": True}
   522→                return
   523→
   524→        except Exception as e:
   525→            self._log(f"Error en streaming: {e}", "error")
   526→            get_metrics().record_error("llm_stream")
   527→
   528→        # Fallback: modo no-streaming
   529→        try:
   530→            response = ollama.generate(messages, tools=TOOL_SCHEMAS)
   531→            # generate() already records llm_call via @timed, so no duplicate
   532→            if isinstance(response, str):
   533→                yield {"type": "token", "data": response}
   534→                yield {"type": "done", "data": True}
   535→                return
   536→
   537→            message = response.get("message", response)
   538→            tool_calls = message.get("tool_calls", [])
   539→            content = message.get("content", "")
   540→
   541→            if tool_calls:
   542→                parsed_calls = []
   543→                for tc in tool_calls:
   544→                    tool_name = tc.get("function", {}).get("name", "")
   545→                    tool_params = tc.get("function", {}).get("arguments", {})
   546→                    self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
   547→                    parsed_calls.append({"name": tool_name, "params": tool_params})
   548→                yield {"type": "tool_calls", "data": parsed_calls}
   549→                yield {"type": "done", "data": False}
   550→                return
   551→
   552→            if content:
   553→                yield {"type": "token", "data": content}
   554→                yield {"type": "done", "data": True}
   555→                return
   556→
   557→        except Exception as e:
   558→            self._log(f"Error en fallback: {e}", "error")
   559→
   560→        yield {"type": "done", "data": False}

   613→    def _react_with_json(self, messages, iteration):
   614→        """ReAct usando JSON parsing (fallback)."""
   615→        # generate_chat() already records llm_call via @timed decorator
   616→        if not any("HERRAMIENTAS DISPONIBLES" in str(m.get("content", "")) for m in messages):
   617→            system_msg_idx = next((i for i, m in enumerate(messages) if m["role"] == "system"), -1)
   618→            if system_msg_idx >= 0:
   619→                messages[system_msg_idx]["content"] += JSON_TOOLS_PROMPT
   620→
   621→        try:
   622→            response = ollama.generate_chat(messages)
   623→            if not response:
   624→                return ("error", "El LLM no respondio")
   625→
   626→            parsed = self._parse_json(response)
   627→            if not parsed:
   628→                # No es JSON - probablemente respuesta directa del modelo
   629→                # Limpiar posibles restos de formato JSON
   630→                clean = self._clean_json_leak(response)
   631→                return ("respond", clean)
   632→
   633→            # 1. Si hay _multi_tool_calls, ejecutar todas las herramientas
   634→            if parsed.get("_multi_tool_calls") and parsed.get("tool_calls"):
   635→                return ("tool_calls", parsed["tool_calls"])
   636→
   637→            # 2. Si hay respuesta_final con contenido, usarla
   638→            respuesta_final = parsed.get("respuesta_final", "").strip()
   639→            if respuesta_final:
   640→                return ("respond", respuesta_final)
   641→
   642→            accion = parsed.get("accion", "").strip()
   643→            params = parsed.get("params", {})
   644→            pensamiento = parsed.get("pensamiento", "").strip()
   645→
   646→            if pensamiento:
   647→                self._log(f"Pensamiento: {pensamiento}", "thinking")
   648→
   649→            # 3. Si hay accion valida, ejecutar herramienta
   650→            if accion and accion in TOOL_FUNCTIONS:
   651→                # Si tambien hay pensamiento, inyectarlo como contexto
   652→                if pensamiento:
   653→                    messages.append({
   654→                        "role": "assistant",
   655→                        "content": f"[Pensamiento: {pensamiento}] Ejecutando {accion}..."
   656→                    })
   657→                return ("tool_calls", [{"name": accion, "params": params}])
   658→
   659→            # 3. Si hay pensamiento pero no accion ni respuesta_final,
   660→            #    el modelo esta respondiendo en campo equivocado - usar pensamiento
   661→            if pensamiento and not accion:
   662→                self._log("Modelo respondio en 'pensamiento' en vez de 'respuesta_final'", "info")
   663→                return ("respond", pensamiento)
   664→
   665→            # 4. Fallback: devolver texto limpio (no JSON crudo)
   666→            clean = self._clean_json_leak(response)
   667→            return ("respond", clean)
   668→
   669→        except Exception as e:
   670→            return ("error", str(e))

   672→    def _clean_json_leak(self, text):
   673→        """Limpia texto que tiene restos de formato JSON para mostrar al usuario.
   674→        v3: Mas agresivo - tambien limpia JSON parcial al inicio/final del texto.
   675→        """
   676→        # Si el texto es JSON completo, extraer solo el contenido util
   677→        parsed = self._parse_json(text)
   678→        if parsed:
   679→            # Prioridad: respuesta_final > pensamiento > texto original
   680→            if parsed.get("respuesta_final", "").strip():
   681→                return parsed["respuesta_final"].strip()
   682→            if parsed.get("pensamiento", "").strip():
   683→                return parsed["pensamiento"].strip()
   684→        # Si no es JSON completo, intentar limpiar restos
   685→        # Caso: texto que empieza con JSON parcial
   686→        import re as _re
   687→        # Remover JSON parcial al inicio: {"pensamiento": "..." ... restos
   688→        cleaned = _re.sub(r'^\s*\{[^}]*$', '', text).strip()
   689→        # Remover JSON parcial al final
   690→        cleaned = _re.sub(r'\{[^}]*$\s*$', '', cleaned).strip()
   691→        # Si despues de limpiar quedo vacio, devolver texto original
   692→        return cleaned if cleaned else text

   719→    # EJECUCION DE TOOLS (paralelo + retry)
   720→    # ----------------------------------------------------------
   721→    def _execute_tool_calls(self, tool_calls, messages):
   722→        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   723→        from concurrent.futures import ThreadPoolExecutor, as_completed
   724→
   725→        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   726→        filtered_calls = []
   727→        for tc in tool_calls:
   728→            tool_name = tc["name"]
   729→            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   730→            self._total_tool_calls += 1
   731→
   732→            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   733→                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   734→                break
   735→
   736→            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   737→                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   738→                continue
   739→
   740→            filtered_calls.append(tc)
   741→
   742→        tool_calls = filtered_calls
   743→
   744→        # Validar parametros de todos los tool calls primero
   745→        for tc in tool_calls:
   746→            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   747→
   748→        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   749→        if len(tool_calls) == 1:
   750→            return [self._execute_single_tool(tool_calls[0], messages)]
   751→
   752→        # Si hay multiples tools, ejecutar en paralelo
   753→        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   754→        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   755→        
   756→        if has_sequential:
   757→            # Ejecutar secuencialmente para evitar race conditions
   758→            results = []
   759→            for tc in tool_calls:
   760→                results.append(self._execute_single_tool(tc, messages))
   761→            return results
   762→
   763→        # Ejecucion paralela para tools de solo lectura
   764→        results = [None] * len(tool_calls)
   765→        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   766→            futures = {}
   767→            for i, tc in enumerate(tool_calls):
   768→                future = executor.submit(self._execute_single_tool, tc, messages)
   769→                futures[future] = i
   770→            
   771→            for future in as_completed(futures):
   772→                idx = futures[future]
   773→                try:
   774→                    results[idx] = future.result(timeout=30)
   775→                except Exception as e:
   776→                    results[idx] = f"ERROR: {e}"
   777→
   778→        return results

   780→    def _execute_single_tool(self, tc, messages, max_retries=1):
   781→        """Ejecuta un solo tool call con retry automatico."""
   782→        import time as _time
   783→        tool_name = tc["name"]
   784→        tool_params = tc["params"]
   785→
   786→        self._log(f"Ejecutando: {tool_name}({tool_params})", "execution")
   787→
   788→        _tool_start = _time.monotonic()
   789→        tool_result = self._execute_tool(tool_name, tool_params)
   790→
   791→        # Retry si fallo y el tool es reintentable
   792→        if "ERROR" in tool_result and max_retries > 0:
   793→            retryable_errors = ["TIMEOUT", "ConnectionError", "empty", "vacia"]
   794→            if any(err in tool_result for err in retryable_errors):
   795→                self._log(f"Reintentando {tool_name} (error transitorio)...", "execution")
   796→                tool_result = self._execute_tool(tool_name, tool_params)
   797→
   798→        _tool_elapsed_ms = (_time.monotonic() - _tool_start) * 1000.0
   799→        get_metrics().record_tool_call(tool_name, _tool_elapsed_ms)
   800→
   801→        if "ERROR" in tool_result:
   802→            get_metrics().record_error("tool:" + tool_name)
   803→
   804→        self._log(f"Resultado: {tool_result[:150]}...", "observation")
   805→        # Registrar en formato [TOOL] para _collect_tool_results (v14.7)
   806→        self._log(f"[TOOL] {tool_name}: {tool_result[:200]}", "tool_result")
   807→
   808→        # Alimentar memoria de trabajo
   809→        self.memory.add_step(f"{tool_name}({tool_params})", tool_result[:200])
   810→        if "ERROR" in tool_result:
   811→            self.memory.set_error(f"{tool_name}: {tool_result[:100]}")
   812→        if "PELIGROSO" in tool_result:
   813→            self._log(f"Comando peligroso bloqueado: {tool_name}", "warning")
   814→        if len(tool_result) > 50 and "ERROR" not in tool_result:
   815→            self.memory.remember(
   816→                f"Resultado de {tool_name}: {tool_result[:300]}",
   817→                metadata={"type": "tool_result", "tool": tool_name}
   818→            )
   819→
   820→        return tool_result

   965→            dirs = [d for d in os.listdir(REPOS_DIR)

   974→    def _parse_json(self, text):
   975→        try:
   976→            return json.loads(text)
   977→        except Exception:
   978→            pass
   979→        patterns = [
   980→            r'```json\s*(.*?)\s*```',
   981→            r'```\s*(.*?)\s*```',
   982→        ]
   983→        for pattern in patterns:
   984→            match = re.search(pattern, text, re.DOTALL)
   985→            if match:
   986→                try:
   987→                    return json.loads(match.group(1))
   988→                except Exception:
   989→                    continue
   990→        # Intentar parsear multiples JSONs en el texto
   991→        # Cada JSON es una accion separada del ReAct
   992→        jsons = self._extract_all_jsons(text)
   993→        if len(jsons) == 1:
   994→            return jsons[0]
   995→        elif len(jsons) > 1:
   996→            # Multiples JSONs = multiples tool calls
   997→            # Combinar en una lista de tool calls
   998→            tool_calls = []
   999→            for j in jsons:
  1000→                accion = j.get("accion", "").strip()
  1001→                if accion:
  1002→                    tool_calls.append({"name": accion, "params": j.get("params", {})})
  1003→            if tool_calls:
  1004→                # Retornar formato especial para multiples tool calls
  1005→                return {"_multi_tool_calls": True, "tool_calls": tool_calls}
  1006→            return jsons[0]  # Fallback al primer JSON
  1007→        # Ultimo intento: buscar un solo objeto JSON
  1008→        match = re.search(r'(\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\})', text, re.DOTALL)
  1009→        if match:
  1010→            try:
  1011→                return json.loads(match.group(1))
  1012→            except Exception:
  1013→                pass
  1014→        return None

  1016→    def _extract_all_jsons(self, text):
  1017→        """Extrae todos los objetos JSON validos de un texto."""
  1018→        results = []
  1019→        # Buscar secuencias que parezcan JSON
  1020→        i = 0
  1021→        while i < len(text):
  1022→            # Encontrar el proximo '{'
  1023→            start = text.find('{', i)
  1024→            if start == -1:
  1025→                break
  1026→            # Intentar parsear desde este punto
  1027→            depth = 0
  1028→            end = start
  1029→            for j in range(start, len(text)):
  1030→                if text[j] == '{':
  1031→                    depth += 1
  1032→                elif text[j] == '}':
  1033→                    depth -= 1
  1034→                if depth == 0:
  1035→                    end = j + 1
  1036→                    break
  1037→            if end > start:
  1038→                try:
  1039→                    parsed = json.loads(text[start:end])
  1040→                    if isinstance(parsed, dict):
  1041→                        results.append(parsed)
  1042→                except Exception:
  1043→                    pass
  1044→                i = end
  1045→            else:
  1046→                i = start + 1
  1047→        return results

  1049→    def _looks_like_tool_json(self, text):
  1050→        """Detecta si el texto parece ser JSON de tool call (no respuesta al usuario).
  1051→        v3: Mas robusto - detecta JSON que empieza despues de whitespace o texto corto.
  1052→        """
  1053→        text = text.strip()
  1054→        if not text:
  1055→            return False
  1056→        # Si el texto empieza con { probablemente es JSON de tool call
  1057→        if text.startswith('{'):
  1058→            return True
  1059→        # Si despues de whitespace/newline hay {, tambien es JSON
  1060→        # Esto captura casos donde el modelo genera: "\n{" o "  {"
  1061→        stripped = text.lstrip()
  1062→        if stripped.startswith('{'):
  1063→            return True
  1064→        # Si contiene patrones de tool calls en cualquier parte
  1065→        tool_indicators = ['"accion"', '"pensamiento"', '"params"', '"respuesta_final"',
  1066→                          '"abrir_', '"ejecutar_', '"leer_', '"escribir_', '"buscar_"']
  1067→        for indicator in tool_indicators:
  1068→            if indicator in text:
  1069→                return True
  1070→        # Si el texto es corto y parece inicio de JSON
  1071→        if len(text) < 5 and text in ['{', '"', '\n', '\r']:
  1072→            return True
  1073→        return False
