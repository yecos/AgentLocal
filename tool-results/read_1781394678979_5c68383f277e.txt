   245тЖТ   245тЖТdef _chart_bar(ax, data, opts, **kwargs):
   246тЖТ   246тЖТ    """Grafico de barras (horizontal/vertical, agrupado, apilado)."""
   247тЖТ   247тЖТ    labels = data.get("labels", [])
   248тЖТ   248тЖТ    values = data.get("values", [])
   249тЖТ   249тЖТ    if not values:
   250тЖТ   250тЖТ        return
   251тЖТ   251тЖТ
   252тЖТ   252тЖТ    horizontal = opts.get("horizontal", False)
   253тЖТ   253тЖТ    stacked = opts.get("stacked", False)
   254тЖТ   254тЖТ
   255тЖТ   255тЖТ    # Multiple series?
   256тЖТ   256тЖТ    if isinstance(values[0], list):
   257тЖТ   257тЖТ        # Datos multi-serie
   258тЖТ   258тЖТ        x = np.arange(len(labels))
   259тЖТ   259тЖТ        width = 0.8 / len(values)
   260тЖТ   260тЖТ        colors = opts.get("colors", CHART_COLORS)
   261тЖТ   261тЖТ        series_names = opts.get("series_names", [f"Serie {i+1}" for i in range(len(values))])
   262тЖТ   262тЖТ
   263тЖТ   263тЖТ        for i, serie in enumerate(values):
   264тЖТ   264тЖТ            offset = (i - len(values)/2 + 0.5) * width
   265тЖТ   265тЖТ            if stacked:
   266тЖТ   266тЖТ                if i == 0:
   267тЖТ   267тЖТ                    bottom = np.zeros(len(labels))
   268тЖТ   268тЖТ                else:
   269тЖТ   269тЖТ                    bottom = np.array(values[i-1]) if i == 1 else np.add.reduce(values[:i])
   270тЖТ   270тЖТ                ax.bar(x, serie, width, label=series_names[i],
   271тЖТ   271тЖТ                       color=colors[i % len(colors)], bottom=bottom)
   272тЖТ   272тЖТ            else:
   273тЖТ   273тЖТ                if horizontal:
   274тЖТ   274тЖТ                    ax.barh(x + offset, serie, width, label=series_names[i],
   275тЖТ   275тЖТ                            color=colors[i % len(colors)])
   276тЖТ   276тЖТ                else:
   277тЖТ   277тЖТ                    ax.bar(x + offset, serie, width, label=series_names[i],
   278тЖТ   278тЖТ                           color=colors[i % len(colors)])
   279тЖТ   279тЖТ        ax.set_xticks(x)
   280тЖТ   280тЖТ        ax.set_xticklabels(labels, rotation=45, ha='right')
   281тЖТ   281тЖТ        ax.legend(loc='best', fontsize=8)
   282тЖТ   282тЖТ    else:
   283тЖТ   283тЖТ        color = opts.get("color", PALETTE["primary"])
   284тЖТ   284тЖТ        if horizontal:
   285тЖТ   285тЖТ            ax.barh(labels, values, color=color)
   286тЖТ   286тЖТ        else:
   287тЖТ   287тЖТ            ax.bar(labels, values, color=color)
   288тЖТ   288тЖТ            plt.xticks(rotation=45, ha='right')

   303тЖТ   303тЖТ            marker = markers[i % len(markers)] if opts.get("show_markers", True) else None

   307тЖТ   307тЖТ                ax.fill_between(range(len(labels)), serie, alpha=0.1,
   308тЖТ   308тЖТ                                color=colors[i % len(colors)])
   309тЖТ   309тЖТ        ax.legend(loc='best', fontsize=8)
   310тЖТ   310тЖТ    else:
   311тЖТ   311тЖТ        color = opts.get("color", PALETTE["primary"])
   312тЖТ   312тЖТ        marker = 'o' if opts.get("show_markers", True) else None
   313тЖТ   313тЖТ        ax.plot(labels, values, marker=marker, color=color, linewidth=2, markersize=5)
   314тЖТ   314тЖТ        if opts.get("fill", False):
   315тЖТ   315тЖТ            ax.fill_between(range(len(labels)), values, alpha=0.1, color=color)

   613тЖТ   613тЖТdef _chart_regression(ax, data, opts, **kwargs):
   614тЖТ   614тЖТ    """Grafico de dispersion con linea de regresion."""
   615тЖТ   615тЖТ    import numpy as np
   616тЖТ   616тЖТ
   617тЖТ   617тЖТ    x_data = data.get("x", data.get("values", []))
   618тЖТ   618тЖТ    y_data = data.get("y", [])
   619тЖТ   619тЖТ
   620тЖТ   620тЖТ    if not y_data:
   621тЖТ   621тЖТ        if isinstance(x_data[0], (list, tuple)) and len(x_data[0]) >= 2:
   622тЖТ   622тЖТ            y_data = [p[1] for p in x_data]
   623тЖТ   623тЖТ            x_data = [p[0] for p in x_data]
   624тЖТ   624тЖТ        else:
   625тЖТ   625тЖТ            y_data = x_data
   626тЖТ   626тЖТ            x_data = list(range(len(y_data)))
   627тЖТ   627тЖТ
   628тЖТ   628тЖТ    x = np.array(x_data, dtype=float)
   629тЖТ   629тЖТ    y = np.array(y_data, dtype=float)
   630тЖТ   630тЖТ
   631тЖТ   631тЖТ    # Scatter
   632тЖТ   632тЖТ    color = opts.get("color", PALETTE["primary"])
   633тЖТ   633тЖТ    ax.scatter(x, y, s=40, c=color, alpha=0.6, edgecolors='white')
   634тЖТ   634тЖТ
   635тЖТ   635тЖТ    # Regresion lineal
   636тЖТ   636тЖТ    if len(x) > 1:
   637тЖТ   637тЖТ        z = np.polyfit(x, y, 1)
   638тЖТ   638тЖТ        p = np.poly1d(z)
   639тЖТ   639тЖТ        x_line = np.linspace(x.min(), x.max(), 100)
   640тЖТ   640тЖТ        ax.plot(x_line, p(x_line), color=PALETTE["danger"], linewidth=2,
   641тЖТ   641тЖТ                linestyle='--', label=f'y = {z[0]:.2f}x + {z[1]:.2f}')
   642тЖТ   642тЖТ
   643тЖТ   643тЖТ        # R-squared
   644тЖТ   644тЖТ        y_pred = p(x)
   645тЖТ   645тЖТ        ss_res = np.sum((y - y_pred) ** 2)
   646тЖТ   646тЖТ        ss_tot = np.sum((y - np.mean(y)) ** 2)
   647тЖТ   647тЖТ        r_squared = 1 - (ss_res / ss_tot) if ss_tot > 0 else 0
   648тЖТ   648тЖТ        ax.text(0.05, 0.95, f'R┬▓ = {r_squared:.3f}',
   649тЖТ   649тЖТ                transform=ax.transAxes, fontsize=10, va='top',
   650тЖТ   650тЖТ                bbox=dict(boxstyle='round', facecolor='wheat', alpha=0.5))
   651тЖТ   651тЖТ
   652тЖТ   652тЖТ        ax.legend(loc='best', fontsize=9)

   677тЖТ   677тЖТ        from scipy.stats import norm

   788тЖТ   788тЖТdef _parse_chart_data(datos_str: str) -> dict:
   789тЖТ   789тЖТ    """Parsea datos de grafico desde JSON o CSV a estructura estandar.
   790тЖТ   790тЖТ
   791тЖТ   791тЖТ    Formatos soportados:
   792тЖТ   792тЖТ    - JSON con keys: labels, values, x, y, matrix, candles, sizes, colors, row_labels, col_labels
   793тЖТ   793тЖТ    - CSV simple: etiqueta,valor por linea
   794тЖТ   794тЖТ    - Lista de puntos: x,y por linea
   795тЖТ   795тЖТ    """
   796тЖТ   796тЖТ    if not datos_str:
   797тЖТ   797тЖТ        return {"labels": [], "values": []}
   798тЖТ   798тЖТ
   799тЖТ   799тЖТ    datos_str = datos_str.strip()
   800тЖТ   800тЖТ
   801тЖТ   801тЖТ    # Intentar JSON
   802тЖТ   802тЖТ    if datos_str.startswith('{') or datos_str.startswith('['):
   803тЖТ   803тЖТ        try:
   804тЖТ   804тЖТ            parsed = json.loads(datos_str)
   805тЖТ   805тЖТ            if isinstance(parsed, dict):
   806тЖТ   806тЖТ                return parsed
   807тЖТ   807тЖТ            elif isinstance(parsed, list):
   808тЖТ   808тЖТ                if len(parsed) > 0 and isinstance(parsed[0], dict):
   809тЖТ   809тЖТ                    # Lista de dicts -> extraer labels y values
   810тЖТ   810тЖТ                    keys = list(parsed[0].keys())
   811тЖТ   811тЖТ                    return {
   812тЖТ   812тЖТ                        "labels": [str(d.get(keys[0], "")) for d in parsed],
   813тЖТ   813тЖТ                        "values": [d.get(keys[1], 0) if len(keys) > 1 else 0 for d in parsed],
   814тЖТ   814тЖТ                    }
   815тЖТ   815тЖТ                elif len(parsed) > 0 and isinstance(parsed[0], (list, tuple)):
   816тЖТ   816тЖТ                    return {"labels": [str(r[0]) for r in parsed],
   817тЖТ   817тЖТ                            "values": [r[1] if len(r) > 1 else 0 for r in parsed]}
   818тЖТ   818тЖТ                else:
   819тЖТ   819тЖТ                    return {"labels": [str(i) for i in range(len(parsed))],
   820тЖТ   820тЖТ                            "values": parsed}
   821тЖТ   821тЖТ        except json.JSONDecodeError:
   822тЖТ   822тЖТ            pass
   823тЖТ   823тЖТ
   824тЖТ   824тЖТ    # Fallback: CSV
   825тЖТ   825тЖТ    result = {"labels": [], "values": []}
   826тЖТ   826тЖТ    for line in datos_str.split("\n"):
   827тЖТ   827тЖТ        line = line.strip()
   828тЖТ   828тЖТ        if not line or line.startswith("#"):
   829тЖТ   829тЖТ            continue
   830тЖТ   830тЖТ
   831тЖТ   831тЖТ        # Intentar punto x,y
   832тЖТ   832тЖТ        if "," in line:
   833тЖТ   833тЖТ            parts = line.rsplit(",", 1)
   834тЖТ   834тЖТ            if len(parts) == 2:
   835тЖТ   835тЖТ                result["labels"].append(parts[0].strip())
   836тЖТ   836тЖТ                try:
   837тЖТ   837тЖТ                    result["values"].append(float(parts[1].strip()))
   838тЖТ   838тЖТ                except ValueError:
   839тЖТ   839тЖТ                    result["values"].append(0)
   840тЖТ   840тЖТ        else:
   841тЖТ   841тЖТ            try:
   842тЖТ   842тЖТ                result["values"].append(float(line))
   843тЖТ   843тЖТ            except ValueError:
   844тЖТ   844тЖТ                result["labels"].append(line)
   845тЖТ   845тЖТ
   846тЖТ   846тЖТ    return result
