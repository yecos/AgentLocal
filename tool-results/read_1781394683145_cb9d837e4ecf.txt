   245тЖТ   245тЖТdef _skewness(values):
   246тЖТ   246тЖТ    """Calcula asimetria (skewness) sin scipy."""
   247тЖТ   247тЖТ    import numpy as np
   248тЖТ   248тЖТ    n = len(values)
   249тЖТ   249тЖТ    if n < 3:
   250тЖТ   250тЖТ        return 0.0
   251тЖТ   251тЖТ    mean = np.mean(values)
   252тЖТ   252тЖТ    std = np.std(values, ddof=1)
   253тЖТ   253тЖТ    if std == 0:
   254тЖТ   254тЖТ        return 0.0
   255тЖТ   255тЖТ    return (1/n) * np.sum(((values - mean) / std) ** 3)
   256тЖТ   256тЖТ

   258тЖТ   258тЖТdef _kurtosis(values):
   259тЖТ   259тЖТ    """Calcula curtosis sin scipy."""
   260тЖТ   260тЖТ    import numpy as np
   261тЖТ   261тЖТ    n = len(values)
   262тЖТ   262тЖТ    if n < 4:
   263тЖТ   263тЖТ        return 0.0
   264тЖТ   264тЖТ    mean = np.mean(values)
   265тЖТ   265тЖТ    std = np.std(values, ddof=1)
   266тЖТ   266тЖТ    if std == 0:
   267тЖТ   267тЖТ        return 0.0
   268тЖТ   268тЖТ    k = (1/n) * np.sum(((values - mean) / std) ** 4) - 3
   269тЖТ   269тЖТ    return k

   300тЖТ   300тЖТdef tabla_pivote(datos: str, filas: str = "", columnas: str = "",
   301тЖТ   301тЖТ                 valores: str = "", funcion: str = "sum") -> str:
   302тЖТ   302тЖТ    """Crea una tabla pivote a partir de datos tabulares.
   303тЖТ   303тЖТ
   304тЖТ   304тЖТ    Args:
   305тЖТ   305тЖТ        datos: Datos en CSV o JSON
   306тЖТ   306тЖТ        filas: Columna para las filas de la pivot
   307тЖТ   307тЖТ        columnas: Columna para las columnas de la pivot
   308тЖТ   308тЖТ        valores: Columna de valores a agregar
   309тЖТ   309тЖТ        funcion: Funcion de agregacion: sum, mean, count, min, max
   310тЖТ   310тЖТ    """
   311тЖТ   311тЖТ    try:
   312тЖТ   312тЖТ        df = _load_dataframe(datos)
   313тЖТ   313тЖТ        if df is None:
   314тЖТ   314тЖТ            return "ERROR: No se pudieron parsear los datos."
   315тЖТ   315тЖТ
   316тЖТ   316тЖТ        if not filas:
   317тЖТ   317тЖТ            return "ERROR: Debes especificar al menos la columna de filas."
   318тЖТ   318тЖТ
   319тЖТ   319тЖТ        # Verificar columnas
   320тЖТ   320тЖТ        for col_name, col_label in [(filas, "filas"), (columnas, "columnas"), (valores, "valores")]:
   321тЖТ   321тЖТ            if col_name and col_name not in df.columns:
   322тЖТ   322тЖТ                return f"ERROR: Columna '{col_name}' no encontrada. Disponibles: {list(df.columns)}"
   323тЖТ   323тЖТ
   324тЖТ   324тЖТ        # Funcion de agregacion
   325тЖТ   325тЖТ        agg_map = {
   326тЖТ   326тЖТ            "sum": "sum", "suma": "sum",
   327тЖТ   327тЖТ            "mean": "mean", "promedio": "mean", "media": "mean",
   328тЖТ   328тЖТ            "count": "count", "conteo": "count",

   331тЖТ   331тЖТ            "std": "std", "desv": "std",
   332тЖТ   332тЖТ        }
   333тЖТ   333тЖТ        agg_func = agg_map.get(funcion.lower(), "sum")
   334тЖТ   334тЖТ
   335тЖТ   335тЖТ        # Crear pivot
   336тЖТ   336тЖТ        if columnas and valores:
   337тЖТ   337тЖТ            pivot = df.pivot_table(index=filas, columns=columnas, values=valores,

   339тЖТ   339тЖТ        elif valores:

   341тЖТ   341тЖТ        else:
   342тЖТ   342тЖТ            pivot = df.pivot_table(index=filas, aggfunc=agg_func)
   343тЖТ   343тЖТ
   344тЖТ   344тЖТ        # Formatear como tabla
   345тЖТ   345тЖТ        return f"Tabla Pivote ({funcion}):\n\n{pivot.to_string()}"
   346тЖТ   346тЖТ
   347тЖТ   347тЖТ    except ImportError:

   422тЖТ   422тЖТdef limpiar_datos(datos: str, operaciones: str = "todo") -> str:
   423тЖТ   423тЖТ    """Limpia un dataset aplicando operaciones de limpieza.
   424тЖТ   424тЖТ
   425тЖТ   425тЖТ    Args:
   426тЖТ   426тЖТ        datos: Datos en CSV o JSON
   427тЖТ   427тЖТ        operaciones: Operaciones separadas por coma: duplicados, nulos, outliers, normalizar, tipos, todo
   428тЖТ   428тЖТ    """
   429тЖТ   429тЖТ    try:
   430тЖТ   430тЖТ        df = _load_dataframe(datos)
   431тЖТ   431тЖТ        if df is None:
   432тЖТ   432тЖТ            return "ERROR: No se pudieron parsear los datos."
   433тЖТ   433тЖТ
   434тЖТ   434тЖТ        original_shape = df.shape
   435тЖТ   435тЖТ        ops = [op.strip().lower() for op in operaciones.split(",")]
   436тЖТ   436тЖТ        if "todo" in ops:
   437тЖТ   437тЖТ            ops = ["duplicados", "nulos", "outliers", "normalizar", "tipos"]

   439тЖТ   439тЖТ        report = [f"Limpieza de datos - Original: {original_shape[0]} filas, {original_shape[1]} columnas\n"]
   440тЖТ   440тЖТ
   441тЖТ   441тЖТ        # 1. Eliminar duplicados
   442тЖТ   442тЖТ        if "duplicados" in ops:
   443тЖТ   443тЖТ            before = len(df)
   444тЖТ   444тЖТ            df = df.drop_duplicates()
   445тЖТ   445тЖТ            removed = before - len(df)
   446тЖТ   446тЖТ            report.append(f"[duplicados] {removed} filas duplicadas eliminadas")
   447тЖТ   447тЖТ
   448тЖТ   448тЖТ        # 2. Tratar nulos
   449тЖТ   449тЖТ        if "nulos" in ops:
   450тЖТ   450тЖТ            null_counts = df.isnull().sum()
   451тЖТ   451тЖТ            total_nulls = null_counts.sum()
   452тЖТ   452тЖТ            if total_nulls > 0:
   453тЖТ   453тЖТ                # Para numericos: rellenar con mediana
   454тЖТ   454тЖТ                for col in df.select_dtypes(include='number').columns:
   455тЖТ   455тЖТ                    if df[col].isnull().any():
   456тЖТ   456тЖТ                        median = df[col].median()
   457тЖТ   457тЖТ                        df[col].fillna(median, inplace=True)
   458тЖТ   458тЖТ                # Para categoricos: rellenar con moda
   459тЖТ   459тЖТ                for col in df.select_dtypes(exclude='number').columns:
   460тЖТ   460тЖТ                    if df[col].isnull().any():
   461тЖТ   461тЖТ                        mode = df[col].mode().iloc[0] if len(df[col].mode()) > 0 else "N/A"
   462тЖТ   462тЖТ                        df[col].fillna(mode, inplace=True)
   463тЖТ   463тЖТ                report.append(f"[nulos] {total_nulls} valores nulos tratados (mediana/moda)")
   464тЖТ   464тЖТ            else:
   465тЖТ   465тЖТ                report.append(f"[nulos] Sin valores nulos encontrados")
   466тЖТ   466тЖТ
   467тЖТ   467тЖТ        # 3. Eliminar outliers (IQR)
   468тЖТ   468тЖТ        if "outliers" in ops:
   469тЖТ   469тЖТ            total_removed = 0
   470тЖТ   470тЖТ            for col in df.select_dtypes(include='number').columns:
   471тЖТ   471тЖТ                Q1 = df[col].quantile(0.25)
   472тЖТ   472тЖТ                Q3 = df[col].quantile(0.75)
   473тЖТ   473тЖТ                IQR = Q3 - Q1
   474тЖТ   474тЖТ                lower = Q1 - 1.5 * IQR
   475тЖТ   475тЖТ                upper = Q3 + 1.5 * IQR
   476тЖТ   476тЖТ                outliers = ((df[col] < lower) | (df[col] > upper)).sum()
   477тЖТ   477тЖТ                if outliers > 0:
   478тЖТ   478тЖТ                    df = df[(df[col] >= lower) & (df[col] <= upper)]
   479тЖТ   479тЖТ                    total_removed += outliers
   480тЖТ   480тЖТ            report.append(f"[outliers] {total_removed} valores atipicos eliminados (IQR)")
   481тЖТ   481тЖТ
   482тЖТ   482тЖТ        # 4. Normalizar (min-max)
   483тЖТ   483тЖТ        if "normalizar" in ops:
   484тЖТ   484тЖТ            for col in df.select_dtypes(include='number').columns:
   485тЖТ   485тЖТ                col_min = df[col].min()
   486тЖТ   486тЖТ                col_max = df[col].max()
   487тЖТ   487тЖТ                if col_max > col_min:
   488тЖТ   488тЖТ                    df[col] = (df[col] - col_min) / (col_max - col_min)
   489тЖТ   489тЖТ            report.append(f"[normalizar] Columnas numericas normalizadas a [0,1]")
   490тЖТ   490тЖТ
   491тЖТ   491тЖТ        # 5. Detectar y convertir tipos
   492тЖТ   492тЖТ        if "tipos" in ops:
   493тЖТ   493тЖТ            converted = []
   494тЖТ   494тЖТ            for col in df.columns:
   495тЖТ   495тЖТ                if df[col].dtype == 'object':
   496тЖТ   496тЖТ                    try:
   497тЖТ   497тЖТ                        df[col] = df[col].astype(float)
   498тЖТ   498тЖТ                        converted.append(f"{col}: texto -> numerico")
   499тЖТ   499тЖТ                    except (ValueError, TypeError):
   500тЖТ   500тЖТ                        pass
   501тЖТ   501тЖТ            if converted:
   502тЖТ   502тЖТ                report.append(f"[tipos] Conversiones: {', '.join(converted)}")
   503тЖТ   503тЖТ            else:
   504тЖТ   504тЖТ                report.append(f"[tipos] Sin conversiones necesarias")
   505тЖТ   505тЖТ
   506тЖТ   506тЖТ        report.append(f"\nResultado: {df.shape[0]} filas, {df.shape[1]} columnas")
   507тЖТ   507тЖТ        report.append(f"Filas eliminadas: {original_shape[0] - df.shape[0]}")
   508тЖТ   508тЖТ
   509тЖТ   509тЖТ        # Preview
   510тЖТ   510тЖТ        report.append(f"\nPreview (5 filas):\n{df.head().to_string()}")
   511тЖТ   511тЖТ
   512тЖТ   512тЖТ        return "\n".join(report)
   513тЖТ   513тЖТ
   514тЖТ   514тЖТ    except ImportError:
   515тЖТ   515тЖТ        return "ERROR: pandas no instalado. Instala: pip install pandas"
   516тЖТ   516тЖТ    except Exception as e:
   517тЖТ   517тЖТ        return f"ERROR limpiando datos: {e}"

   524тЖТ   524тЖТdef transformar_datos(datos: str, operacion: str = "", parametros: str = "{}") -> str:
   525тЖТ   525тЖТ    """Transforma un dataset aplicando operaciones de filtrado, ordenamiento, agrupacion, etc.
   526тЖТ   526тЖТ
   527тЖТ   527тЖТ    Args:
   528тЖТ   528тЖТ        datos: Datos en CSV o JSON
   529тЖТ   529тЖТ        operacion: Operacion a realizar: filtrar, ordenar, agrupar, seleccionar, renombrar, agregar_columna, head, sample
   530тЖТ   530тЖТ        parametros: Parametros de la operacion en JSON
   531тЖТ   531тЖТ    """
   532тЖТ   532тЖТ    try:
   533тЖТ   533тЖТ        df = _load_dataframe(datos)
   534тЖТ   534тЖТ        if df is None:
   535тЖТ   535тЖТ            return "ERROR: No se pudieron parsear los datos."
   536тЖТ   536тЖТ
   537тЖТ   537тЖТ        params = _parse_json(parametros, {})
   538тЖТ   538тЖТ        op = operacion.lower().strip()
   539тЖТ   539тЖТ

   541тЖТ   541тЖТ            columna = params.get("columna", params.get("column", ""))
   542тЖТ   542тЖТ            condicion = params.get("condicion", params.get("condition", ""))
   543тЖТ   543тЖТ            valor = params.get("valor", params.get("value", ""))

   546тЖТ   546тЖТ                return f"ERROR: Columna '{columna}' no encontrada."
   547тЖТ   547тЖТ
   548тЖТ   548тЖТ            if condicion in (">", "mayor"):

   550тЖТ   550тЖТ            elif condicion in ("<", "menor"):
   551тЖТ   551тЖТ                df = df[df[columna] < float(valor)]
   552тЖТ   552тЖТ            elif condicion in ("==", "igual"):

   554тЖТ   554тЖТ            elif condicion in ("!=", "diferente"):

   556тЖТ   556тЖТ            elif condicion in (">=", "mayor_igual"):
   557тЖТ   557тЖТ                df = df[df[columna] >= float(valor)]
   558тЖТ   558тЖТ            elif condicion in ("<=", "menor_igual"):
   559тЖТ   559тЖТ                df = df[df[columna] <= float(valor)]
   560тЖТ   560тЖТ            elif condicion in ("contains", "contiene"):
   561тЖТ   561тЖТ                df = df[df[columna].astype(str).str.contains(str(valor), case=False)]
   562тЖТ   562тЖТ            else:
   563тЖТ   563тЖТ                return f"ERROR: Condicion '{condicion}' no reconocida. Usar: >, <, ==, !=, >=, <=, contiene"
   564тЖТ   564тЖТ
   565тЖТ   565тЖТ        elif op == "ordenar" or op == "sort":
   566тЖТ   566тЖТ            columna = params.get("columna", params.get("column", ""))
   567тЖТ   567тЖТ            ascendente = params.get("ascendente", params.get("ascending", True))
   568тЖТ   568тЖТ
   569тЖТ   569тЖТ            if not columna:
   570тЖТ   570тЖТ                return "ERROR: Especifica columna para ordenar."

   573тЖТ   573тЖТ
   574тЖТ   574тЖТ            df = df.sort_values(by=columna, ascending=ascendente)
   575тЖТ   575тЖТ
   576тЖТ   576тЖТ        elif op == "agrupar" or op == "groupby":
   577тЖТ   577тЖТ            columna = params.get("columna", params.get("column", ""))

   595тЖТ   595тЖТ                return f"ERROR: Columnas no encontradas: {missing}"

   598тЖТ   598тЖТ
   599тЖТ   599тЖТ        elif op == "renombrar" or op == "rename":
   600тЖТ   600тЖТ            mapeo = params.get("mapeo", params.get("mapping", {}))

   602тЖТ   602тЖТ
   603тЖТ   603тЖТ        elif op == "agregar_columna" or op == "add_column":
   604тЖТ   604тЖТ            nombre = params.get("nombre", params.get("name", "nueva"))
   605тЖТ   605тЖТ            formula = params.get("formula", "")
   606тЖТ   606тЖТ
   607тЖТ   607тЖТ            if not formula:
   608тЖТ   608тЖТ                return "ERROR: Especifica formula para la nueva columna."
   609тЖТ   609тЖТ
   610тЖТ   610тЖТ            # Evaluar formula de forma segura
   611тЖТ   611тЖТ            try:
   612тЖТ   612тЖТ                df[nombre] = df.eval(formula)
   613тЖТ   613тЖТ            except Exception as e:
   614тЖТ   614тЖТ                return f"ERROR evaluando formula: {e}"
   615тЖТ   615тЖТ
   616тЖТ   616тЖТ        elif op == "head":
   617тЖТ   617тЖТ            n = int(params.get("n", 10))
   618тЖТ   618тЖТ            df = df.head(n)
   619тЖТ   619тЖТ
   620тЖТ   620тЖТ        elif op == "sample":
   621тЖТ   621тЖТ            n = int(params.get("n", 5))
   622тЖТ   622тЖТ            df = df.sample(min(n, len(df)))

   624тЖТ   624тЖТ        else:
   625тЖТ   625тЖТ            return (f"ERROR: Operacion '{op}' no reconocida.\n"

   627тЖТ   627тЖТ
   628тЖТ   628тЖТ        # Formatear resultado
   629тЖТ   629тЖТ        result = f"Transformacion [{op}]:\n{df.to_string()}"
   630тЖТ   630тЖТ        if len(result) > MAX_TOOL_OUTPUT * 5:
   631тЖТ   631тЖТ            result = result[:MAX_TOOL_OUTPUT * 5] + "\n... [truncado]"
   632тЖТ   632тЖТ        return result
   633тЖТ   633тЖТ
   634тЖТ   634тЖТ    except ImportError:
   635тЖТ   635тЖТ        return "ERROR: pandas no instalado. Instala: pip install pandas"
   636тЖТ   636тЖТ    except Exception as e:
   637тЖТ   637тЖТ        return f"ERROR transformando datos: {e}"
