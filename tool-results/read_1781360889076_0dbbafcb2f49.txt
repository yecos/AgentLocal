    34→
    35→class ChromaVectorStore:
    36→    """
    37→    Vector store basado en ChromaDB con decaimiento temporal y deduplicacion.
    38→    Escalable a miles de documentos con busqueda semantica rapida.
    39→    v3: Auto-deteccion robusta y recreacion ante mismatch de dimensiones de embeddings.
    40→        Catch-all de errores de dimension para que NUNCA crashee el agente.
    41→    """
    42→
    43→    DECAY_HALF_LIFE_DAYS = 30  # Los recuerdos pierden la mitad de relevancia en 30 dias
    44→    MAX_RECREATE_RETRIES = 2   # Max reintentos al recrear coleccion por dimension mismatch
    45→
    46→    def __init__(self, store_dir=None):
    47→        self.store_dir = store_dir or os.path.join(LEARN_DIR, "vectors")
    48→        os.makedirs(self.store_dir, exist_ok=True)
    49→
    50→        # Detectar dimension del modelo de embedding actual
    51→        self._embedding_dim = self._detect_embedding_dimension()
    52→
    53→        # Inicializar ChromaDB
    54→        self._client = chromadb.PersistentClient(path=self.store_dir)
    55→
    56→        # Obtener o crear coleccion con validacion de dimensiones
    57→        self._collection = self._get_or_create_collection()
    58→
    59→        # Validacion post-init: verificar que las dimensiones coinciden
    60→        self._validate_collection_dimension()
    61→
    62→        self._index_meta = self._load_meta()
    63→        logger.info(f"ChromaDB inicializado: {self._collection.count()} documentos (dim={self._embedding_dim})")
    64→
    65→    def _detect_embedding_dimension(self):
    66→        """Detecta la dimension de los embeddings del modelo actual."""
    67→        try:
    68→            test_embedding = ollama.get_embedding("test")
    69→            if test_embedding and isinstance(test_embedding, list):
    70→                return len(test_embedding)
    71→        except Exception as e:
    72→            logger.debug(f"No se pudo detectar dimension del embedding: {e}")
    73→        return None
    74→
    75→    def _get_collection_dimension(self, collection):
    76→        """Intenta obtener la dimension de embeddings de una coleccion existente."""
    77→        try:
    78→            metadata = collection.metadata
    79→            if metadata and "hnsw:dim" in metadata:
    80→                return int(metadata["hnsw:dim"])
    81→        except Exception:
    82→            pass
    83→        try:
    84→            peek_result = collection.peek(limit=1)
    85→            if peek_result and peek_result.get("embeddings") and len(peek_result["embeddings"]) > 0:
    86→                return len(peek_result["embeddings"][0])
    87→        except Exception:
    88→            pass
    89→        return None
    90→
    91→    def _validate_collection_dimension(self):
    92→        """Validacion post-inicializacion: prueba insertar y buscar un vector de prueba.
    93→        Si falla por dimension mismatch, recrea la coleccion automaticamente.
    94→        Esto atrapa casos donde _get_or_create_collection no detecto el mismatch.
    95→        """
    96→        if self._embedding_dim is None:
    97→            return
    98→        try:
    99→            # Intentar una operacion de query con un vector de la dimension actual
   100→            test_vec = [0.0] * self._embedding_dim
   101→            self._collection.query(
   102→                query_embeddings=[test_vec],
   103→                n_results=1,
   104→                include=["distances"]
   105→            )
   106→            logger.debug("Validacion de dimensiones ChromaDB: OK")
   107→        except Exception as e:
   108→            err_msg = str(e).lower()
   109→            if "dimension" in err_msg or "dim" in err_msg:
   110→                logger.warning(
   111→                    f"⚠️ MISMATCH DETECTADO EN VALIDACION POST-INIT: {e}. "
   112→                    f"Recreando coleccion con dim={self._embedding_dim}..."
   113→                )
   114→                self._force_recreate_collection()
   115→            else:
   116→                logger.debug(f"Validacion ChromaDB fallo (no es dimension): {e}")
   117→
   118→    def _force_recreate_collection(self):
   119→        """Fuerza la recreacion de la coleccion, eliminando datos previos."""
   120→        try:
   121→            self._client.delete_collection(name="agent_memory")
   122→            logger.info("Coleccion 'agent_memory' eliminada para recreacion")
   123→        except Exception:
   124→            pass
   125→        metadata = {"hnsw:space": "cosine"}
   126→        if self._embedding_dim is not None:
   127→            metadata["hnsw:dim"] = self._embedding_dim
   128→        self._collection = self._client.create_collection(name="agent_memory", metadata=metadata)
   129→        logger.info(f"Coleccion recreada con dim={self._embedding_dim}")
   130→
   131→    def _get_or_create_collection(self):
   132→        """Obtiene o crea la coleccion, manejando mismatch de dimensiones."""
   133→        try:
   134→            existing = self._client.get_collection(name="agent_memory")
   135→
   136→            # Verificar dimensiones si las conocemos
   137→            if self._embedding_dim is not None:
   138→                existing_dim = self._get_collection_dimension(existing)
   139→
   140→                if existing_dim is not None and existing_dim != self._embedding_dim:
   141→                    logger.warning(
   142→                        f"⚠️ MISMATCH DE DIMENSIONES: coleccion espera {existing_dim} dim, "
   143→                        f"modelo actual produce {self._embedding_dim} dim. Recreando coleccion..."
   144→                    )
   145→                    self._client.delete_collection(name="agent_memory")
   146→                    logger.info(f"Coleccion eliminada. Creando nueva con dim={self._embedding_dim}")
   147→                    return self._client.create_collection(
   148→                        name="agent_memory",
   149→                        metadata={"hnsw:space": "cosine", "hnsw:dim": self._embedding_dim}
   150→                    )
   151→
   152→            return existing
   153→
   154→        except Exception:
   155→            # La coleccion no existe — crear nueva
   156→            metadata = {"hnsw:space": "cosine"}
   157→            if self._embedding_dim is not None:
   158→                metadata["hnsw:dim"] = self._embedding_dim
   159→            return self._client.create_collection(name="agent_memory", metadata=metadata)

   161→    def _load_meta(self):
   162→        """Carga metadatos adicionales (timestamp para decaimiento)."""
   163→        meta_file = os.path.join(self.store_dir, "meta.json")
   164→        try:
   165→            if os.path.exists(meta_file):
   166→                with open(meta_file, "r", encoding="utf-8") as f:
   167→                    return json.load(f)
   168→        except Exception as e:
   169→            logger.debug(f"Error cargando metadatos: {e}")
   170→        return {}
   171→
   172→    def _save_meta(self):
   173→        """Guarda metadatos adicionales."""
   174→        meta_file = os.path.join(self.store_dir, "meta.json")
   175→        try:
   176→            with open(meta_file, "w", encoding="utf-8") as f:
   177→                json.dump(self._index_meta, f, ensure_ascii=False, indent=2)
   178→        except Exception as e:
   179→            logger.debug(f"Error guardando metadatos: {e}")

   181→    def _compute_decay(self, created_at, now=None):
   182→        """
   183→        Computa el factor de decaimiento temporal.
   184→        Los recuerdos recientes pesan mas que los viejos.
   185→        Retorna un valor entre 0 y 1.
   186→        """
   187→        if not created_at:
   188→            return 0.5
   189→        try:
   190→            created = datetime.fromisoformat(created_at)
   191→            if now is None:
   192→                now = datetime.now()
   193→            days_old = (now - created).total_seconds() / 86400
   194→            # Decaimiento exponencial
   195→            import math
   196→            decay = math.exp(-0.693 * days_old / self.DECAY_HALF_LIFE_DAYS)
   197→            return max(decay, 0.1)  # Minimo 10% de relevancia
   198→        except Exception as e:
   199→            logger.debug(f"Error computando decaimiento: {e}")
   200→            return 0.5

   202→    def _is_duplicate(self, text, threshold=0.95):
   203→        """
   204→        Verifica si un texto es duplicado semantico de uno existente.
   205→        Usa embeddings para comparar.
   206→        v3: Catch de errores de dimension para no crashear.
   207→        """
   208→        embedding = ollama.get_embedding(text)
   209→        if not embedding:
   210→            return False
   211→
   212→        # Verificar dimension antes de consultar
   213→        if self._embedding_dim is not None and len(embedding) != self._embedding_dim:
   214→            logger.debug(f"Dimension mismatch en _is_duplicate, saltando verificacion")
   215→            return False
   216→
   217→        try:
   218→            results = self._collection.query(
   219→                query_embeddings=[embedding],
   220→                n_results=1,
   221→                include=["distances"]
   222→            )
   223→            if results and results["distances"] and results["distances"][0]:
   224→                # ChromaDB con cosine space: distance < (1 - threshold) = duplicado
   225→                distance = results["distances"][0][0]
   226→                similarity = 1 - distance
   227→                return similarity >= threshold
   228→        except Exception as e:
   229→            err_msg = str(e).lower()
   230→            if "dimension" in err_msg or "dim" in err_msg:
   231→                logger.warning(f"Dimension mismatch en _is_duplicate, recreando coleccion...")
   232→                try:
   233→                    self._handle_dimension_error(embedding)
   234→                except Exception:
   235→                    pass
   236→            else:
   237→                logger.debug(f"Error verificando duplicado semantico: {e}")
   238→        return False

   240→    def add(self, text, metadata=None, entry_id=None, skip_embedding=False):
   241→        """Agrega un texto al vector store con su embedding.
   242→        
   243→        Args:
   244→            skip_embedding: Si True, NO calcula embedding (mas rapido).
   245→                           La entrada solo aparecera en busquedas por texto.
   246→        v3: Catch-all de errores de dimension con reintento automatico.
   247→        """
   248→        if not entry_id:
   249→            entry_id = hashlib.md5(text.encode()).hexdigest()[:12]
   250→
   251→        # Verificar duplicado existente en ChromaDB
   252→        try:
   253→            existing = self._collection.get(ids=[entry_id])
   254→            if existing and existing["ids"]:
   255→                return entry_id
   256→        except Exception as e:
   257→            logger.debug(f"Error verificando entrada existente en ChromaDB: {e}")
   258→
   259→        # Verificar duplicado semantico (solo si tenemos embedding y no skip)
   260→        if not skip_embedding:
   261→            try:
   262→                if self._is_duplicate(text):
   263→                    logger.debug(f"Texto duplicado semantico, saltando: {text[:50]}...")
   264→                    return entry_id
   265→            except Exception as e:
   266→                logger.debug(f"Error verificando duplicado semantico: {e}")
   267→
   268→        now = datetime.now().isoformat()
   269→
   270→        # Metadatos con timestamp para decaimiento
   271→        meta = metadata or {}
   272→        meta["created"] = now
   273→        meta["text_preview"] = text[:100]
   274→
   275→        # Si skip_embedding, guardar solo con metadatos (sin calculo de embedding)
   276→        if skip_embedding:
   277→            meta["no_embedding"] = True
   278→            try:
   279→                self._collection.add(
   280→                    ids=[entry_id],
   281→                    documents=[text[:500]],
   282→                    metadatas=[meta]
   283→                )
   284→            except Exception as e:
   285→                logger.debug(f"Error agregando sin embedding a ChromaDB: {e}")
   286→            return entry_id
   287→
   288→        # Obtener embedding
   289→        embedding = ollama.get_embedding(text)
   290→
   291→        if embedding:
   292→            # Actualizar dimension si es la primera vez que obtenemos un embedding valido
   293→            if self._embedding_dim is None:
   294→                self._embedding_dim = len(embedding)
   295→                logger.info(f"Dimension de embedding detectada dinamicamente: {self._embedding_dim}")
   296→
   297→            # Validar dimension del embedding antes de insertar
   298→            if len(embedding) != self._embedding_dim:
   299→                logger.warning(
   300→                    f"Embedding con dimension incorrecta ({len(embedding)} vs {self._embedding_dim}). "
   301→                    f"Actualizando dimension y recreando coleccion..."
   302→                )
   303→                self._handle_dimension_error(embedding)
   304→
   305→            # Intentar insertar con catch-all de errores de dimension
   306→            for attempt in range(self.MAX_RECREATE_RETRIES + 1):
   307→                try:
   308→                    self._collection.add(
   309→                        ids=[entry_id],
   310→                        embeddings=[embedding],
   311→                        documents=[text[:500]],
   312→                        metadatas=[meta]
   313→                    )
   314→                    return entry_id
   315→                except Exception as e:
   316→                    err_msg = str(e).lower()
   317→                    if "dimension" in err_msg or "dim" in err_msg:
   318→                        if attempt < self.MAX_RECREATE_RETRIES:
   319→                            logger.warning(
   320→                                f"Error de dimension al insertar (intento {attempt + 1}). "
   321→                                f"Recreando coleccion..."
   322→                            )
   323→                            self._handle_dimension_error(embedding)
   324→                        else:
   325→                            logger.error(
   326→                                f"No se pudo resolver error de dimension tras {attempt + 1} intentos. "
   327→                                f"Guardando sin embedding."
   328→                            )
   329→                            # Fallback: guardar sin embedding para no perder el dato
   330→                            meta["no_embedding"] = True
   331→                            try:
   332→                                self._collection.add(
   333→                                    ids=[entry_id],
   334→                                    documents=[text[:500]],
   335→                                    metadatas=[meta]
   336→                                )
   337→                            except Exception:
   338→                                pass
   339→                            return entry_id
   340→                    else:
   341→                        logger.debug(f"Error al insertar en ChromaDB (no dimension): {e}")
   342→                        return entry_id
   343→        else:
   344→            # Sin embedding, guardar solo con metadatos
   345→            meta["no_embedding"] = True
   346→            try:
   347→                self._collection.add(
   348→                    ids=[entry_id],
   349→                    documents=[text[:500]],
   350→                    metadatas=[meta]
   351→                )
   352→            except Exception as e:
   353→                logger.debug(f"Error agregando sin embedding a ChromaDB: {e}")
   354→
   355→        return entry_id

   357→    def _handle_dimension_error(self, sample_embedding=None):
   358→        """Maneja error de dimension recreando la coleccion.
   359→        v3: Actualiza _embedding_dim con la dimension real del embedding recibido.
   360→        """
   361→        new_dim = len(sample_embedding) if sample_embedding else self._embedding_dim
   362→        if new_dim:
   363→            self._embedding_dim = new_dim
   364→        logger.warning(
   365→            f"Recreando coleccion 'agent_memory' con dim={new_dim} "
   366→            f"(datos previos se pierden para evitar errores de dimension)"
   367→        )
   368→        try:
   369→            self._client.delete_collection(name="agent_memory")
   370→        except Exception:
   371→            pass
   372→        metadata = {"hnsw:space": "cosine"}
   373→        if new_dim:
   374→            metadata["hnsw:dim"] = new_dim
   375→        try:
   376→            self._collection = self._client.create_collection(name="agent_memory", metadata=metadata)
   377→            logger.info(f"Coleccion recreada exitosamente con dim={new_dim}")
   378→        except Exception as e:
   379→            logger.error(f"Error recreando coleccion: {e}")

   383→        existing_dim = self._get_collection_dimension(self._collection)
   384→        return {

   386→            "collection_dimension": existing_dim,
   387→            "model_dimension": self._embedding_dim,
   388→            "dimensions_match": existing_dim == self._embedding_dim if existing_dim and self._embedding_dim else "unknown",
   389→            "persist_dir": self.store_dir,

   392→    def search(self, query, limit=5, min_similarity=0.3):
   393→        """Busca entradas semanticamente similares con decaimiento temporal.
   394→        v3: Catch-all de errores de dimension con fallback a busqueda por texto.
   395→        """
   396→        query_embedding = ollama.get_embedding(query)
   397→        if not query_embedding:
   398→            return self._text_search(query, limit)
   399→
   400→        # Actualizar dimension si es la primera vez que obtenemos un embedding valido
   401→        if self._embedding_dim is None:
   402→            self._embedding_dim = len(query_embedding)
   403→            logger.info(f"Dimension de embedding detectada dinamicamente: {self._embedding_dim}")
   404→
   405→        # Validar dimension del embedding antes de buscar
   406→        if len(query_embedding) != self._embedding_dim:
   407→            logger.warning(
   408→                f"Dimension mismatch en search: {len(query_embedding)} vs {self._embedding_dim}. "
   409→                f"Actualizando dimension y recreando coleccion..."
   410→            )
   411→            try:
   412→                self._handle_dimension_error(query_embedding)
   413→            except Exception as e:
   414→                logger.warning(f"Error recreando coleccion: {e}")
   415→            return self._text_search(query, limit)
   416→
   417→        try:
   418→            # Buscar mas candidatos de los necesarios para re-rankear con decaimiento
   419→            n_candidates = min(limit * 3, 20)
   420→            results = self._collection.query(
   421→                query_embeddings=[query_embedding],
   422→                n_results=n_candidates,
   423→                include=["documents", "metadatas", "distances"]
   424→            )
   425→
   426→            if not results or not results["ids"] or not results["ids"][0]:
   427→                return self._text_search(query, limit)
   428→
   429→            scored = []
   430→            now = datetime.now()
   431→            for i, doc_id in enumerate(results["ids"][0]):
   432→                distance = results["distances"][0][i]
   433→                similarity = 1 - distance
   434→
   435→                # Aplicar decaimiento temporal
   436→                meta = results["metadatas"][0][i] if results["metadatas"] else {}
   437→                created_at = meta.get("created", "")
   438→                decay = self._compute_decay(created_at, now)
   439→
   440→                # Score final = similitud * decaimiento
   441→                final_score = similarity * decay
   442→
   443→                if final_score >= min_similarity * 0.5:  # Umbral mas bajo tras decaimiento
   444→                    scored.append({
   445→                        "id": doc_id,
   446→                        "text": results["documents"][0][i] if results["documents"] else "",
   447→                        "metadata": meta,
   448→                        "has_vector": True,
   449→                        "score": round(final_score, 3),
   450→                        "raw_similarity": round(similarity, 3),
   451→                        "decay": round(decay, 3),
   452→                    })
   453→
   454→            scored.sort(key=lambda x: x["score"], reverse=True)
   455→            return scored[:limit]
   456→
   457→        except Exception as e:
   458→            err_msg = str(e).lower()
   459→            if "dimension" in err_msg or "dim" in err_msg:
   460→                logger.warning(f"Dimension mismatch en query, recreando coleccion...")
   461→                try:
   462→                    self._handle_dimension_error(query_embedding)
   463→                except Exception:
   464→                    pass
   465→            else:
   466→                logger.warning(f"Error en busqueda ChromaDB: {e}")
   467→            return self._text_search(query, limit)
   468→
   469→    def _text_search(self, query, limit=5):
   470→        """Busqueda por texto cuando no hay embeddings."""
   471→        query_lower = query.lower()
   472→        query_words = [w for w in query_lower.split() if len(w) > 3]
   473→
   474→        if not query_words:
   475→            return []
   476→
   477→        try:
   478→            # ChromaDB no tiene text search nativo, obtener todos y filtrar
   479→            all_docs = self._collection.get(include=["documents", "metadatas"])
   480→            results = []
   481→
   482→            for i, doc in enumerate(all_docs["documents"]):
   483→                doc_lower = doc.lower()
   484→                matches = sum(1 for w in query_words if w in doc_lower)
   485→                if matches > 0:
   486→                    score = matches / max(len(query_words), 1)
   487→                    meta = all_docs["metadatas"][i] if all_docs["metadatas"] else {}
   488→                    results.append({
   489→                        "id": all_docs["ids"][i],
   490→                        "text": doc,
   491→                        "metadata": meta,
   492→                        "has_vector": not meta.get("no_embedding", False),
   493→                        "score": round(score, 3),
   494→                    })
   495→
   496→            results.sort(key=lambda x: x["score"], reverse=True)
   497→            return results[:limit]
   498→        except Exception as e:
   499→            logger.debug(f"Error en busqueda por texto ChromaDB: {e}")
   500→            return []

   502→    def count(self):
   503→        """Retorna el numero de documentos en el store."""
   504→        try:
   505→            return self._collection.count()
   506→        except Exception as e:
   507→            logger.warning(f"Error contando documentos ChromaDB: {e}")
   508→            return 0

   510→    def cleanup(self, max_entries=1000):
   511→        """Limpia entradas viejas si hay demasiadas."""
   512→        current = self.count()
   513→        if current <= max_entries:
   514→            return
   515→
   516→        try:
   517→            # Obtener todos los documentos con metadatos
   518→            all_docs = self._collection.get(include=["metadatas"])
   519→
   520→            # Ordenar por fecha de creacion
   521→            entries_with_dates = []
   522→            for i, entry_id in enumerate(all_docs["ids"]):
   523→                meta = all_docs["metadatas"][i] if all_docs["metadatas"] else {}
   524→                created = meta.get("created", "2000-01-01")
   525→                entries_with_dates.append((entry_id, created))
   526→
   527→            entries_with_dates.sort(key=lambda x: x[1])
   528→
   529→            # Eliminar los mas viejos
   530→            to_remove = entries_with_dates[:current - max_entries]
   531→            if to_remove:
   532→                ids_to_remove = [e[0] for e in to_remove]
   533→                self._collection.delete(ids=ids_to_remove)
   534→                logger.info(f"Cleanup: eliminados {len(ids_to_remove)} documentos viejos")
   535→        except Exception as e:
   536→            logger.warning(f"Error en cleanup: {e}")

   539→class SimpleVectorStore(VectorStore):
   540→    """
   541→    Vector store casero (fallback cuando ChromaDB no esta instalado).
   542→    Hereda de VectorStore y agrega decaimiento temporal + deduplicacion por texto.
   543→    Solo sobreescribe los metodos que difieren del base.

   545→    """

   547→    DECAY_HALF_LIFE_DAYS = 30
   548→
   549→    def _compute_decay(self, created_at):
   550→        """Computa el factor de decaimiento temporal (0.1 - 1.0)."""
   551→        if not created_at:
   552→            return 0.5
   553→        try:
   554→            created = datetime.fromisoformat(created_at)
   555→            days_old = (datetime.now() - created).total_seconds() / 86400
   556→            import math
   557→            decay = math.exp(-0.693 * days_old / self.DECAY_HALF_LIFE_DAYS)
   558→            return max(decay, 0.1)
   559→        except Exception as e:
   560→            logger.debug(f"Error computando decaimiento: {e}")
   561→            return 0.5
   562→
   563→    def add(self, text, metadata=None, entry_id=None, skip_embedding=False):
   564→        """Agrega un texto con deduplicacion basica por texto similar."""
   565→        if not entry_id:
   566→            entry_id = hashlib.md5(text.encode()).hexdigest()[:12]
   567→
   568→        # Verificar si ya existe
   569→        for entry in self.index:
   570→            if entry["id"] == entry_id:
   571→                return entry_id
   572→
   573→        # Deduplicacion basica por texto similar (100 chars lowercase)
   574→        text_lower = text[:100].lower()
   575→        for entry in self.index:
   576→            if entry["text"][:100].lower() == text_lower:
   577→                return entry["id"]
   578→
   579→        # Delegar al padre para el resto (guardar con/sin embedding)
   580→        return super().add(text, metadata=metadata, entry_id=entry_id, skip_embedding=skip_embedding)
   581→
   582→    def search(self, query, limit=5, min_similarity=0.3):
   583→        """Busca con decaimiento temporal. Sobreescribe para agregar decay al scoring."""
   584→        if not self.index:
   585→            return []
   586→
   587→        query_embedding = ollama.get_embedding(query)
   588→        if not query_embedding:
   589→            return self._text_search(query, limit)
   590→
   591→        candidates = self._pre_filter(query, max_candidates=50)
   592→        if not candidates:
   593→            candidates = self.index[:50]
   594→
   595→        candidate_ids = [c["id"] for c in candidates if c.get("has_vector")]
   596→        vectors = self._load_vectors_for(candidate_ids)
   597→
   598→        # Scoring vectorizado con decaimiento temporal
   599→        similarities = ollama.cosine_similarity_batch(query_embedding, vectors)
   600→
   601→        scored = []
   602→        for entry in candidates:
   603→            if not entry.get("has_vector") or entry["id"] not in similarities:
   604→                continue
   605→            raw_sim = similarities[entry["id"]]
   606→            decay = self._compute_decay(entry.get("created", ""))
   607→            final_score = raw_sim * decay
   608→
   609→            if final_score >= min_similarity * 0.5:
   610→                scored.append({
   611→                    **entry,
   612→                    "score": round(final_score, 3),
   613→                    "raw_similarity": round(raw_sim, 3),
   614→                    "decay": round(decay, 3),
   615→                })
   616→
   617→        scored.sort(key=lambda x: x["score"], reverse=True)
   618→        return scored[:limit]
   619→
   620→    def _text_search(self, query, limit=5):
   621→        """Busqueda por texto con decaimiento temporal."""
   622→        query_lower = query.lower()
   623→        query_words = [w for w in query_lower.split() if len(w) > 3]
   624→        results = []
   625→        for entry in self.index:
   626→            text_lower = entry["text"].lower()
   627→            matches = sum(1 for w in query_words if w in text_lower)
   628→            if matches > 0:
   629→                decay = self._compute_decay(entry.get("created", ""))
   630→                score = (matches / max(len(query_words), 1)) * decay
   631→                results.append({**entry, "score": round(score, 3)})
   632→        results.sort(key=lambda x: x["score"], reverse=True)
   633→        return results[:limit]
