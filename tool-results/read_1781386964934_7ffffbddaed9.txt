    55→    def run(self, user_message):
    56→        """
    57→        Bucle ReAct principal. Retorna (respuesta, thinking_log).
    58→        v14.6: Incluye fase de pensamiento profundo antes del bucle.
    59→        """
    60→        self.thinking_log = []
    61→        self.metacognition.reset()
    62→        self._tool_call_counts = {}
    63→        self._total_tool_calls = 0
    64→        self._log(f"Mensaje del usuario: {user_message}", "input")
    65→
    66→        messages = self._build_messages(user_message)
    67→
    68→        # ---- PENSAMIENTO PROFUNDO (v14.6) ----
    69→        deep_result = self.deep_thinking.think(
    70→            user_message,
    71→            context=self.memory.get_context_for(user_message)
    72→        )
    73→        if deep_result.should_deep_think:
    74→            _nsteps = len(deep_result.plan)
    75→            self._log(
    76→                f"Pensamiento profundo: complejidad={deep_result.complexity:.2f}, "
    77→                f"tipo={deep_result.query_type}, pasos={_nsteps}",
    78→                "deep_thinking"
    79→            )
    80→            thinking_prompt = self.deep_thinking.get_thinking_prompt(deep_result)
    81→            if thinking_prompt:
    82→                messages.append({"role": "user", "content": thinking_prompt})
    83→                messages.append({"role": "assistant", "content": "Entendido. Analizaré la situación antes de actuar."})
    84→
    85→        if self.supports_tool_calling is None:
    86→            self.supports_tool_calling = self._detect_tool_calling_support()
    87→            self._log(
    88→                f"Tool calling nativo: {'SI' if self.supports_tool_calling else 'NO (usando JSON fallback)'}",
    89→                "info"
    90→            )
    91→
    92→        for iteration in range(MAX_REACT_ITERATIONS):
    93→            self._log(f"--- Iteracion {iteration + 1}/{MAX_REACT_ITERATIONS} ---", "react")
    94→
    95→            # *** FIX: Metacognicion ANTES de decidir la accion ***
    96→            # Antes: se evaluaba DESPUES de actuar (paso 4→5→2)
    97→            # Ahora: se evalua ANTES (paso 2→3→4) para mejor toma de decisiones
    98→            meta_prompt = self.metacognition.get_metacognitive_prompt(iteration)
    99→            if meta_prompt:
   100→                self._log("Alerta metacognitiva inyectada ANTES de accion", "evaluation")
   101→                self._inject_metacognitive_prompt(messages, meta_prompt)
   102→
   103→            # Verificar si metacognicion indica que necesitamos mas contexto
   104→            if self.metacognition.confidence < 0.3 and iteration > 0:
   105→                extra_context = self.memory.get_context_for(user_message)
   106→                if extra_context and not any(extra_context[:50] in str(m.get("content", "")) for m in messages[-3:]):
   107→                    self._log("Metacognicion: inyectando contexto adicional de memoria", "evaluation")
   108→                    # No re-agregar contexto que ya esta en mensajes
   109→                    context_added = False
   110→                    for m in messages[-3:]:
   111→                        if "CONTEXTO DE MEMORIA" in m.get("content", ""):
   112→                            context_added = True
   113→                            break
   114→                    if not context_added:
   115→                        messages.append({
   116→                            "role": "user",
   117→                            "content": f"[Contexto adicional relevante]:\n{extra_context[:500]}\n\nUsa esta informacion si es relevante."
   118→                        })
   119→
   120→            if self.supports_tool_calling:
   121→                action_result = self._react_with_tools(messages, iteration)
   122→            else:
   123→                action_result = self._react_with_json(messages, iteration)
   124→
   125→            if action_result[0] == "respond":
   126→                final_response = action_result[1]
   127→
   128→                # SAFETY CHECK: Si la respuesta es JSON de tool call, ejecutarlo
   129→                # en vez de mostrarlo como texto al usuario
   130→                parsed_json = self._parse_json(final_response)
   131→                if parsed_json:
   132→                    accion = parsed_json.get("accion", "").strip()
   133→                    if accion and accion in TOOL_FUNCTIONS:
   134→                        self._log(f"JSON de tool detectado en respuesta: {accion}", "thinking")
   135→                        action_result = ("tool_calls", [{"name": accion, "params": parsed_json.get("params", {})}])
   136→                        # Procesar como tool call en vez de respuesta
   137→                        for tc in action_result[1]:
   138→                            self.metacognition.record_iteration(
   139→                                iteration=iteration, action_type="tool_call", tool_name=tc["name"]
   140→                            )
   141→                        results = self._execute_tool_calls(action_result[1], messages)
   142→                        for tc, res in zip(action_result[1], results):
   143→                            had_error = "ERROR" in res
   144→                            self.metacognition.record_iteration(
   145→                                iteration=iteration, action_type="tool_result",
   146→                                tool_name=tc["name"], result_summary=res[:200], had_error=had_error
   147→                            )
   148→                        self._feed_tool_results(action_result[1], results, messages)
   149→                        continue  # Siguiente iteracion del bucle ReAct
   150→                    elif not parsed_json.get("respuesta_final", "").strip():
   151→                        # JSON sin respuesta_final ni accion valida - limpiar
   152→                        clean = self._clean_json_leak(final_response)
   153→                        if clean != final_response:
   154→                            final_response = clean
   155→
   156→                # Evaluar resultado para poblar _last_evaluation
   157→                reflection = self.metacognition.evaluate_result(
   158→                    user_message, final_response, iteration + 1
   159→                )
   160→                self._log(
   161→                    f"Evaluacion: {reflection['assessment']} "
   162→                    f"(confianza={reflection['confidence_final']})",
   163→                    "evaluation"
   164→                )
   165→
   166→                # SEGUNDO: Ahora get_final_reflection_prompt() tiene datos reales
   167→                reflection_prompt = self.metacognition.get_final_reflection_prompt()
   168→                if reflection_prompt:
   169→                    self._log("Reflexion metacognitiva inyectada en respuesta final", "evaluation")
   170→                    # Re-generar con contexto de reflexion (solo si la respuesta puede mejorar)
   171→                    if len(final_response) < 100 and self.metacognition.error_count > 0:
   172→                        messages.append({"role": "assistant", "content": final_response})
   173→                        messages.append({"role": "user", "content": reflection_prompt + "\n\nMejora tu respuesta anterior si es incompleta."})

   175→                        if improved and len(improved) > len(final_response):
   176→                            final_response = improved
   177→                            self._log("Respuesta mejorada via metacognicion", "success")
   178→
   179→                for lesson in reflection.get("lessons", []):
   180→                    learning.add_knowledge(lesson, source="metacognition")
   181→
   182→                # ---- POST-REFLEXIÓN DEEP THINKING (v14.6) ----
   183→                had_errors = self.metacognition.error_count > 0
   184→                improved_response, was_improved = self.deep_thinking.reflect(
   185→                    user_message, final_response, had_errors=had_errors
   186→                )
   187→                if was_improved:
   188→                    final_response = improved_response
   189→                    self._log("Respuesta mejorada via pensamiento profundo", "deep_thinking")
   190→
   191→                self._log("Respuesta final generada", "success")
   192→                self._save_interaction(user_message, final_response)
   193→                return final_response, self.thinking_log
   194→
   195→            elif action_result[0] == "tool_calls":
   196→                tool_calls = action_result[1]
   197→                for tc in tool_calls:
   198→                    self.metacognition.record_iteration(
   199→                        iteration=iteration, action_type="tool_call", tool_name=tc["name"]
   200→                    )
   201→                results = self._execute_tool_calls(tool_calls, messages)
   202→                for tc, res in zip(tool_calls, results):

   204→                    self.metacognition.record_iteration(
   205→                        iteration=iteration, action_type="tool_result",
   206→                        tool_name=tc["name"], result_summary=res[:200], had_error=had_error
   207→                    )
   208→                self._feed_tool_results(tool_calls, results, messages)
   209→
   210→            elif action_result[0] == "error":
   211→                self._log(f"Error: {action_result[1]}", "error")
   212→                self.metacognition.record_iteration(
   213→                    iteration=iteration, action_type="error", had_error=True
   214→                )
   215→                if iteration >= MAX_REACT_ITERATIONS - 1:
   216→                    return "Tuve problemas para procesar tu solicitud. Puedes reformularla?", self.thinking_log
   217→
   218→        self._log("Alcanzado limite de iteraciones", "warning")
   219→        return "Alcance el limite de iteraciones. Puede que necesites ser mas especifico.", self.thinking_log

   221→    # ----------------------------------------------------------
   222→    # RUN CON STREAMING REAL (token a token)
   223→    # ----------------------------------------------------------
   224→    def run_stream(self, user_message):
   225→        """
   226→        Bucle ReAct con streaming REAL. Yields cada token al instante.
   227→        Eventos: {"type": "text"|"tool_start"|"tool_result"|"meta"|"thinking"|"done", "data": ...}
   228→        v14.6: Incluye fase de pensamiento profundo y evento "thinking".
   229→        """
   230→        self.thinking_log = []
   231→        self.metacognition.reset()
   232→        self._tool_call_counts = {}
   233→        self._total_tool_calls = 0
   234→        self._log(f"Mensaje del usuario: {user_message}", "input")
   235→
   236→        messages = self._build_messages(user_message)
   237→
   238→        # ---- PENSAMIENTO PROFUNDO (v14.6) ----
   239→        deep_result = self.deep_thinking.think(
   240→            user_message,
   241→            context=self.memory.get_context_for(user_message)
   242→        )
   243→        if deep_result.should_deep_think:
   244→            _nsteps = len(deep_result.plan)
   245→            self._log(
   246→                f"Pensamiento profundo: complejidad={deep_result.complexity:.2f}, "

   248→                "deep_thinking"
   249→            )

   251→            yield {
   252→                "type": "thinking",
   253→                "data": {
   254→                    "reasoning": deep_result.reasoning[:500],
   255→                    "plan": deep_result.plan,
   256→                    "complexity": deep_result.complexity,
   257→                    "query_type": deep_result.query_type,
   258→                }

   263→                messages.append({"role": "user", "content": thinking_prompt})

   267→            self.supports_tool_calling = self._detect_tool_calling_support()
   268→            self._log(
   269→                f"Tool calling nativo: {'SI' if self.supports_tool_calling else 'NO (usando JSON fallback)'}",
   270→                "info"
   271→            )
   272→
   273→        for iteration in range(MAX_REACT_ITERATIONS):
   274→            self._log(f"--- Iteracion {iteration + 1}/{MAX_REACT_ITERATIONS} ---", "react")
   275→
   276→            # *** FIX: Metacognicion ANTES de decidir la accion ***

   282→                    "type": "meta",
   283→                    "data": {
   284→                        "confidence": self.metacognition.confidence,
   285→                        "errors": self.metacognition.error_count,
   286→                        "successes": self.metacognition.success_count,
   287→                        "plan_changes": self.metacognition.plan_changes,
   288→                    }
   289→                }
   290→
   291→            # ---- NUCLEO: Streaming token-a-token ----
   292→            full_text = ""
   293→            tool_calls_found = []
   294→            is_final_response = False
   295→
   296→            if self.supports_tool_calling:
   297→                # Stream del LLM y procesar en tiempo real
   298→                for event in self._stream_llm_with_tools(messages):
   299→                    if event["type"] == "token":
   300→                        full_text += event["data"]
   301→                        yield {"type": "text", "data": event["data"]}
   302→                    elif event["type"] == "tool_calls":
   303→                        tool_calls_found = event["data"]
   304→                    elif event["type"] == "done":
   305→                        is_final_response = event["data"]  # True si es respuesta final
   306→            else:
   307→                # JSON fallback (sin streaming real)
   308→                result_type, result_data = self._react_with_json(messages, iteration)
   309→                if result_type == "respond":
   310→                    full_text = result_data
   311→                    yield {"type": "text", "data": result_data}
   312→                    is_final_response = True
   313→                elif result_type == "tool_calls":
   314→                    tool_calls_found = result_data
   315→                elif result_type == "error":
   316→                    self._log(f"Error: {result_data}", "error")
   317→                    self.metacognition.record_iteration(
   318→                        iteration=iteration, action_type="error", had_error=True
   319→                    )
   320→                    if iteration >= MAX_REACT_ITERATIONS - 1:
   321→                        yield {
   322→                            "type": "done",
   323→                            "data": "Tuve problemas para procesar tu solicitud. Puedes reformularla?",
   324→                            "thinking_log": self.thinking_log,
   325→                            "meta_status": self.metacognition.get_status(),
   326→                        }
   327→                        return
   328→                    continue
   329→
   330→            # Procesar resultado de esta iteracion
   331→            # SAFETY CHECK: Si full_text parece JSON de tool call, parsearlo y ejecutar
   332→            # en vez de mostrarlo como texto al usuario
   333→            if is_final_response and not tool_calls_found and full_text:
   334→                parsed_json = self._parse_json(full_text)
   335→                if parsed_json:
   336→                    accion = parsed_json.get("accion", "").strip()
   337→                    if accion and accion in TOOL_FUNCTIONS:
   338→                        self._log(f"JSON de tool detectado en respuesta final: {accion}", "thinking")
   339→                        tool_calls_found = [{"name": accion, "params": parsed_json.get("params", {})}]
   340→                        is_final_response = False  # No es respuesta final, es tool call
   341→                    elif not parsed_json.get("respuesta_final", "").strip():
   342→                        # JSON sin respuesta_final ni accion valida - extraer contenido util
   343→                        clean = self._clean_json_leak(full_text)
   344→                        if clean != full_text:
   345→                            full_text = clean
   346→
   347→            if is_final_response and not tool_calls_found:
   348→                # PRIMERO: Evaluar resultado para poblar _last_evaluation
   349→                reflection = self.metacognition.evaluate_result(
   350→                    user_message, full_text, iteration + 1
   351→                )
   352→                self._log(
   353→                    f"Evaluacion: {reflection['assessment']} (confianza={reflection['confidence_final']})",
   354→                    "evaluation"
   355→                )
   356→
   357→                # SEGUNDO: Ahora get_final_reflection_prompt() tiene datos reales
   358→                reflection_prompt = self.metacognition.get_final_reflection_prompt()

   361→                    try:
   362→                        messages.append({"role": "assistant", "content": full_text})
   363→                        messages.append({"role": "user", "content": reflection_prompt + "\n\nDa una respuesta mas completa."})

   365→                        if improved and len(improved) > len(full_text):
   366→                            full_text = improved

   369→                        pass
   370→
   371→                # Respuesta final - terminar
   372→                for lesson in reflection.get("lessons", []):
   373→                    learning.add_knowledge(lesson, source="metacognition")
   374→
   375→                # ---- POST-REFLEXIÓN DEEP THINKING (v14.6) ----
   376→                had_errors = self.metacognition.error_count > 0
   377→                improved_response, was_improved = self.deep_thinking.reflect(
   378→                    user_message, full_text, had_errors=had_errors
   379→                )
   380→                if was_improved:
   381→                    full_text = improved_response
   382→                    self._log("Respuesta mejorada via pensamiento profundo", "deep_thinking")
   383→
   384→                self._log("Respuesta final generada (streaming)", "success")
   385→                self._save_interaction(user_message, full_text)
   386→                yield {
   387→                    "type": "done",
   388→                    "data": full_text,
   389→                    "thinking_log": self.thinking_log,
   390→                    "meta_status": self.metacognition.get_status(),
   391→                    "deep_thinking_stats": self.deep_thinking.stats(),
   392→                }
   393→                return
   394→
   395→            elif tool_calls_found:
   396→                # Ejecutar tool calls
   397→                for tc in tool_calls_found:
   398→                    self.metacognition.record_iteration(
   399→                        iteration=iteration, action_type="tool_call", tool_name=tc["name"]
   400→                    )
   401→                    yield {"type": "tool_start", "data": tc}
   402→
   403→                results = self._execute_tool_calls(tool_calls_found, messages)
   404→
   405→                for tc, res in zip(tool_calls_found, results):
   406→                    had_error = "ERROR" in res
   407→                    self.metacognition.record_iteration(
   408→                        iteration=iteration, action_type="tool_result",
   409→                        tool_name=tc["name"], result_summary=res[:200], had_error=had_error
   410→                    )
   411→                    yield {"type": "tool_result", "data": {"tool": tc, "result": res}}
   412→
   413→                self._feed_tool_results(tool_calls_found, results, messages)
   414→
   415→            elif not full_text:
   416→                # Ni respuesta ni tool calls - error
   417→                self._log("Respuesta vacia del modelo", "error")
   418→                self.metacognition.record_iteration(
   419→                    iteration=iteration, action_type="error", had_error=True
   420→                )
   421→                if iteration >= MAX_REACT_ITERATIONS - 1:
   422→                    yield {
   423→                        "type": "done",
   424→                        "data": "Tuve problemas para procesar tu solicitud. Puedes reformularla?",
   425→                        "thinking_log": self.thinking_log,
   426→                        "meta_status": self.metacognition.get_status(),
   427→                    }
   428→                    return
   429→
   430→        # Limite de iteraciones
   431→        yield {
   432→            "type": "done",
   433→            "data": "Alcance el limite de iteraciones. Puede que necesites ser mas especifico.",
   434→            "thinking_log": self.thinking_log,
   435→            "meta_status": self.metacognition.get_status(),
   436→        }

   438→    # ----------------------------------------------------------
   439→    # STREAMING LLM CON TOOL CALLING (token a token)
   440→    # ----------------------------------------------------------
   441→    def _stream_llm_with_tools(self, messages):
   442→        """
   443→        Genera respuesta del LLM con streaming REAL.
   444→        Yields: {"type": "token"|"tool_calls"|"done", "data": ...}
   445→        """
   446→        import time as _time
   447→        _llm_start = _time.monotonic()
   448→        full_content = ""
   449→        full_tool_calls = []
   450→
   451→        try:
   452→            # Intentar streaming HTTP directo (mas rapido, sin overhead de lib)
   453→            for chunk in ollama.generate_stream(messages, tools=TOOL_SCHEMAS):
   454→                if isinstance(chunk, str):
   455→                    # Token de texto - emitir inmediatamente
   456→                    # PERO: si parece JSON de tool call, no emitirlo al usuario
   457→                    full_content += chunk
   458→                    # Detectar si el contenido es JSON de herramienta
   459→                    if self._looks_like_tool_json(full_content):
   460→                        continue  # Acumular sin emitir
   461→                    # Si antes se acumulaba JSON pero ya no, limpiar
   462→                    yield {"type": "token", "data": chunk}
   463→                elif isinstance(chunk, dict):
   464→                    # Resultado final con tool_calls
   465→                    msg = chunk.get("message", chunk)
   466→                    tool_calls = msg.get("tool_calls", [])
   467→                    content = msg.get("content", "")
   468→
   469→                    if content and not full_content:
   470→                        # Contenido que no se streameo (fallback)
   471→                        full_content = content
   472→                        yield {"type": "token", "data": content}
   473→
   474→                    if tool_calls:
   475→                        # Parsear tool calls
   476→                        parsed_calls = []
   477→                        for tc in tool_calls:
   478→                            tool_name = tc.get("function", {}).get("name", "")
   479→                            tool_params = tc.get("function", {}).get("arguments", {})
   480→                            self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
   481→                            parsed_calls.append({"name": tool_name, "params": tool_params})
   482→                        yield {"type": "tool_calls", "data": parsed_calls}
   483→                        yield {"type": "done", "data": False}
   484→                        return
   485→
   486→            # Si llegamos aqui sin tool_calls, verificar si el contenido es JSON de herramientas
   487→            if full_content:
   488→                # Si el contenido acumulado era JSON de tool calls (no se emitio),
   489→                # parsearlo y ejecutar como tool calls
   490→                if self._looks_like_tool_json(full_content):
   491→                    parsed = self._parse_json(full_content)
   492→                    if parsed:
   493→                        # Verificar si son multiples tool calls
   494→                        if parsed.get("_multi_tool_calls") and parsed.get("tool_calls"):
   495→                            yield {"type": "tool_calls", "data": parsed["tool_calls"]}
   496→                            yield {"type": "done", "data": False}
   497→                            return
   498→                        accion = parsed.get("accion", "").strip()
   499→                        if accion and accion in TOOL_FUNCTIONS:
   500→                            yield {"type": "tool_calls", "data": [{"name": accion, "params": parsed.get("params", {})}]}
   501→                            yield {"type": "done", "data": False}
   502→                            return
   503→                    # Si no se pudo parsear como tool call, emitir el texto limpio
   504→                    clean = self._clean_json_leak(full_content)
   505→                    if clean != full_content:
   506→                        # El texto era JSON, emitir version limpia
   507→                        yield {"type": "token", "data": clean}
   508→                _llm_elapsed = (_time.monotonic() - _llm_start) * 1000.0
   509→                get_metrics().record_llm_call(_llm_elapsed)
   510→                yield {"type": "done", "data": True}
   511→                return
   512→
   513→        except Exception as e:
   514→            self._log(f"Error en streaming: {e}", "error")
   515→            get_metrics().record_error("llm_stream")
   516→
   517→        # Fallback: modo no-streaming
   518→        try:
   519→            response = ollama.generate(messages, tools=TOOL_SCHEMAS)
   520→            # generate() already records llm_call via @timed, so no duplicate
   521→            if isinstance(response, str):
   522→                yield {"type": "token", "data": response}
   523→                yield {"type": "done", "data": True}
   524→                return
   525→
   526→            message = response.get("message", response)
   527→            tool_calls = message.get("tool_calls", [])
   528→            content = message.get("content", "")
   529→
   530→            if tool_calls:
   531→                parsed_calls = []
   532→                for tc in tool_calls:
   533→                    tool_name = tc.get("function", {}).get("name", "")
   534→                    tool_params = tc.get("function", {}).get("arguments", {})
   535→                    self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
   536→                    parsed_calls.append({"name": tool_name, "params": tool_params})
   537→                yield {"type": "tool_calls", "data": parsed_calls}
   538→                yield {"type": "done", "data": False}
   539→                return
   540→
   541→            if content:
   542→                yield {"type": "token", "data": content}
   543→                yield {"type": "done", "data": True}
   544→                return
   545→
   546→        except Exception as e:
   547→            self._log(f"Error en fallback: {e}", "error")
   548→
   549→        yield {"type": "done", "data": False}
   550→

   556→        # generate() already records llm_call via @timed decorator
   557→        try:
   558→            response = ollama.generate(messages, tools=TOOL_SCHEMAS)
   559→
   560→            if isinstance(response, str):
   561→                return ("respond", response)
   562→
   563→            message = response.get("message", response)
   564→            tool_calls = message.get("tool_calls", [])
   565→
   566→            if tool_calls:
   567→                parsed_calls = []
   568→                for tc in tool_calls:
   569→                    tool_name = tc.get("function", {}).get("name", "")
   570→                    tool_params = tc.get("function", {}).get("arguments", {})
   571→                    self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
   572→                    parsed_calls.append({"name": tool_name, "params": tool_params})
   573→                return ("tool_calls", parsed_calls)
   574→
   575→            content = message.get("content", "")
   576→            if content:
   577→                # SAFETY CHECK: Si el modelo genero JSON de tool call en vez de
   578→                # usar function calling nativo, parsearlo y ejecutarlo
   579→                parsed = self._parse_json(content)
   580→                if parsed:
   581→                    accion = parsed.get("accion", "").strip()
   582→                    if accion and accion in TOOL_FUNCTIONS:
   583→                        self._log(f"Tool call via JSON (modelo no uso function calling nativo): {accion}", "thinking")
   584→                        params = parsed.get("params", {})
   585→                        return ("tool_calls", [{"name": accion, "params": params}])
   586→                    # Si tiene respuesta_final, usarla
   587→                    respuesta_final = parsed.get("respuesta_final", "").strip()
   588→                    if respuesta_final:
   589→                        return ("respond", respuesta_final)
   590→                return ("respond", content)
   591→
   592→            return ("error", "Respuesta vacia del modelo")
   593→
   594→        except Exception as e:
   595→            self._log(f"Error en tool calling: {e}", "error")
   596→            self.supports_tool_calling = False
   597→            return ("error", str(e))

   602→    def _react_with_json(self, messages, iteration):
   603→        """ReAct usando JSON parsing (fallback)."""
   604→        # generate_chat() already records llm_call via @timed decorator
   605→        if not any("HERRAMIENTAS DISPONIBLES" in str(m.get("content", "")) for m in messages):
   606→            system_msg_idx = next((i for i, m in enumerate(messages) if m["role"] == "system"), -1)
   607→            if system_msg_idx >= 0:
   608→                messages[system_msg_idx]["content"] += JSON_TOOLS_PROMPT
   609→
   610→        try:
   611→            response = ollama.generate_chat(messages)
   612→            if not response:
   613→                return ("error", "El LLM no respondio")
   614→
   615→            parsed = self._parse_json(response)
   616→            if not parsed:
   617→                # No es JSON - probablemente respuesta directa del modelo
   618→                # Limpiar posibles restos de formato JSON
   619→                clean = self._clean_json_leak(response)
   620→                return ("respond", clean)
   621→
   622→            # 1. Si hay _multi_tool_calls, ejecutar todas las herramientas
   623→            if parsed.get("_multi_tool_calls") and parsed.get("tool_calls"):
   624→                return ("tool_calls", parsed["tool_calls"])
   625→
   626→            # 2. Si hay respuesta_final con contenido, usarla
   627→            respuesta_final = parsed.get("respuesta_final", "").strip()
   628→            if respuesta_final:
   629→                return ("respond", respuesta_final)
   630→
   631→            accion = parsed.get("accion", "").strip()
   632→            params = parsed.get("params", {})
   633→            pensamiento = parsed.get("pensamiento", "").strip()
   634→
   635→            if pensamiento:
   636→                self._log(f"Pensamiento: {pensamiento}", "thinking")
   637→
   638→            # 3. Si hay accion valida, ejecutar herramienta
   639→            if accion and accion in TOOL_FUNCTIONS:
   640→                # Si tambien hay pensamiento, inyectarlo como contexto
   641→                if pensamiento:
   642→                    messages.append({
   643→                        "role": "assistant",
   644→                        "content": f"[Pensamiento: {pensamiento}] Ejecutando {accion}..."
   645→                    })
   646→                return ("tool_calls", [{"name": accion, "params": params}])
   647→
   648→            # 3. Si hay pensamiento pero no accion ni respuesta_final,
   649→            #    el modelo esta respondiendo en campo equivocado - usar pensamiento
   650→            if pensamiento and not accion:
   651→                self._log("Modelo respondio en 'pensamiento' en vez de 'respuesta_final'", "info")
   652→                return ("respond", pensamiento)
   653→
   654→            # 4. Fallback: devolver texto limpio (no JSON crudo)
   655→            clean = self._clean_json_leak(response)
   656→            return ("respond", clean)
   657→
   658→        except Exception as e:
   659→            return ("error", str(e))

   661→    def _clean_json_leak(self, text):
   662→        """Limpia texto que tiene restos de formato JSON para mostrar al usuario.
   663→        v3: Mas agresivo - tambien limpia JSON parcial al inicio/final del texto.
   664→        """
   665→        # Si el texto es JSON completo, extraer solo el contenido util
   666→        parsed = self._parse_json(text)
   667→        if parsed:
   668→            # Prioridad: respuesta_final > pensamiento > texto original
   669→            if parsed.get("respuesta_final", "").strip():
   670→                return parsed["respuesta_final"].strip()
   671→            if parsed.get("pensamiento", "").strip():
   672→                return parsed["pensamiento"].strip()
   673→        # Si no es JSON completo, intentar limpiar restos
   674→        # Caso: texto que empieza con JSON parcial
   675→        import re as _re
   676→        # Remover JSON parcial al inicio: {"pensamiento": "..." ... restos
   677→        cleaned = _re.sub(r'^\s*\{[^}]*$', '', text).strip()
   678→        # Remover JSON parcial al final
   679→        cleaned = _re.sub(r'\{[^}]*$\s*$', '', cleaned).strip()
   680→        # Si despues de limpiar quedo vacio, devolver texto original
   681→        return cleaned if cleaned else text

   684→    # EJECUCION DE TOOLS (paralelo + retry)
   685→    # ----------------------------------------------------------
   686→    def _execute_tool_calls(self, tool_calls, messages):
   687→        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   688→        from concurrent.futures import ThreadPoolExecutor, as_completed
   689→
   690→        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   691→        filtered_calls = []
   692→        for tc in tool_calls:
   693→            tool_name = tc["name"]
   694→            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   695→            self._total_tool_calls += 1
   696→
   697→            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   698→                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   699→                break
   700→
   701→            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   702→                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   703→                continue
   704→
   705→            filtered_calls.append(tc)
   706→
   707→        tool_calls = filtered_calls
   708→
   709→        # Validar parametros de todos los tool calls primero
   710→        for tc in tool_calls:
   711→            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   712→
   713→        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   714→        if len(tool_calls) == 1:
   715→            return [self._execute_single_tool(tool_calls[0], messages)]
   716→
   717→        # Si hay multiples tools, ejecutar en paralelo
   718→        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   719→        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   720→        
   721→        if has_sequential:
   722→            # Ejecutar secuencialmente para evitar race conditions
   723→            results = []
   724→            for tc in tool_calls:
   725→                results.append(self._execute_single_tool(tc, messages))
   726→            return results
   727→
   728→        # Ejecucion paralela para tools de solo lectura
   729→        results = [None] * len(tool_calls)
   730→        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   731→            futures = {}
   732→            for i, tc in enumerate(tool_calls):
   733→                future = executor.submit(self._execute_single_tool, tc, messages)
   734→                futures[future] = i
   735→            
   736→            for future in as_completed(futures):
   737→                idx = futures[future]
   738→                try:
   739→                    results[idx] = future.result(timeout=30)
   740→                except Exception as e:
   741→                    results[idx] = f"ERROR: {e}"
   742→
   743→        return results
   744→
   745→    def _execute_single_tool(self, tc, messages, max_retries=1):
   746→        """Ejecuta un solo tool call con retry automatico."""
   747→        import time as _time
   748→        tool_name = tc["name"]
   749→        tool_params = tc["params"]
   750→
   751→        self._log(f"Ejecutando: {tool_name}({tool_params})", "execution")
   752→
   753→        _tool_start = _time.monotonic()
   754→        tool_result = self._execute_tool(tool_name, tool_params)
   755→
   756→        # Retry si fallo y el tool es reintentable
   757→        if "ERROR" in tool_result and max_retries > 0:
   758→            retryable_errors = ["TIMEOUT", "ConnectionError", "empty", "vacia"]
   759→            if any(err in tool_result for err in retryable_errors):
   760→                self._log(f"Reintentando {tool_name} (error transitorio)...", "execution")
   761→                tool_result = self._execute_tool(tool_name, tool_params)
   762→
   763→        _tool_elapsed_ms = (_time.monotonic() - _tool_start) * 1000.0
   764→        get_metrics().record_tool_call(tool_name, _tool_elapsed_ms)
   765→
   766→        if "ERROR" in tool_result:
   767→            get_metrics().record_error("tool:" + tool_name)
   768→
   769→        self._log(f"Resultado: {tool_result[:150]}...", "observation")
   770→
   771→        # Alimentar memoria de trabajo
   772→        self.memory.add_step(f"{tool_name}({tool_params})", tool_result[:200])
   773→        if "ERROR" in tool_result:
   774→            self.memory.set_error(f"{tool_name}: {tool_result[:100]}")
   775→        if "PELIGROSO" in tool_result:
   776→            self._log(f"Comando peligroso bloqueado: {tool_name}", "warning")
   777→        if len(tool_result) > 50 and "ERROR" not in tool_result:
   778→            self.memory.remember(
   779→                f"Resultado de {tool_name}: {tool_result[:300]}",
   780→                metadata={"type": "tool_result", "tool": tool_name}
   781→            )
   782→
   783→        return tool_result

   785→    def _validate_tool_params(self, tool_name, params):
   786→        """Valida y limpia parametros de un tool call antes de ejecutar."""
   787→        if not isinstance(params, dict):

   789→        
   790→        # Asegurar que params tenga las claves correctas

   802→                else:
   803→                    clean_params[key] = value
   804→        
   805→        return clean_params

   926→    def _find_repo_path(self):
   927→        try:
   928→            dirs = [d for d in os.listdir(REPOS_DIR)
   929→                    if os.path.isdir(os.path.join(REPOS_DIR, d)) and not d.startswith(".")]
   930→            if dirs:
   931→                latest = max(dirs, key=lambda d: os.path.getmtime(os.path.join(REPOS_DIR, d)))
   932→                return os.path.join(REPOS_DIR, latest)

   937→    def _parse_json(self, text):
   938→        try:
   939→            return json.loads(text)
   940→        except Exception:
   941→            pass
   942→        patterns = [
   943→            r'```json\s*(.*?)\s*```',
   944→            r'```\s*(.*?)\s*```',
   945→        ]
   946→        for pattern in patterns:
   947→            match = re.search(pattern, text, re.DOTALL)
   948→            if match:
   949→                try:
   950→                    return json.loads(match.group(1))
   951→                except Exception:
   952→                    continue
   953→        # Intentar parsear multiples JSONs en el texto
   954→        # Cada JSON es una accion separada del ReAct
   955→        jsons = self._extract_all_jsons(text)
   956→        if len(jsons) == 1:
   957→            return jsons[0]
   958→        elif len(jsons) > 1:
   959→            # Multiples JSONs = multiples tool calls
   960→            # Combinar en una lista de tool calls
   961→            tool_calls = []
   962→            for j in jsons:
   963→                accion = j.get("accion", "").strip()
   964→                if accion:
   965→                    tool_calls.append({"name": accion, "params": j.get("params", {})})
   966→            if tool_calls:
   967→                # Retornar formato especial para multiples tool calls
   968→                return {"_multi_tool_calls": True, "tool_calls": tool_calls}
   969→            return jsons[0]  # Fallback al primer JSON
   970→        # Ultimo intento: buscar un solo objeto JSON
   971→        match = re.search(r'(\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\})', text, re.DOTALL)
   972→        if match:
   973→            try:
   974→                return json.loads(match.group(1))
   975→            except Exception:
   976→                pass
   977→        return None

   979→    def _extract_all_jsons(self, text):
   980→        """Extrae todos los objetos JSON validos de un texto."""
   981→        results = []
   982→        # Buscar secuencias que parezcan JSON
   983→        i = 0
   984→        while i < len(text):
   985→            # Encontrar el proximo '{'
   986→            start = text.find('{', i)
   987→            if start == -1:
   988→                break
   989→            # Intentar parsear desde este punto
   990→            depth = 0
   991→            end = start
   992→            for j in range(start, len(text)):
   993→                if text[j] == '{':
   994→                    depth += 1
   995→                elif text[j] == '}':
   996→                    depth -= 1
   997→                if depth == 0:
   998→                    end = j + 1
   999→                    break
  1000→            if end > start:
  1001→                try:
  1002→                    parsed = json.loads(text[start:end])
  1003→                    if isinstance(parsed, dict):
  1004→                        results.append(parsed)
  1005→                except Exception:
  1006→                    pass
  1007→                i = end
  1008→            else:
  1009→                i = start + 1
  1010→        return results

  1012→    def _looks_like_tool_json(self, text):
  1013→        """Detecta si el texto parece ser JSON de tool call (no respuesta al usuario).
  1014→        v3: Mas robusto - detecta JSON que empieza despues de whitespace o texto corto.
  1015→        """
  1016→        text = text.strip()
  1017→        if not text:
  1018→            return False
  1019→        # Si el texto empieza con { probablemente es JSON de tool call
  1020→        if text.startswith('{'):
  1021→            return True
  1022→        # Si despues de whitespace/newline hay {, tambien es JSON
  1023→        # Esto captura casos donde el modelo genera: "\n{" o "  {"
  1024→        stripped = text.lstrip()
  1025→        if stripped.startswith('{'):
  1026→            return True

  1031→            if indicator in text:

  1033→        # Si el texto es corto y parece inicio de JSON
