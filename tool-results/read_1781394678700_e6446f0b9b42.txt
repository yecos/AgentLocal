    93тЖТ    93тЖТdef _tts_edge(texto, ruta, voz, velocidad):

    96тЖТ    96тЖТ        import asyncio
    97тЖТ    97тЖТ
    98тЖТ    98тЖТ        # Mapear idioma a voz de Edge
    99тЖТ    99тЖТ        voice_map = {
   100тЖТ   100тЖТ            "es": "es-ES-ElviraNeural",
   101тЖТ   101тЖТ            "es-MX": "es-MX-DaliaNeural",
   102тЖТ   102тЖТ            "en": "en-US-AriaNeural",

   106тЖТ   106тЖТ            "pt": "pt-BR-FranciscaNeural",

   108тЖТ   108тЖТ            "ja": "ja-JP-NanamiNeural",
   109тЖТ   109тЖТ            "ko": "ko-KR-SunHiNeural",
   110тЖТ   110тЖТ            "zh": "zh-CN-XiaoxiaoNeural",
   111тЖТ   111тЖТ            "ru": "ru-RU-SvetlanaNeural",
   112тЖТ   112тЖТ        }
   113тЖТ   113тЖТ
   114тЖТ   114тЖТ        voice_name = voice_map.get(voz, voz)
   115тЖТ   115тЖТ        # Si es solo un codigo de 2 letras, usar el mapeo
   116тЖТ   116тЖТ        if len(voz) == 2 and voz in voice_map:
   117тЖТ   117тЖТ            voice_name = voice_map[voz]
   118тЖТ   118тЖТ
   119тЖТ   119тЖТ        rate = f"{'+'
   120тЖТ   120тЖТ                       if velocidad >= 1.0 else ''}{int((velocidad - 1.0) * 100)}%"
   121тЖТ   121тЖТ
   122тЖТ   122тЖТ        try:
   123тЖТ   123тЖТ            import edge_tts
   124тЖТ   124тЖТ        except ImportError:
   125тЖТ   125тЖТ            return None
   126тЖТ   126тЖТ
   127тЖТ   127тЖТ        async def _generate():
   128тЖТ   128тЖТ            communicate = edge_tts.Communicate(texto, voice_name, rate=rate)
   129тЖТ   129тЖТ            await communicate.save(ruta)
   130тЖТ   130тЖТ
   131тЖТ   131тЖТ        # Ejecutar async
   132тЖТ   132тЖТ        try:
   133тЖТ   133тЖТ            loop = asyncio.get_event_loop()
   134тЖТ   134тЖТ            if loop.is_running():
   135тЖТ   135тЖТ                import concurrent.futures
   136тЖТ   136тЖТ                with concurrent.futures.ThreadPoolExecutor() as pool:
   137тЖТ   137тЖТ                    pool.submit(lambda: asyncio.run(_generate())).result()
   138тЖТ   138тЖТ            else:
   139тЖТ   139тЖТ                loop.run_until_complete(_generate())
   140тЖТ   140тЖТ        except RuntimeError:
   141тЖТ   141тЖТ            asyncio.run(_generate())
   142тЖТ   142тЖТ
   143тЖТ   143тЖТ        if os.path.exists(ruta) and os.path.getsize(ruta) > 0:
   144тЖТ   144тЖТ            size_kb = os.path.getsize(ruta) / 1024
   145тЖТ   145тЖТ            return f"Audio TTS creado: {ruta} ({size_kb:.0f} KB, voz: {voice_name})"
   146тЖТ   146тЖТ
   147тЖТ   147тЖТ    except ImportError:
   148тЖТ   148тЖТ        return None
   149тЖТ   149тЖТ    except Exception as e:
   150тЖТ   150тЖТ        logger.debug(f"edge-tts fallo: {e}")
   151тЖТ   151тЖТ        return f"ERROR TTS con edge-tts: {e}"

   154тЖТ   154тЖТdef _tts_pyttsx3(texto, ruta, voz, velocidad):
   155тЖТ   155тЖТ    """TTS con pyttsx3 (offline, voces del sistema)."""
   156тЖТ   156тЖТ    try:
   157тЖТ   157тЖТ        import pyttsx3
   158тЖТ   158тЖТ
   159тЖТ   159тЖТ        engine = pyttsx3.init()
   160тЖТ   160тЖТ
   161тЖТ   161тЖТ        # Configurar velocidad
   162тЖТ   162тЖТ        rate = engine.getProperty('rate')
   163тЖТ   163тЖТ        engine.setProperty('rate', int(rate * velocidad))
   164тЖТ   164тЖТ
   165тЖТ   165тЖТ        # Buscar voz en el idioma deseado
   166тЖТ   166тЖТ        voices = engine.getProperty('voices')
   167тЖТ   167тЖТ        for v in voices:
   168тЖТ   168тЖТ            if voz.lower() in v.id.lower() or voz.lower() in v.name.lower():
   169тЖТ   169тЖТ                engine.setProperty('voice', v.id)
   170тЖТ   170тЖТ                break
   171тЖТ   171тЖТ
   172тЖТ   172тЖТ        # Guardar
   173тЖТ   173тЖТ        engine.save_to_file(texto, ruta)
   174тЖТ   174тЖТ        engine.runAndWait()
   175тЖТ   175тЖТ
   176тЖТ   176тЖТ        if os.path.exists(ruta) and os.path.getsize(ruta) > 0:
   177тЖТ   177тЖТ            size_kb = os.path.getsize(ruta) / 1024
   178тЖТ   178тЖТ            return f"Audio TTS creado: {ruta} ({size_kb:.0f} KB, pyttsx3)"
   179тЖТ   179тЖТ
   180тЖТ   180тЖТ    except ImportError:
   181тЖТ   181тЖТ        return None
   182тЖТ   182тЖТ    except Exception as e:
   183тЖТ   183тЖТ        logger.debug(f"pyttsx3 fallo: {e}")
   184тЖТ   184тЖТ        return f"ERROR TTS con pyttsx3: {e}"

   677тЖТ   677тЖТdef _video_extract_frames(ruta, params):
   678тЖТ   678тЖТ    """Extrae frames del video como imagenes PNG."""
   679тЖТ   679тЖТ    try:
   680тЖТ   680тЖТ        interval = params.get("intervalo", params.get("interval", 5))  # cada N segundos
   681тЖТ   681тЖТ        max_frames = params.get("max", 10)
   682тЖТ   682тЖТ        output_dir = params.get("output_dir", LEARN_DIR)
   683тЖТ   683тЖТ

   685тЖТ   685тЖТ
   686тЖТ   686тЖТ        base_name = os.path.splitext(os.path.basename(ruta))[0]
   687тЖТ   687тЖТ        pattern = os.path.join(output_dir, f"{base_name}_frame_%03d.png")
   688тЖТ   688тЖТ
   689тЖТ   689тЖТ        cmd = [
   690тЖТ   690тЖТ            'ffmpeg', '-i', ruta,
   691тЖТ   691тЖТ            '-vf', f'fps=1/{interval}',
   692тЖТ   692тЖТ            '-frames:v', str(max_frames),
   693тЖТ   693тЖТ            '-q:v', '2',
   694тЖТ   694тЖТ            pattern
   695тЖТ   695тЖТ        ]
   696тЖТ   696тЖТ
   697тЖТ   697тЖТ        result = subprocess.run(cmd, capture_output=True, text=True, timeout=60)
   698тЖТ   698тЖТ
   699тЖТ   699тЖТ        # Contar frames generados
   700тЖТ   700тЖТ        frames = [f for f in os.listdir(output_dir)
   701тЖТ   701тЖТ                  if f.startswith(base_name) and f.endswith('.png')]
   702тЖТ   702тЖТ
   703тЖТ   703тЖТ        if frames:
   704тЖТ   704тЖТ            return (f"Frames extraidos: {len(frames)} de {ruta}\n"
   705тЖТ   705тЖТ                    f"  Intervalo: cada {interval}s\n"
   706тЖТ   706тЖТ                    f"  Directorio: {output_dir}\n"
   707тЖТ   707тЖТ                    f"  Archivos: {', '.join(sorted(frames)[:5])}"
   708тЖТ   708тЖТ                    + (f" ... y {len(frames)-5} mas" if len(frames) > 5 else ""))
   709тЖТ   709тЖТ
   710тЖТ   710тЖТ        return "ERROR: No se pudieron extraer frames del video."
   711тЖТ   711тЖТ
   712тЖТ   712тЖТ    except FileNotFoundError:
   713тЖТ   713тЖТ        return "ERROR: ffmpeg no instalado. Instala: sudo apt install ffmpeg"
   714тЖТ   714тЖТ    except Exception as e:
   715тЖТ   715тЖТ        return f"ERROR extrayendo frames: {e}"

   718тЖТ   718тЖТdef _video_analyze_with_vlm(ruta, params):
   719тЖТ   719тЖТ    """Analiza video extrayendo frames y usando VLM para describir."""
   720тЖТ   720тЖТ    try:
   721тЖТ   721тЖТ        # Extraer 3 frames clave (inicio, medio, final)
   722тЖТ   722тЖТ        temp_dir = tempfile.mkdtemp()
   723тЖТ   723тЖТ        base_name = "frame"
   724тЖТ   724тЖТ
   725тЖТ   725тЖТ        # Obtener duracion
   726тЖТ   726тЖТ        probe = subprocess.run(
   727тЖТ   727тЖТ            ['ffprobe', '-v', 'quiet', '-print_format', 'json',
   728тЖТ   728тЖТ             '-show_format', ruta],
   729тЖТ   729тЖТ            capture_output=True, text=True, timeout=10
   730тЖТ   730тЖТ        )
   731тЖТ   731тЖТ        duration = float(json.loads(probe.stdout).get("format", {}).get("duration", 0))
   732тЖТ   732тЖТ
   733тЖТ   733тЖТ        frames_info = []
   734тЖТ   734тЖТ        positions = [0, duration * 0.5, duration * 0.9] if duration > 0 else [0]
   735тЖТ   735тЖТ
   736тЖТ   736тЖТ        for i, pos in enumerate(positions[:3]):
   737тЖТ   737тЖТ            frame_path = os.path.join(temp_dir, f"{base_name}_{i}.png")
   738тЖТ   738тЖТ            cmd = ['ffmpeg', '-ss', str(pos), '-i', ruta,
   739тЖТ   739тЖТ                   '-frames:v', '1', '-q:v', '2', frame_path]
   740тЖТ   740тЖТ            subprocess.run(cmd, capture_output=True, timeout=30)
   741тЖТ   741тЖТ
   742тЖТ   742тЖТ            if os.path.exists(frame_path):
   743тЖТ   743тЖТ                frames_info.append((pos, frame_path))
   744тЖТ   744тЖТ
   745тЖТ   745тЖТ        if not frames_info:
   746тЖТ   746тЖТ            return "ERROR: No se pudieron extraer frames para analizar."
   747тЖТ   747тЖТ
   748тЖТ   748тЖТ        # Analizar cada frame con VLM
   749тЖТ   749тЖТ        results = [f"Analisis de video: {ruta}\nDuracion: {duration:.1f}s\n"]
   750тЖТ   750тЖТ
   751тЖТ   751тЖТ        for pos, frame_path in frames_info:
   752тЖТ   752тЖТ            try:
   753тЖТ   753тЖТ                from llm import ollama
   754тЖТ   754тЖТ                desc = ollama.generate_with_image(
   755тЖТ   755тЖТ                    f"Describe lo que ves en este frame de video. "
   756тЖТ   756тЖТ                    f"Timestamp: {pos:.1f}s",
   757тЖТ   757тЖТ                    frame_path
   758тЖТ   758тЖТ                )
   759тЖТ   759тЖТ                results.append(f"[{pos:.1f}s] {desc}")
   760тЖТ   760тЖТ            except Exception as e:
   761тЖТ   761тЖТ                results.append(f"[{pos:.1f}s] (VLM no disponible: {e})")
   762тЖТ   762тЖТ
   763тЖТ   763тЖТ        # Limpiar
   764тЖТ   764тЖТ        import shutil
   765тЖТ   765тЖТ        shutil.rmtree(temp_dir, ignore_errors=True)
   766тЖТ   766тЖТ
   767тЖТ   767тЖТ        return "\n".join(results)
   768тЖТ   768тЖТ
   769тЖТ   769тЖТ    except Exception as e:
   770тЖТ   770тЖТ        return f"ERROR analizando video: {e}"
