    40тЖТ    40тЖТ    """Realiza una busqueda profunda multi-ronda sobre un tema. Busca en multiples
    41тЖТ    41тЖТ    fuentes, sigue enlaces relevantes y sintetiza un informe completo.
    42тЖТ    42тЖТ
    43тЖТ    43тЖТ    Es como un investigador que busca en Google, abre los mejores resultados,

    67тЖТ    67тЖТ        search_queries.append(f"{tema} que es definicion")

    84тЖТ    84тЖТ

    86тЖТ    86тЖТ            continue

    93тЖТ    93тЖТ
    94тЖТ    94тЖТ        # Visitar las URLs mas relevantes
    95тЖТ    95тЖТ        for url in urls_to_visit:
    96тЖТ    96тЖТ            if url in visited_urls:
    97тЖТ    97тЖТ                continue
    98тЖТ    98тЖТ            visited_urls.add(url)

   102тЖТ   102тЖТ                if content and len(content) > 100:
   103тЖТ   103тЖТ                    all_findings.append(f"--- Fuente: {url} ---\n{content}")
   104тЖТ   104тЖТ            except Exception as e:
   105тЖТ   105тЖТ                logger.debug(f"Error leyendo {url}: {e}")
   106тЖТ   106тЖТ
   107тЖТ   107тЖТ        # No hacer demasiadas busquedas seguidas
   108тЖТ   108тЖТ        if round_num < min(max_rounds, len(search_queries)) - 1:
   109тЖТ   109тЖТ            time.sleep(1)
   110тЖТ   110тЖТ
   111тЖТ   111тЖТ    if not all_findings:
   112тЖТ   112тЖТ        return f"No se encontraron resultados para '{tema}'. Intenta con otro termino."

   161тЖТ   161тЖТdef _extract_urls(text: str) -> list:
   162тЖТ   162тЖТ    """Extrae URLs de un texto."""
   163тЖТ   163тЖТ    url_pattern = r'https?://[^\s<>"\')\]]+'
   164тЖТ   164тЖТ    urls = re.findall(url_pattern, text)
   165тЖТ   165тЖТ    # Filtrar URLs no utiles
   166тЖТ   166тЖТ    skip_domains = ['duckduckgo.com', 'google.com', 'bing.com']
   167тЖТ   167тЖТ    filtered = []
   168тЖТ   168тЖТ    for url in urls:
   169тЖТ   169тЖТ        if not any(d in url for d in skip_domains):
   170тЖТ   170тЖТ            filtered.append(url.rstrip('.,;:'))
   171тЖТ   171тЖТ    return filtered[:10]

   261тЖТ   261тЖТdef editar_multiples(ediciones: str, crear_archivos: bool = True) -> str:

   265тЖТ   265тЖТ
   266тЖТ   266тЖТ    Args:

   289тЖТ   289тЖТ
   290тЖТ   290тЖТ        if not archivo:
   291тЖТ   291тЖТ            results.append(f"  [{i+1}] ERROR: No se especifico archivo")
   292тЖТ   292тЖТ            error_count += 1
   293тЖТ   293тЖТ            continue
   294тЖТ   294тЖТ
   295тЖТ   295тЖТ        # Resolver ruta
   296тЖТ   296тЖТ        ruta = _resolve_file_path(archivo)
   297тЖТ   297тЖТ
   298тЖТ   298тЖТ        # Si el archivo no existe

   306тЖТ   306тЖТ                    success_count += 1
   307тЖТ   307тЖТ                except Exception as e:
   308тЖТ   308тЖТ                    results.append(f"  [{i+1}] ERROR creando {archivo}: {e}")
   309тЖТ   309тЖТ                    error_count += 1
   310тЖТ   310тЖТ                continue
   311тЖТ   311тЖТ            else:
   312тЖТ   312тЖТ                results.append(f"  [{i+1}] ERROR: {archivo} no existe")
   313тЖТ   313тЖТ                error_count += 1
   314тЖТ   314тЖТ                continue
   315тЖТ   315тЖТ
   316тЖТ   316тЖТ        # Leer archivo actual
   317тЖТ   317тЖТ        try:
   318тЖТ   318тЖТ            with open(ruta, "r", encoding="utf-8", errors="replace") as f:
   319тЖТ   319тЖТ                content = f.read()
   320тЖТ   320тЖТ        except Exception as e:
   321тЖТ   321тЖТ            results.append(f"  [{i+1}] ERROR leyendo {archivo}: {e}")
   322тЖТ   322тЖТ            error_count += 1
   323тЖТ   323тЖТ            continue
   324тЖТ   324тЖТ
   325тЖТ   325тЖТ        # Realizar reemplazo
   326тЖТ   326тЖТ        if buscar:
   327тЖТ   327тЖТ            if buscar not in content:

   329тЖТ   329тЖТ                error_count += 1
   330тЖТ   330тЖТ                continue
   331тЖТ   331тЖТ
   332тЖТ   332тЖТ            count = content.count(buscar)
   333тЖТ   333тЖТ            replace_all = edit.get("reemplazar_todo", edit.get("replaceAll", False))
   334тЖТ   334тЖТ            if replace_all:
   335тЖТ   335тЖТ                content = content.replace(buscar, reemplazar)

   337тЖТ   337тЖТ            else:
   338тЖТ   338тЖТ                content = content.replace(buscar, reemplazar, 1)
   339тЖТ   339тЖТ                results.append(f"  [{i+1}] OK: {archivo} (1 de {count} reemplazos)")
   340тЖТ   340тЖТ        else:
   341тЖТ   341тЖТ            # Si no hay texto a buscar, agregar al final
   342тЖТ   342тЖТ            content += reemplazar

   345тЖТ   345тЖТ        # Escribir archivo

   374тЖТ   374тЖТdef generacion_batch(archivos: str) -> str:
   375тЖТ   375тЖТ    """Genera multiples archivos en una sola operacion. Ideal para crear
   376тЖТ   376тЖТ    estructuras de proyecto, templates, o conjuntos de archivos relacionados.
   377тЖТ   377тЖТ
   378тЖТ   378тЖТ    Args:
   379тЖТ   379тЖТ        archivos: Lista JSON de archivos: [{"ruta": "dir/archivo.py", "contenido": "..."}, ...]
   380тЖТ   380тЖТ    """
   381тЖТ   381тЖТ    try:
   382тЖТ   382тЖТ        file_list = json.loads(archivos)
   383тЖТ   383тЖТ    except json.JSONDecodeError as e:
   384тЖТ   384тЖТ        return f"ERROR: Formato JSON invalido: {e}"

   386тЖТ   386тЖТ    if not file_list:
   387тЖТ   387тЖТ        return "ERROR: Lista de archivos vacia."

   389тЖТ   389тЖТ    if len(file_list) > 30:
   390тЖТ   390тЖТ        return "ERROR: Maximo 30 archivos por operacion batch."
   391тЖТ   391тЖТ
   392тЖТ   392тЖТ    results = []
   393тЖТ   393тЖТ    created = 0
   394тЖТ   394тЖТ    skipped = 0

   396тЖТ   396тЖТ    for i, file_info in enumerate(file_list):
   397тЖТ   397тЖТ        ruta = file_info.get("ruta", file_info.get("path", file_info.get("file", "")))
   398тЖТ   398тЖТ        contenido = file_info.get("contenido", file_info.get("content", ""))
   399тЖТ   399тЖТ        sobreescribir = file_info.get("sobreescribir", file_info.get("overwrite", True))
   400тЖТ   400тЖТ
   401тЖТ   401тЖТ        if not ruta:
   402тЖТ   402тЖТ            results.append(f"  [{i+1}] ERROR: No se especifico ruta")
   403тЖТ   403тЖТ            continue
   404тЖТ   404тЖТ
   405тЖТ   405тЖТ        # Resolver ruta
   406тЖТ   406тЖТ        full_path = _resolve_file_path(ruta)
   407тЖТ   407тЖТ
   408тЖТ   408тЖТ        # Verificar si ya existe
   409тЖТ   409тЖТ        if os.path.exists(full_path) and not sobreescribir:
   410тЖТ   410тЖТ            results.append(f"  [{i+1}] SKIP: {ruta} (ya existe)")
   411тЖТ   411тЖТ            skipped += 1
   412тЖТ   412тЖТ            continue
   413тЖТ   413тЖТ
   414тЖТ   414тЖТ        # Crear directorios necesarios
   415тЖТ   415тЖТ        dir_path = os.path.dirname(full_path)
   416тЖТ   416тЖТ        if dir_path:
   417тЖТ   417тЖТ            os.makedirs(dir_path, exist_ok=True)
   418тЖТ   418тЖТ
   419тЖТ   419тЖТ        # Escribir archivo
   420тЖТ   420тЖТ        try:
   421тЖТ   421тЖТ            with open(full_path, "w", encoding="utf-8") as f:
   422тЖТ   422тЖТ                f.write(contenido)
   423тЖТ   423тЖТ            size = len(contenido)
   424тЖТ   424тЖТ            results.append(f"  [{i+1}] OK: {ruta} ({size} chars)")
   425тЖТ   425тЖТ            created += 1
   426тЖТ   426тЖТ        except Exception as e:
   427тЖТ   427тЖТ            results.append(f"  [{i+1}] ERROR: {ruta}: {e}")
   428тЖТ   428тЖТ
   429тЖТ   429тЖТ    summary = f"Generacion batch: {created} creados, {skipped} omitidos de {len(file_list)} totales\n"
   430тЖТ   430тЖТ    return summary + "\n".join(results)

   447тЖТ   447тЖТ

   453тЖТ   453тЖТ        ignorar_case: Ignorar mayusculas/minusculas (default True)

   455тЖТ   455тЖТ    """

   457тЖТ   457тЖТ    if not os.path.isdir(directorio):

   469тЖТ   469тЖТdef _search_ripgrep(patron, directorio, tipo_archivo, max_resultados, ignorar_case, contexto):
   470тЖТ   470тЖТ    """Busqueda usando ripgrep (rg)."""
   471тЖТ   471тЖТ    try:
   472тЖТ   472тЖТ        cmd = ["rg"]
   473тЖТ   473тЖТ        if ignorar_case:
   474тЖТ   474тЖТ            cmd.append("-i")
   475тЖТ   475тЖТ        cmd.extend(["-n", "--max-count", str(max_resultados)])
   476тЖТ   476тЖТ        if contexto > 0:
   477тЖТ   477тЖТ            cmd.extend(["-C", str(contexto)])
   478тЖТ   478тЖТ        if tipo_archivo:
   479тЖТ   479тЖТ            cmd.extend(["--glob", f"*{tipo_archivo}"])
   480тЖТ   480тЖТ        cmd.extend([patron, directorio])
   481тЖТ   481тЖТ
   482тЖТ   482тЖТ        result = subprocess.run(cmd, capture_output=True, text=True, timeout=30)
   483тЖТ   483тЖТ
   484тЖТ   484тЖТ        if result.returncode == 0 and result.stdout.strip():
   485тЖТ   485тЖТ            lines = result.stdout.strip().split('\n')
   486тЖТ   486тЖТ            if len(lines) > max_resultados:
   487тЖТ   487тЖТ                lines = lines[:max_resultados]
   488тЖТ   488тЖТ                lines.append(f"... ({len(result.stdout.strip().split(chr(10)))} resultados totales, mostrando {max_resultados})")
   489тЖТ   489тЖТ
   490тЖТ   490тЖТ            output = "\n".join(lines)
   491тЖТ   491тЖТ            if len(output) > MAX_TOOL_OUTPUT:
   492тЖТ   492тЖТ                output = output[:MAX_TOOL_OUTPUT] + "\n... [truncado]"
   493тЖТ   493тЖТ            return f"Resultados de busqueda '{patron}':\n{output}"
   494тЖТ   494тЖТ        elif result.returncode == 1:
   495тЖТ   495тЖТ            return f"No se encontraron coincidencias para '{patron}' en {directorio}"
   496тЖТ   496тЖТ        else:
   497тЖТ   497тЖТ            return None  # rg fallo, usar fallback
   498тЖТ   498тЖТ
   499тЖТ   499тЖТ    except FileNotFoundError:
   500тЖТ   500тЖТ        return None  # rg no instalado
   501тЖТ   501тЖТ    except subprocess.TimeoutExpired:
   502тЖТ   502тЖТ        return "ERROR: Busqueda excedio el timeout."
   503тЖТ   503тЖТ    except Exception:
   504тЖТ   504тЖТ        return None

   507тЖТ   507тЖТdef _search_python(patron, directorio, tipo_archivo, max_resultados, ignorar_case, contexto):
   508тЖТ   508тЖТ    """Busqueda fallback usando Python puro."""
   509тЖТ   509тЖТ    flags = re.IGNORECASE if ignorar_case else 0
   510тЖТ   510тЖТ    try:
   511тЖТ   511тЖТ        pattern = re.compile(patron, flags)
   512тЖТ   512тЖТ    except re.error as e:
   513тЖТ   513тЖТ        return f"ERROR: Patron regex invalido: {e}"
   514тЖТ   514тЖТ
   515тЖТ   515тЖТ    # Directorios a ignorar
   516тЖТ   516тЖТ    skip_dirs = {'.git', 'node_modules', '__pycache__', '.venv', 'venv', 'dist', 'build', '.next', '.cache'}
   517тЖТ   517тЖТ
   518тЖТ   518тЖТ    results = []
   519тЖТ   519тЖТ    count = 0
   520тЖТ   520тЖТ
   521тЖТ   521тЖТ    for root, dirs, files in os.walk(directorio):
   522тЖТ   522тЖТ        # Filtrar directorios
   523тЖТ   523тЖТ        dirs[:] = [d for d in dirs if d not in skip_dirs]
   524тЖТ   524тЖТ
   525тЖТ   525тЖТ        for fname in files:
   526тЖТ   526тЖТ            # Filtrar por extension
   527тЖТ   527тЖТ            if tipo_archivo and not fname.endswith(tipo_archivo):
   528тЖТ   528тЖТ                continue
   529тЖТ   529тЖТ
   530тЖТ   530тЖТ            fpath = os.path.join(root, fname)
   531тЖТ   531тЖТ
   532тЖТ   532тЖТ            # Solo leer archivos de texto
   533тЖТ   533тЖТ            try:
   534тЖТ   534тЖТ                with open(fpath, "r", encoding="utf-8", errors="replace") as f:
   535тЖТ   535тЖТ                    lines = f.readlines()
   536тЖТ   536тЖТ            except Exception:
   537тЖТ   537тЖТ                continue
   538тЖТ   538тЖТ
   539тЖТ   539тЖТ            for line_num, line in enumerate(lines, 1):
   540тЖТ   540тЖТ                if pattern.search(line):
   541тЖТ   541тЖТ                    # Mostrar con contexto
   542тЖТ   542тЖТ                    context_lines = []
   543тЖТ   543тЖТ                    start = max(0, line_num - 1 - contexto)
   544тЖТ   544тЖТ                    end = min(len(lines), line_num + contexto)
   545тЖТ   545тЖТ
   546тЖТ   546тЖТ                    for ctx_i in range(start, end):
   547тЖТ   547тЖТ                        prefix = ">>>" if ctx_i == line_num - 1 else "   "
   548тЖТ   548тЖТ                        context_lines.append(f"{prefix} {fpath}:{ctx_i+1}: {lines[ctx_i].rstrip()}")
   549тЖТ   549тЖТ
   550тЖТ   550тЖТ                    results.append("\n".join(context_lines))
   551тЖТ   551тЖТ                    count += 1
   552тЖТ   552тЖТ
   553тЖТ   553тЖТ                    if count >= max_resultados:
   554тЖТ   554тЖТ                        break
   555тЖТ   555тЖТ
   556тЖТ   556тЖТ            if count >= max_resultados:
   557тЖТ   557тЖТ                break
   558тЖТ   558тЖТ
   559тЖТ   559тЖТ        if count >= max_resultados:
   560тЖТ   560тЖТ            break
   561тЖТ   561тЖТ
   562тЖТ   562тЖТ    if not results:
   563тЖТ   563тЖТ        return f"No se encontraron coincidencias para '{patron}' en {directorio}"
   564тЖТ   564тЖТ
   565тЖТ   565тЖТ    output = "\n\n".join(results)
   566тЖТ   566тЖТ    if len(output) > MAX_TOOL_OUTPUT:
   567тЖТ   567тЖТ        output = output[:MAX_TOOL_OUTPUT] + "\n... [truncado]"
   568тЖТ   568тЖТ
   569тЖТ   569тЖТ    return f"Resultados de busqueda '{patron}' ({count} encontrados):\n{output}"

   575тЖТ   575тЖТ
   576тЖТ   576тЖТdef listar_glob(
   577тЖТ   577тЖТ    patron: str = "**/*",
   578тЖТ   578тЖТ    directorio: str = ".",
   579тЖТ   579тЖТ    solo_tipo: str = "todos",
   580тЖТ   580тЖТ    max_resultados: int = 100,
   581тЖТ   581тЖТ) -> str:
   582тЖТ   582тЖТ    """Lista archivos usando glob patterns (como **/*.py, **/test_*.js, etc.).
   583тЖТ   583тЖТ    Permite busqueda flexible de archivos por nombre, extension o patron.
   584тЖТ   584тЖТ
   585тЖТ   585тЖТ    Args:
   586тЖТ   586тЖТ        patron: Patron glob (ej: **/*.py, src/**/*.ts, **/test_*.js, *.md)
   587тЖТ   587тЖТ        directorio: Directorio base para la busqueda (default: actual)
   588тЖТ   588тЖТ        solo_tipo: Filtrar: todos, archivos, directorios
   589тЖТ   589тЖТ        max_resultados: Maximo resultados (default 100)

   592тЖТ   592тЖТ    if not os.path.isdir(directorio):
   593тЖТ   593тЖТ        return f"ERROR: Directorio no encontrado: {directorio}"
   594тЖТ   594тЖТ
   595тЖТ   595тЖТ    try:
   596тЖТ   596тЖТ        matches = glob_module.glob(os.path.join(directorio, patron), recursive=True)
   597тЖТ   597тЖТ    except Exception as e:

   599тЖТ   599тЖТ
   600тЖТ   600тЖТ    # Filtrar por tipo
   601тЖТ   601тЖТ    if solo_tipo == "archivos":
   602тЖТ   602тЖТ        matches = [m for m in matches if os.path.isfile(m)]
   603тЖТ   603тЖТ    elif solo_tipo == "directorios":
   604тЖТ   604тЖТ        matches = [m for m in matches if os.path.isdir(m)]
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ    # Ordenar por modificacion (mas recientes primero)
   607тЖТ   607тЖТ    try:
   608тЖТ   608тЖТ        matches.sort(key=lambda x: os.path.getmtime(x), reverse=True)
   609тЖТ   609тЖТ    except Exception:
   610тЖТ   610тЖТ        matches.sort()
   611тЖТ   611тЖТ
   612тЖТ   612тЖТ    if not matches:
   613тЖТ   613тЖТ        return f"No se encontraron archivos con patron '{patron}' en {directorio}"
   614тЖТ   614тЖТ
   615тЖТ   615тЖТ    # Limitar resultados
   616тЖТ   616тЖТ    total = len(matches)
   617тЖТ   617тЖТ    if len(matches) > max_resultados:
   618тЖТ   618тЖТ        matches = matches[:max_resultados]
   619тЖТ   619тЖТ
   620тЖТ   620тЖТ    # Formatear resultados
   621тЖТ   621тЖТ    results = [f"Archivos encontrados: {total} (mostrando {len(matches)})\n"]
   622тЖТ   622тЖТ
   623тЖТ   623тЖТ    for match in matches:
   624тЖТ   624тЖТ        rel_path = os.path.relpath(match, directorio)
   625тЖТ   625тЖТ        if os.path.isfile(match):
   626тЖТ   626тЖТ            try:
   627тЖТ   627тЖТ                size = os.path.getsize(match)
   628тЖТ   628тЖТ                size_str = f"{size:,} bytes" if size < 1024 else f"{size/1024:.1f} KB"
   629тЖТ   629тЖТ            except Exception:
   630тЖТ   630тЖТ                size_str = ""
   631тЖТ   631тЖТ            results.append(f"  {rel_path}  ({size_str})")
   632тЖТ   632тЖТ        else:
   633тЖТ   633тЖТ            results.append(f"  {rel_path}/")
   634тЖТ   634тЖТ
   635тЖТ   635тЖТ    output = "\n".join(results)
   636тЖТ   636тЖТ    if len(output) > MAX_TOOL_OUTPUT:
   637тЖТ   637тЖТ        output = output[:MAX_TOOL_OUTPUT] + "\n... [truncado]"
   638тЖТ   638тЖТ
   639тЖТ   639тЖТ    return output

   682тЖТ   682тЖТ        return f"ERROR: El directorio {project_dir} ya existe. Elige otro nombre o elimina el existente."

   684тЖТ   684тЖТ    ext = ".ts" if use_typescript else ".js"
   685тЖТ   685тЖТ    extx = ".tsx" if use_typescript else ".jsx"
   686тЖТ   686тЖТ
   687тЖТ   687тЖТ    # Generar archivos segun tipo
   688тЖТ   688тЖТ    files = {}
   689тЖТ   689тЖТ
   690тЖТ   690тЖТ    if tipo == "nextjs":

   692тЖТ   692тЖТ    elif tipo == "react":
   693тЖТ   693тЖТ        files = _generate_react_project(nombre, ext, extx, use_typescript, use_tailwind)
   694тЖТ   694тЖТ    elif tipo == "vue":

   696тЖТ   696тЖТ    elif tipo == "express":
   697тЖТ   697тЖТ        files = _generate_express_project(nombre, ext, use_typescript)
   698тЖТ   698тЖТ    elif tipo == "static":

   700тЖТ   700тЖТ

   704тЖТ   704тЖТ
   705тЖТ   705тЖТ    for rel_path, content in files.items():
   706тЖТ   706тЖТ        full_path = os.path.join(project_dir, rel_path)
   707тЖТ   707тЖТ        os.makedirs(os.path.dirname(full_path), exist_ok=True)
   708тЖТ   708тЖТ

   722тЖТ   722тЖТ

   728тЖТ   728тЖТ            f"  cd {project_dir}\n"

   730тЖТ   730тЖТ        )
