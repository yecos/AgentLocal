    30→from collections import OrderedDict
    31→
    32→
    33→class LRUCache:
    34→    """Cache LRU real usando OrderedDict. Reemplaza el FIFO del v13."""
    35→
    36→    def __init__(self, maxsize=200):
    37→        self._cache = OrderedDict()
    38→        self._maxsize = maxsize
    39→
    40→    def get(self, key):
    41→        if key in self._cache:
    42→            self._cache.move_to_end(key)  # Mas reciente al final
    43→            return self._cache[key]
    44→        return None
    45→
    46→    def put(self, key, value):
    47→        if key in self._cache:
    48→            self._cache.move_to_end(key)
    49→            self._cache[key] = value  # Actualizar valor existente
    50→        else:
    51→            if len(self._cache) >= self._maxsize:
    52→                self._cache.popitem(last=False)  # Elimina el mas viejo
    53→            self._cache[key] = value
    54→
    55→    def __len__(self):
    56→        return len(self._cache)
    57→
    58→    def clear(self):
    59→        self._cache.clear()
    60→
    61→
    62→# Instancia global del cache de embeddings
    63→embed_cache = LRUCache(maxsize=200)

   118→    def detect_models(self):
   119→        """Detecta el mejor modelo disponible. Se ejecuta 1 vez."""
   120→        if self._detected:
   121→            return self.model
   122→
   123→        # Intentar cargar cache persistente primero
   124→        cached = self._load_connection_cache()
   125→        if cached:
   126→            self.model = cached.get("model")
   127→            self.host = cached.get("host")
   128→            self.method = cached.get("method")
   129→            logger.info(f"Cache de conexion cargado: {self.model}@{self.host}")
   130→            # IMPORTANTE: Verificar que el modelo cacheado siga instalado
   131→            available_check = self._fetch_available_models(refresh=True)
   132→            if self.model and self.model not in available_check:
   133→                logger.warning(f"Modelo cacheado '{self.model}' ya no esta instalado. Reseteando cache.")
   134→                self.model = None
   135→                self.host = None
   136→                self.method = None
   137→                # Borrar cache invalido
   138→                try:
   139→                    if os.path.exists(CONNECTION_CACHE_FILE):
   140→                        os.remove(CONNECTION_CACHE_FILE)
   141→                except Exception as e:
   142→                    logger.debug(f"Error eliminando cache invalido: {e}")
   143→
   144→        available = self._fetch_available_models()
   145→        if not available:
   146→            self.model = "qwen2.5:14b"
   147→            self.fallback_model = "llama3.1:8b"
   148→            self.chat_model = "llama3.1:8b"
   149→            self.code_model = "qwen2.5:14b"
   150→            self._detected = True
   151→            return self.model
   152→
   153→        # Buscar el mejor modelo en orden de preferencia (match EXACTO primero)
   154→        for preferred in PREFERRED_MODELS:
   155→            # 1. Match exacto: "qwen3:4b" == "qwen3:4b"
   156→            for avail in available:
   157→                if avail == preferred or avail.startswith(preferred + ":"):
   158→                    self.model = avail
   159→                    break
   160→            if not self.model:
   161→                # 2. Match parcial: modelo base coincide ("qwen3" en "qwen3:30b-a3b")
   162→                #    PERO solo si no hay ya un match exacto de otro preferred
   163→                preferred_base = preferred.split(":")[0]
   164→                for avail in available:
   165→                    if avail.split(":")[0] == preferred_base:
   166→                        self.model = avail
   167→                        break
   168→            if self.model:
   169→                # El fallback es el segundo mejor
   170→                for fb in PREFERRED_MODELS:
   171→                    if fb != preferred:
   172→                        for avail2 in available:
   173→                            if avail2 == fb or avail2.startswith(fb + ":"):
   174→                                self.fallback_model = avail2
   175→                                break
   176→                        if self.fallback_model:
   177→                            break
   178→                if not self.fallback_model:
   179→                    # Buscar fallback por modelo base
   180→                    for fb in PREFERRED_MODELS:
   181→                        if fb != preferred:
   182→                            fb_base = fb.split(":")[0]
   183→                            for avail2 in available:
   184→                                if avail2.split(":")[0] == fb_base:
   185→                                    self.fallback_model = avail2
   186→                                    break
   187→                            if self.fallback_model:
   188→                                break
   189→                if not self.fallback_model:
   190→                    self.fallback_model = available[0] if len(available) > 1 else self.model
   191→                break
   192→
   193→        if not self.model:
   194→            self.model = available[0]
   195→            self.fallback_model = available[1] if len(available) > 1 else available[0]
   196→
   197→        # Detectar modelos especializados
   198→        self._detect_specialized_models(available)
   199→
   200→        # Detectar modelo de embeddings
   201→        self._detect_embed_model(available)
   202→
   203→        self._detected = True
   204→        logger.info(f"Modelo principal: {self.model}, Fallback: {self.fallback_model}")
   205→        logger.info(f"Chat: {self.chat_model}, Code: {self.code_model}, Embed: {self.embed_model}")
   206→        return self.model

   208→    def _detect_specialized_models(self, available):
   209→        """Detecta modelos para chat rapido y codigo potente."""
   210→        available_lower = [m.lower() for m in available]
   211→        # Chat model: el mas rapido
   212→        for pattern in CHAT_MODEL_PATTERNS:
   213→            for i, al in enumerate(available_lower):
   214→                if pattern.lower() in al:
   215→                    self.chat_model = available[i]
   216→                    break
   217→            if self.chat_model:
   218→                break
   219→        if not self.chat_model:
   220→            self.chat_model = self.fallback_model or self.model
   221→
   222→        # Code model: el mas potente
   223→        for pattern in CODE_MODEL_PATTERNS:
   224→            for i, al in enumerate(available_lower):
   225→                if pattern.lower() in al:
   226→                    self.code_model = available[i]
   227→                    break
   228→            if self.code_model:
   229→                break
   230→        if not self.code_model:
   231→            self.code_model = self.model

   233→    def _detect_embed_model(self, available):
   234→        """Detecta que modelo de embeddings esta disponible.
   235→        v3: Guarda el modelo detectado en cache para consistencia entre sesiones.
   236→        """
   237→        # Primero intentar cargar modelo de embeddings desde cache
   238→        cached_embed = self._load_embed_model_cache()
   239→        if cached_embed:
   240→            # Verificar que el modelo cacheado siga instalado
   241→            available_lower = [m.lower() for m in available]
   242→            if any(cached_embed.lower() in al for al in available_lower):
   243→                self.embed_model = cached_embed
   244→                logger.info(f"Modelo de embeddings (desde cache): {cached_embed}")
   245→                return
   246→            else:
   247→                logger.warning(f"Modelo de embeddings cacheado '{cached_embed}' ya no esta instalado. Detectando nuevo...")
   248→        
   249→        available_lower = [m.lower() for m in available]
   250→        for candidate in EMBED_MODEL_CANDIDATES:
   251→            for al in available_lower:
   252→                if candidate in al:
   253→                    self.embed_model = candidate
   254→                    logger.info(f"Modelo de embeddings detectado: {candidate}")
   255→                    # Guardar en cache para consistencia
   256→                    self._save_embed_model_cache(candidate)
   257→                    return
   258→        self.embed_model = "nomic-embed-text"  # Default
   259→        self._save_embed_model_cache(self.embed_model)

   286→    def check_gpu_status(self):
   287→        """
   288→        Verifica si Ollama esta usando la GPU.
   289→        Retorna: True (GPU), False (CPU), None (no se pudo determinar)
   290→        """
   291→        if self._gpu_status is not None:
   292→            return self._gpu_status
   293→
   294→        try:
   295→            import subprocess
   296→            result = subprocess.run(
   297→                ["ollama", "ps"],
   298→                capture_output=True, text=True, timeout=5
   299→            )
   300→            output = result.stdout
   301→            if "100% CPU" in output:
   302→                self._gpu_status = False
   303→                logger.warning("GPU NO detectada por Ollama - corriendo en CPU (LENTO)")
   304→            elif "GPU" in output:
   305→                self._gpu_status = True
   306→                logger.info("Ollama esta usando GPU")
   307→            else:
   308→                # No hay modelo cargado, verificar con nvidia-smi
   309→                try:
   310→                    nvidia = subprocess.run(
   311→                        ["nvidia-smi", "--query-gpu=name", "--format=csv,noheader"],
   312→                        capture_output=True, text=True, timeout=5
   313→                    )
   314→                    if nvidia.returncode == 0 and nvidia.stdout.strip():
   315→                        logger.info(f"GPU NVIDIA detectada: {nvidia.stdout.strip()}")
   316→                        # GPU existe pero no sabemos si Ollama la usa
   317→                        self._gpu_status = None
   318→                    else:
   319→                        self._gpu_status = False
   320→                except Exception as e:
   321→                    logger.debug(f"Error verificando GPU con nvidia-smi: {e}")
   322→                    self._gpu_status = False
   323→        except Exception as e:
   324→            logger.debug(f"No se pudo verificar GPU: {e}")
   325→            self._gpu_status = None
   326→
   327→        return self._gpu_status

   373→    @timed("llm")
   374→    def generate(self, messages, tools=None, model_override=None, timeout_overwrite=None):
   375→        """
   376→        Genera respuesta del LLM. Usa cache de conexion persistente.
   377→        Retorna: str (texto) o dict (respuesta completa con tool_calls).
   378→        Retorna "" si todo falla.
   379→        """
   380→        self.detect_models()
   381→        model = model_override or self.model
   382→        timeout = timeout_overwrite or self._get_timeout(model)
   383→
   384→        errors = []
   385→
   386→        # ---- ESTRATEGIA 1: Probar conexion cacheada ----
   387→        if self.host and self.method:
   388→            result = self._try_method(self.host, self.method, model, messages, tools, timeout)
   389→            if result is not None:
   390→                return result
   391→            # Cache fallo, resetear
   392→            self.method = None
   393→            errors.append("cached connection failed")
   394→
   395→        # ---- ESTRATEGIA 2: Buscar conexion que funcione ----
   396→        models_to_try = [model]
   397→        fallback = self.fallback_model
   398→        if fallback and fallback != model:
   399→            models_to_try.append(fallback)
   400→
   401→        hosts = ['http://localhost:11434', 'http://127.0.0.1:11434']
   402→
   403→        for m in models_to_try:
   404→            for host in hosts:
   405→                for method in ['client', 'http']:
   406→                    result = self._try_method(host, method, m, messages, tools, timeout)
   407→                    if result is not None:
   408→                        # Guardar conexion exitosa
   409→                        self.host = host
   410→                        self.method = method
   411→                        self._save_connection_cache()
   412→                        logger.info(f"LLM conectado: {m}@{host} via {method}")
   413→                        return result
   414→                    errors.append(f"{m}@{host}/{method}: fallo")
   415→
   416→        # Si llegamos aqui, todo fallo
   417→        self._log_errors(errors)
   418→        return ""

   522→    def generate_stream(self, messages, tools=None, model_override=None):
   523→        """
   524→        Genera respuesta del LLM con streaming.
   525→        Yields: str (tokens de texto) o dict (resultado final con tool_calls).
   526→        El llamador itera sobre los chunks para streaming en tiempo real.
   527→        Retorna None si todo falla (el llamador debe usar generate() como fallback).
   528→        """
   529→        self.detect_models()
   530→        model = model_override or self.model
   531→
   532→        # Probar hosts en orden
   533→        hosts = [self.host] if self.host else ['http://localhost:11434', 'http://127.0.0.1:11434']
   534→        hosts = [h for h in hosts if h]  # Filtrar None
   535→        if not hosts:
   536→            hosts = ['http://localhost:11434', 'http://127.0.0.1:11434']
   537→
   538→        for host in hosts:
   539→            # Intentar streaming HTTP directo (mas rapido)
   540→            stream = self._try_stream_http(host, model, messages, tools)
   541→            if stream is not None:
   542→                yield from stream
   543→                return
   544→
   545→            # Intentar streaming via client
   546→            stream = self._try_stream_client(host, model, messages, tools)
   547→            if stream is not None:
   548→                yield from stream
   549→                return
   550→
   551→        # Ultimo fallback: sin streaming
   552→        result = self.generate(messages, tools=tools, model_override=model_override)
   553→        if result:
   554→            if isinstance(result, str):
   555→                yield result
   556→            elif isinstance(result, dict):
   557→                yield result
   558→            return
   559→        return

   561→    def _try_stream_http(self, host, model, messages, tools):
   562→        """Intenta streaming HTTP. Retorna generador o None si falla."""
   563→        try:
   564→            import urllib.request
   565→            payload = {
   566→                "model": model,

   569→            }

   574→                f"{host}/api/chat",

   579→            def _stream():

   603→                        if chunk.get("done", False):
   604→                            self.host = host

   606→                            self._save_connection_cache()

   608→                            if full_tool_calls:
   609→                                yield {
   610→                                    "message": {
   611→                                        "content": full_content,
   612→                                        "tool_calls": full_tool_calls

   617→            # Test rapido: verificar que podemos abrir la conexion

   683→    def _try_method(self, host, method, model, messages, tools, timeout, think=None):
   684→        """Intenta una combinacion de host/metodo/modelo. Retorna None si falla.
   685→
   686→        Args:
   687→            think: Si True, activa pensamiento nativo del modelo (qwen3, deepseek-r1).
   688→                   Si None, se detecta automaticamente segun el modelo.
   689→        """
   690→        # Auto-detectar soporte de think nativo
   691→        if think is None:
   692→            try:
   693→                from agent.deep_thinking import detect_native_thinking_support
   694→                think = detect_native_thinking_support(model)
   695→            except ImportError:
   696→                think = False
   697→
   698→        if method == 'client':
   699→            return self._try_client(host, model, messages, tools, timeout, think=think)
   700→        elif method == 'http':
   701→            return self._try_http(host, model, messages, tools, timeout, think=think)
   702→        return None

   704→    def _try_client(self, host, model, messages, tools, timeout, think=False):
   705→        """Intenta via ollama.Client. Soporta think nativo para modelos compatibles."""
   706→        try:
   707→            import ollama
   708→            client = self._get_or_create_client(host)
   709→
   710→            # Construir kwargs
   711→            kwargs = {"model": model, "messages": messages}
   712→            if tools:
   713→                kwargs["tools"] = tools
   714→            if think and not tools:
   715→                # Think nativo: solo sin tools (Ollama no soporta ambos)
   716→                kwargs["think"] = True
   717→
   718→            response = client.chat(**kwargs)
   719→
   720→            # Procesar respuesta con posible thinking
   721→            msg = response.get("message", response)
   722→            content = msg.get("content", "") if isinstance(msg, dict) else ""
   723→            thinking_content = msg.get("thinking", "") if isinstance(msg, dict) else ""
   724→
   725→            # Si hay thinking, guardarlo para referencia
   726→            if thinking_content:
   727→                self._last_thinking = thinking_content
   728→                logger.debug(f"Native thinking: {len(thinking_content)} chars")
   729→
   730→            if tools:
   731→                if content or msg.get("tool_calls"):
   732→                    return response
   733→            else:
   734→                if content:
   735→                    return content
   736→        except TypeError as e:
   737→            # Si el client no soporta 'think', reintentar sin él
   738→            if 'think' in str(e).lower():
   739→                logger.debug("Cliente no soporta param 'think', reintentando sin él")
   740→                try:
   741→                    import ollama as ollama_lib
   742→                    client = self._get_or_create_client(host)
   743→                    if tools:
   744→                        response = client.chat(model=model, messages=messages, tools=tools)
   745→                    else:
   746→                        response = client.chat(model=model, messages=messages)
   747→                    msg = response.get("message", response)
   748→                    content = msg.get("content", "") if isinstance(msg, dict) else ""
   749→                    if tools:
   750→                        if content or msg.get("tool_calls"):
   751→                            return response
   752→                    else:
   753→                        if content:
   754→                            return content
   755→                except Exception:
   756→                    pass
   757→        except Exception as e:
   758→            logger.debug(f"Client method failed: {e}")
   759→        return None

   830→    def get_embedding(self, text):
   831→        """Obtiene el embedding de un texto usando Ollama. Con cache LRU.
   832→        v3: Retry con modelo alternativo si falla el principal.
   833→        """
   834→        import hashlib
   835→        cache_key = hashlib.md5(text[:500].encode()).hexdigest()[:16]
   836→
   837→        # Cache hit
   838→        cached = embed_cache.get(cache_key)
   839→        if cached is not None:
   840→            return cached
   841→
   842→        self.detect_models()
   843→        
   844→        # Intentar con el modelo principal de embeddings
   845→        embedding = self._try_get_embedding(text, self.embed_model)
   846→        if embedding:
   847→            embed_cache.put(cache_key, embedding)
   848→            get_metrics().record_embedding_call()
   849→            return embedding
   850→        
   851→        # Fallback: intentar con otros modelos de embedding disponibles
   852→        for fallback_model in EMBED_MODEL_CANDIDATES:
   853→            if fallback_model == self.embed_model:
   854→                continue
   855→            embedding = self._try_get_embedding(text, fallback_model)
   856→            if embedding:
   857→                logger.warning(f"Embedding fallback exitoso con modelo: {fallback_model}")
   858→                # Actualizar modelo de embeddings al que funciona
   859→                self.embed_model = fallback_model
   860→                self._save_embed_model_cache(fallback_model)
   861→                embed_cache.put(cache_key, embedding)
   862→                get_metrics().record_embedding_call()
   863→                return embedding
   864→        
   865→        logger.warning("No se pudo obtener embedding con ningun modelo")
   866→        get_metrics().record_error("embedding")
   867→        return []

   889→    @staticmethod
   890→    def cosine_similarity(vec1, vec2):
   891→        """Calcula similitud coseno entre dos vectores.
   892→        Usa numpy si esta disponible (10-50x mas rapido en batch).
   893→        """
   894→        if not vec1 or not vec2 or len(vec1) != len(vec2):
   895→            return 0.0
   896→        try:
   897→            import numpy as np
   898→            a = np.array(vec1, dtype=np.float32)
   899→            b = np.array(vec2, dtype=np.float32)
   900→            norm_a = np.linalg.norm(a)
   901→            norm_b = np.linalg.norm(b)
   902→            if norm_a == 0 or norm_b == 0:
   903→                return 0.0
   904→            return float(np.dot(a, b) / (norm_a * norm_b))
   905→        except ImportError:
   906→            # Fallback: Python puro
   907→            dot_product = sum(a * b for a, b in zip(vec1, vec2))
   908→            norm1 = sum(a * a for a in vec1) ** 0.5
   909→            norm2 = sum(b * b for b in vec2) ** 0.5
   910→            if norm1 == 0 or norm2 == 0:
   911→                return 0.0
   912→            return dot_product / (norm1 * norm2)

   914→    @staticmethod
   915→    def cosine_similarity_batch(query_vec, vectors_dict):
   916→        """Calcula similitud coseno de un vector contra muchos vectores de una vez.
   917→        
   918→        Args:
   919→            query_vec: Lista[float] - vector de consulta
   920→            vectors_dict: Dict[id, Lista[float]] - diccionario de vectores
   921→            
   922→        Returns:
   923→            Dict[id, float] - similitud de cada vector vs query
   924→            
   925→        Usa numpy para calculo vectorizado (10-50x mas rapido que bucle Python).
   926→        Fallback a Python puro si numpy no esta disponible.
   927→        """
   928→        if not query_vec or not vectors_dict:
   929→            return {}
   930→
   931→        ids = list(vectors_dict.keys())
   932→        vecs = list(vectors_dict.values())
   933→
   934→        # Filtrar vectores de tamano incompatible
   935→        valid_pairs = [(eid, v) for eid, v in zip(ids, vecs)
   936→                       if v and len(v) == len(query_vec)]
   937→        if not valid_pairs:
   938→            return {}
   939→
   940→        valid_ids, valid_vecs = zip(*valid_pairs)
   941→
   942→        try:
   943→            import numpy as np
   944→            query = np.array(query_vec, dtype=np.float32)
   945→            matrix = np.array(valid_vecs, dtype=np.float32)  # shape: (N, D)
   946→
   947→            # Normalizar query
   948→            query_norm = np.linalg.norm(query)
   949→            if query_norm == 0:
   950→                return {eid: 0.0 for eid in valid_ids}
   951→            query_normalized = query / query_norm
   952→
   953→            # Normalizar filas de la matriz
   954→            row_norms = np.linalg.norm(matrix, axis=1, keepdims=True)
   955→            row_norms = np.where(row_norms == 0, 1.0, row_norms)  # Evitar div/0
   956→            matrix_normalized = matrix / row_norms
   957→
   958→            # Producto punto vectorizado: (D,) @ (N, D).T -> (N,)
   959→            similarities = matrix_normalized @ query_normalized
   960→
   961→            return {eid: float(sim) for eid, sim in zip(valid_ids, similarities)}
   962→
   963→        except ImportError:
   964→            # Fallback: Python puro (bucle)
   965→            results = {}
   966→            for eid, vec in valid_pairs:
   967→                dot_product = sum(a * b for a, b in zip(query_vec, vec))
   968→                norm1 = sum(a * a for a in query_vec) ** 0.5
   969→                norm2 = sum(b * b for b in vec) ** 0.5
   970→                if norm1 > 0 and norm2 > 0:
   971→                    results[eid] = dot_product / (norm1 * norm2)
   972→                else:
   973→                    results[eid] = 0.0
   974→            return results
