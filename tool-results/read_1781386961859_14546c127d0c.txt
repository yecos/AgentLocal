   191тЖТ   191тЖТ    def _compute_decay(self, created_at, now=None, metadata=None):
   192тЖТ   192тЖТ        """"
   193тЖТ   193тЖТ        Computa el factor de decaimiento temporal diferenciado por tipo.
   194тЖТ   194тЖТ        Los recuerdos recientes pesan mas que los viejos.
   195тЖТ   195тЖТ        El half-life varia segun tipo de contenido:
   196тЖТ   196тЖТ        - knowledge: 365 dias, correction: 180, lesson: 90
   197тЖТ   197тЖТ        - experience: 60, task: 14, conversation: 7
   198тЖТ   198тЖТ        Retorna un valor entre 0 y 1.
   199тЖТ   199тЖТ        """
   200тЖТ   200тЖТ        if not created_at:
   201тЖТ   201тЖТ            return 0.5
   202тЖТ   202тЖТ        try:
   203тЖТ   203тЖТ            created = datetime.fromisoformat(created_at)
   204тЖТ   204тЖТ            if now is None:
   205тЖТ   205тЖТ                now = datetime.now()
   206тЖТ   206тЖТ            days_old = (now - created).total_seconds() / 86400
   207тЖТ   207тЖТ
   208тЖТ   208тЖТ            # Determinar half-life segun tipo de contenido
   209тЖТ   209тЖТ            half_life = self.DEFAULT_DECAY_HALF_LIFE
   210тЖТ   210тЖТ            if metadata:
   211тЖТ   211тЖТ                content_type = metadata.get("type", "").lower()
   212тЖТ   212тЖТ                if content_type in self.DECAY_HALF_LIFE_BY_TYPE:
   213тЖТ   213тЖТ                    half_life = self.DECAY_HALF_LIFE_BY_TYPE[content_type]
   214тЖТ   214тЖТ
   215тЖТ   215тЖТ            # Decaimiento exponencial con half-life diferenciado
   216тЖТ   216тЖТ            import math
   217тЖТ   217тЖТ            decay = math.exp(-0.693 * days_old / half_life)
   218тЖТ   218тЖТ            return max(decay, 0.1)  # Minimo 10% de relevancia
   219тЖТ   219тЖТ        except Exception as e:
   220тЖТ   220тЖТ            logger.debug(f"Error computando decaimiento: {e}")
   221тЖТ   221тЖТ            return 0.5

   223тЖТ   223тЖТ    def _is_duplicate(self, text, threshold=0.95):
   224тЖТ   224тЖТ        """
   225тЖТ   225тЖТ        Verifica si un texto es duplicado semantico de uno existente.
   226тЖТ   226тЖТ        Usa embeddings para comparar.
   227тЖТ   227тЖТ        v3: Catch de errores de dimension para no crashear.
   228тЖТ   228тЖТ        """
   229тЖТ   229тЖТ        embedding = ollama.get_embedding(text)
   230тЖТ   230тЖТ        if not embedding:
   231тЖТ   231тЖТ            return False
   232тЖТ   232тЖТ
   233тЖТ   233тЖТ        # Verificar dimension antes de consultar
   234тЖТ   234тЖТ        if self._embedding_dim is not None and len(embedding) != self._embedding_dim:
   235тЖТ   235тЖТ            logger.debug(f"Dimension mismatch en _is_duplicate, saltando verificacion")
   236тЖТ   236тЖТ            return False
   237тЖТ   237тЖТ
   238тЖТ   238тЖТ        try:
   239тЖТ   239тЖТ            results = self._collection.query(
   240тЖТ   240тЖТ                query_embeddings=[embedding],
   241тЖТ   241тЖТ                n_results=1,
   242тЖТ   242тЖТ                include=["distances"]
   243тЖТ   243тЖТ            )
   244тЖТ   244тЖТ            if results and results["distances"] and results["distances"][0]:
   245тЖТ   245тЖТ                # ChromaDB con cosine space: distance < (1 - threshold) = duplicado
   246тЖТ   246тЖТ                distance = results["distances"][0][0]
   247тЖТ   247тЖТ                similarity = 1 - distance
   248тЖТ   248тЖТ                return similarity >= threshold
   249тЖТ   249тЖТ        except Exception as e:
   250тЖТ   250тЖТ            err_msg = str(e).lower()
   251тЖТ   251тЖТ            if "dimension" in err_msg or "dim" in err_msg:
   252тЖТ   252тЖТ                logger.warning(f"Dimension mismatch en _is_duplicate, recreando coleccion...")
   253тЖТ   253тЖТ                try:
   254тЖТ   254тЖТ                    self._handle_dimension_error(embedding)
   255тЖТ   255тЖТ                except Exception:
   256тЖТ   256тЖТ                    pass
   257тЖТ   257тЖТ            else:
   258тЖТ   258тЖТ                logger.debug(f"Error verificando duplicado semantico: {e}")
   259тЖТ   259тЖТ        return False

   261тЖТ   261тЖТ    def add(self, text, metadata=None, entry_id=None, skip_embedding=False):
   262тЖТ   262тЖТ        """Agrega un texto al vector store con su embedding.
   263тЖТ   263тЖТ        
   264тЖТ   264тЖТ        Args:
   265тЖТ   265тЖТ            skip_embedding: Si True, NO calcula embedding (mas rapido).
   266тЖТ   266тЖТ                           La entrada solo aparecera en busquedas por texto.
   267тЖТ   267тЖТ        v3: Catch-all de errores de dimension con reintento automatico.
   268тЖТ   268тЖТ        """
   269тЖТ   269тЖТ        if not entry_id:
   270тЖТ   270тЖТ            entry_id = hashlib.md5(text.encode()).hexdigest()[:12]
   271тЖТ   271тЖТ
   272тЖТ   272тЖТ        # Verificar duplicado existente en ChromaDB

   274тЖТ   274тЖТ            existing = self._collection.get(ids=[entry_id])
   275тЖТ   275тЖТ            if existing and existing["ids"]:
   276тЖТ   276тЖТ                return entry_id
   277тЖТ   277тЖТ        except Exception as e:
   278тЖТ   278тЖТ            logger.debug(f"Error verificando entrada existente en ChromaDB: {e}")
   279тЖТ   279тЖТ
   280тЖТ   280тЖТ        # Verificar duplicado semantico (solo si tenemos embedding y no skip)
   281тЖТ   281тЖТ        if not skip_embedding:
   282тЖТ   282тЖТ            try:
   283тЖТ   283тЖТ                if self._is_duplicate(text):
   284тЖТ   284тЖТ                    logger.debug(f"Texto duplicado semantico, saltando: {text[:50]}...")
   285тЖТ   285тЖТ                    return entry_id
   286тЖТ   286тЖТ            except Exception as e:
   287тЖТ   287тЖТ                logger.debug(f"Error verificando duplicado semantico: {e}")
   288тЖТ   288тЖТ
   289тЖТ   289тЖТ        now = datetime.now().isoformat()
   290тЖТ   290тЖТ
   291тЖТ   291тЖТ        # Metadatos con timestamp para decaimiento
   292тЖТ   292тЖТ        meta = metadata or {}
   293тЖТ   293тЖТ        meta["created"] = now
   294тЖТ   294тЖТ        meta["text_preview"] = text[:100]
   295тЖТ   295тЖТ
   296тЖТ   296тЖТ        # Si skip_embedding, guardar solo con metadatos (sin calculo de embedding)
   297тЖТ   297тЖТ        if skip_embedding:
   298тЖТ   298тЖТ            meta["no_embedding"] = True
   299тЖТ   299тЖТ            try:
   300тЖТ   300тЖТ                self._collection.add(
   301тЖТ   301тЖТ                    ids=[entry_id],
   302тЖТ   302тЖТ                    documents=[text[:500]],
   303тЖТ   303тЖТ                    metadatas=[meta]
   304тЖТ   304тЖТ                )
   305тЖТ   305тЖТ            except Exception as e:
   306тЖТ   306тЖТ                logger.debug(f"Error agregando sin embedding a ChromaDB: {e}")
   307тЖТ   307тЖТ            return entry_id
   308тЖТ   308тЖТ
   309тЖТ   309тЖТ        # Obtener embedding
   310тЖТ   310тЖТ        embedding = ollama.get_embedding(text)
   311тЖТ   311тЖТ
   312тЖТ   312тЖТ        if embedding:
   313тЖТ   313тЖТ            # Actualizar dimension si es la primera vez que obtenemos un embedding valido
   314тЖТ   314тЖТ            if self._embedding_dim is None:
   315тЖТ   315тЖТ                self._embedding_dim = len(embedding)
   316тЖТ   316тЖТ                logger.info(f"Dimension de embedding detectada dinamicamente: {self._embedding_dim}")
   317тЖТ   317тЖТ
   318тЖТ   318тЖТ            # Validar dimension del embedding antes de insertar
   319тЖТ   319тЖТ            if len(embedding) != self._embedding_dim:
   320тЖТ   320тЖТ                logger.warning(
   321тЖТ   321тЖТ                    f"Embedding con dimension incorrecta ({len(embedding)} vs {self._embedding_dim}). "
   322тЖТ   322тЖТ                    f"Actualizando dimension y recreando coleccion..."
   323тЖТ   323тЖТ                )
   324тЖТ   324тЖТ                self._handle_dimension_error(embedding)
   325тЖТ   325тЖТ
   326тЖТ   326тЖТ            # Intentar insertar con catch-all de errores de dimension
   327тЖТ   327тЖТ            for attempt in range(self.MAX_RECREATE_RETRIES + 1):
   328тЖТ   328тЖТ                try:
   329тЖТ   329тЖТ                    self._collection.add(
   330тЖТ   330тЖТ                        ids=[entry_id],
   331тЖТ   331тЖТ                        embeddings=[embedding],
   332тЖТ   332тЖТ                        documents=[text[:500]],
   333тЖТ   333тЖТ                        metadatas=[meta]
   334тЖТ   334тЖТ                    )
   335тЖТ   335тЖТ                    return entry_id

   420тЖТ   420тЖТ
   421тЖТ   421тЖТ        # Actualizar dimension si es la primera vez que obtenemos un embedding valido

   423тЖТ   423тЖТ            self._embedding_dim = len(query_embedding)

   435тЖТ   435тЖТ                logger.warning(f"Error recreando coleccion: {e}")
   436тЖТ   436тЖТ            return self._text_search(query, limit)
   437тЖТ   437тЖТ
   438тЖТ   438тЖТ        try:
   439тЖТ   439тЖТ            # Buscar mas candidatos de los necesarios para re-rankear con decaimiento
   440тЖТ   440тЖТ            n_candidates = min(limit * 3, 20)
   441тЖТ   441тЖТ            results = self._collection.query(
   442тЖТ   442тЖТ                query_embeddings=[query_embedding],
   443тЖТ   443тЖТ                n_results=n_candidates,
   444тЖТ   444тЖТ                include=["documents", "metadatas", "distances"]
   445тЖТ   445тЖТ            )
   446тЖТ   446тЖТ
   447тЖТ   447тЖТ            if not results or not results["ids"] or not results["ids"][0]:
   448тЖТ   448тЖТ                return self._text_search(query, limit)
   449тЖТ   449тЖТ
   450тЖТ   450тЖТ            scored = []
   451тЖТ   451тЖТ            now = datetime.now()
   452тЖТ   452тЖТ            for i, doc_id in enumerate(results["ids"][0]):
   453тЖТ   453тЖТ                distance = results["distances"][0][i]
   454тЖТ   454тЖТ                similarity = 1 - distance
   455тЖТ   455тЖТ
   456тЖТ   456тЖТ                # Aplicar decaimiento temporal
   457тЖТ   457тЖТ                meta = results["metadatas"][0][i] if results["metadatas"] else {}
   458тЖТ   458тЖТ                created_at = meta.get("created", "")
   459тЖТ   459тЖТ                decay = self._compute_decay(created_at, now, metadata=meta)
   460тЖТ   460тЖТ
   461тЖТ   461тЖТ                # Score final = similitud * decaimiento
   462тЖТ   462тЖТ                final_score = similarity * decay
   463тЖТ   463тЖТ
   464тЖТ   464тЖТ                if final_score >= min_similarity * 0.5:  # Umbral mas bajo tras decaimiento
   465тЖТ   465тЖТ                    scored.append({
   466тЖТ   466тЖТ                        "id": doc_id,
   467тЖТ   467тЖТ                        "text": results["documents"][0][i] if results["documents"] else "",
   468тЖТ   468тЖТ                        "metadata": meta,
   469тЖТ   469тЖТ                        "has_vector": True,
   470тЖТ   470тЖТ                        "score": round(final_score, 3),
   471тЖТ   471тЖТ                        "raw_similarity": round(similarity, 3),
   472тЖТ   472тЖТ                        "decay": round(decay, 3),
   473тЖТ   473тЖТ                    })
   474тЖТ   474тЖТ
   475тЖТ   475тЖТ            scored.sort(key=lambda x: x["score"], reverse=True)
   476тЖТ   476тЖТ            return scored[:limit]
   477тЖТ   477тЖТ
   478тЖТ   478тЖТ        except Exception as e:
   479тЖТ   479тЖТ            err_msg = str(e).lower()
   480тЖТ   480тЖТ            if "dimension" in err_msg or "dim" in err_msg:
   481тЖТ   481тЖТ                logger.warning(f"Dimension mismatch en query, recreando coleccion...")
   482тЖТ   482тЖТ                try:
   483тЖТ   483тЖТ                    self._handle_dimension_error(query_embedding)
   484тЖТ   484тЖТ                except Exception:
   485тЖТ   485тЖТ                    pass
   486тЖТ   486тЖТ            else:
   487тЖТ   487тЖТ                logger.warning(f"Error en busqueda ChromaDB: {e}")
   488тЖТ   488тЖТ            return self._text_search(query, limit)

   490тЖТ   490тЖТ    def _text_search(self, query, limit=5):
   491тЖТ   491тЖТ        """Busqueda por texto cuando no hay embeddings."""
   492тЖТ   492тЖТ        query_lower = query.lower()
   493тЖТ   493тЖТ        query_words = [w for w in query_lower.split() if len(w) > 3]
   494тЖТ   494тЖТ
   495тЖТ   495тЖТ        if not query_words:
   496тЖТ   496тЖТ            return []
   497тЖТ   497тЖТ
   498тЖТ   498тЖТ        try:
   499тЖТ   499тЖТ            # ChromaDB no tiene text search nativo, obtener todos y filtrar
   500тЖТ   500тЖТ            all_docs = self._collection.get(include=["documents", "metadatas"])
   501тЖТ   501тЖТ            results = []
   502тЖТ   502тЖТ
   503тЖТ   503тЖТ            for i, doc in enumerate(all_docs["documents"]):
   504тЖТ   504тЖТ                doc_lower = doc.lower()
   505тЖТ   505тЖТ                matches = sum(1 for w in query_words if w in doc_lower)
   506тЖТ   506тЖТ                if matches > 0:
   507тЖТ   507тЖТ                    score = matches / max(len(query_words), 1)
   508тЖТ   508тЖТ                    meta = all_docs["metadatas"][i] if all_docs["metadatas"] else {}
   509тЖТ   509тЖТ                    results.append({
   510тЖТ   510тЖТ                        "id": all_docs["ids"][i],
   511тЖТ   511тЖТ                        "text": doc,
   512тЖТ   512тЖТ                        "metadata": meta,
   513тЖТ   513тЖТ                        "has_vector": not meta.get("no_embedding", False),
   514тЖТ   514тЖТ                        "score": round(score, 3),
   515тЖТ   515тЖТ                    })
   516тЖТ   516тЖТ
   517тЖТ   517тЖТ            results.sort(key=lambda x: x["score"], reverse=True)
   518тЖТ   518тЖТ            return results[:limit]
   519тЖТ   519тЖТ        except Exception as e:
   520тЖТ   520тЖТ            logger.debug(f"Error en busqueda por texto ChromaDB: {e}")

   560тЖТ   560тЖТclass SimpleVectorStore(VectorStore):
   561тЖТ   561тЖТ    """
   562тЖТ   562тЖТ    Vector store casero (fallback cuando ChromaDB no esta instalado).
   563тЖТ   563тЖТ    Hereda de VectorStore y agrega decaimiento temporal + deduplicacion por texto.
   564тЖТ   564тЖТ    Solo sobreescribe los metodos que difieren del base.
   565тЖТ   565тЖТ    Elimina ~150 lineas de codigo duplicado vs la version anterior.
   566тЖТ   566тЖТ    """
   567тЖТ   567тЖТ
   568тЖТ   568тЖТ    # Decay diferenciado por tipo de contenido (en dias half-life)
   569тЖТ   569тЖТ    DECAY_HALF_LIFE_BY_TYPE = {
   570тЖТ   570тЖТ        "knowledge": 365, "correction": 180, "lesson": 90,
   571тЖТ   571тЖТ        "experience": 60, "task": 14, "conversation": 7, "note": 30,
   572тЖТ   572тЖТ    }
   573тЖТ   573тЖТ    DEFAULT_DECAY_HALF_LIFE = 30
   574тЖТ   574тЖТ
   575тЖТ   575тЖТ    def _compute_decay(self, created_at, metadata=None):
   576тЖТ   576тЖТ        """Computa el factor de decaimiento temporal diferenciado por tipo (0.1 - 1.0)."""
   577тЖТ   577тЖТ        if not created_at:
   578тЖТ   578тЖТ            return 0.5
   579тЖТ   579тЖТ        try:
   580тЖТ   580тЖТ            created = datetime.fromisoformat(created_at)
   581тЖТ   581тЖТ            days_old = (datetime.now() - created).total_seconds() / 86400
   582тЖТ   582тЖТ
   583тЖТ   583тЖТ            # Determinar half-life segun tipo de contenido
   584тЖТ   584тЖТ            half_life = self.DEFAULT_DECAY_HALF_LIFE
   585тЖТ   585тЖТ            if metadata:
   586тЖТ   586тЖТ                content_type = metadata.get("type", "").lower()
   587тЖТ   587тЖТ                if content_type in self.DECAY_HALF_LIFE_BY_TYPE:
   588тЖТ   588тЖТ                    half_life = self.DECAY_HALF_LIFE_BY_TYPE[content_type]
   589тЖТ   589тЖТ
   590тЖТ   590тЖТ            import math
   591тЖТ   591тЖТ            decay = math.exp(-0.693 * days_old / half_life)
   592тЖТ   592тЖТ            return max(decay, 0.1)
   593тЖТ   593тЖТ        except Exception as e:
   594тЖТ   594тЖТ            logger.debug(f"Error computando decaimiento: {e}")
   595тЖТ   595тЖТ            return 0.5

   606тЖТ   606тЖТ
   607тЖТ   607тЖТ        # Deduplicacion basica por texto similar (100 chars lowercase)

   609тЖТ   609тЖТ        for entry in self.index:

   616тЖТ   616тЖТ    def search(self, query, limit=5, min_similarity=0.3):
   617тЖТ   617тЖТ        """Busca con decaimiento temporal. Sobreescribe para agregar decay al scoring."""
   618тЖТ   618тЖТ        if not self.index:
   619тЖТ   619тЖТ            return []
   620тЖТ   620тЖТ
   621тЖТ   621тЖТ        query_embedding = ollama.get_embedding(query)
   622тЖТ   622тЖТ        if not query_embedding:
   623тЖТ   623тЖТ            return self._text_search(query, limit)
   624тЖТ   624тЖТ
   625тЖТ   625тЖТ        candidates = self._pre_filter(query, max_candidates=50)
   626тЖТ   626тЖТ        if not candidates:
   627тЖТ   627тЖТ            candidates = self.index[:50]
   628тЖТ   628тЖТ
   629тЖТ   629тЖТ        candidate_ids = [c["id"] for c in candidates if c.get("has_vector")]
   630тЖТ   630тЖТ        vectors = self._load_vectors_for(candidate_ids)
   631тЖТ   631тЖТ
   632тЖТ   632тЖТ        # Scoring vectorizado con decaimiento temporal
   633тЖТ   633тЖТ        similarities = ollama.cosine_similarity_batch(query_embedding, vectors)
   634тЖТ   634тЖТ
   635тЖТ   635тЖТ        scored = []
   636тЖТ   636тЖТ        for entry in candidates:
   637тЖТ   637тЖТ            if not entry.get("has_vector") or entry["id"] not in similarities:
   638тЖТ   638тЖТ                continue
   639тЖТ   639тЖТ            raw_sim = similarities[entry["id"]]
   640тЖТ   640тЖТ            decay = self._compute_decay(entry.get("created", ""), metadata=entry.get("metadata"))
   641тЖТ   641тЖТ            final_score = raw_sim * decay
   642тЖТ   642тЖТ
   643тЖТ   643тЖТ            if final_score >= min_similarity * 0.5:
   644тЖТ   644тЖТ                scored.append({
   645тЖТ   645тЖТ                    **entry,
   646тЖТ   646тЖТ                    "score": round(final_score, 3),
   647тЖТ   647тЖТ                    "raw_similarity": round(raw_sim, 3),
   648тЖТ   648тЖТ                    "decay": round(decay, 3),
   649тЖТ   649тЖТ                })
   650тЖТ   650тЖТ
   651тЖТ   651тЖТ        scored.sort(key=lambda x: x["score"], reverse=True)

   654тЖТ   654тЖТ    def _text_search(self, query, limit=5):

   656тЖТ   656тЖТ        # Usar stemming del VectorStore base (que ya lo soporta)
   657тЖТ   657тЖТ        base_results = super()._text_search(query, limit=limit * 2)
   658тЖТ   658тЖТ        results = []
