    34тЖТ    34тЖТ    """Cache LRU real usando OrderedDict. Reemplaza el FIFO del v13."""
    35тЖТ    35тЖТ
    36тЖТ    36тЖТ    def __init__(self, maxsize=200):
    37тЖТ    37тЖТ        self._cache = OrderedDict()
    38тЖТ    38тЖТ        self._maxsize = maxsize
    39тЖТ    39тЖТ
    40тЖТ    40тЖТ    def get(self, key):
    41тЖТ    41тЖТ        if key in self._cache:
    42тЖТ    42тЖТ            self._cache.move_to_end(key)  # Mas reciente al final
    43тЖТ    43тЖТ            return self._cache[key]
    44тЖТ    44тЖТ        return None
    45тЖТ    45тЖТ
    46тЖТ    46тЖТ    def put(self, key, value):
    47тЖТ    47тЖТ        if key in self._cache:
    48тЖТ    48тЖТ            self._cache.move_to_end(key)
    49тЖТ    49тЖТ            self._cache[key] = value  # Actualizar valor existente
    50тЖТ    50тЖТ        else:
    51тЖТ    51тЖТ            if len(self._cache) >= self._maxsize:
    52тЖТ    52тЖТ                self._cache.popitem(last=False)  # Elimina el mas viejo
    53тЖТ    53тЖТ            self._cache[key] = value
    54тЖТ    54тЖТ
    55тЖТ    55тЖТ    def __len__(self):
    56тЖТ    56тЖТ        return len(self._cache)
    57тЖТ    57тЖТ

   172тЖТ   172тЖТ                        for avail2 in available:

   215тЖТ   215тЖТ                    self.chat_model = available[i]

   220тЖТ   220тЖТ            self.chat_model = self.fallback_model or self.model
   221тЖТ   221тЖТ
   222тЖТ   222тЖТ        # Code model: el mas potente

   227тЖТ   227тЖТ                    break

   408тЖТ   408тЖТ                        # Guardar conexion exitosa
   409тЖТ   409тЖТ                        self.host = host
   410тЖТ   410тЖТ                        self.method = method

   415тЖТ   415тЖТ

   523тЖТ   523тЖТ        """
   524тЖТ   524тЖТ        Genera respuesta del LLM con streaming.
   525тЖТ   525тЖТ        Yields: str (tokens de texto) o dict (resultado final con tool_calls).

   535тЖТ   535тЖТ        if not hosts:

   553тЖТ   553тЖТ        if result:

   555тЖТ   555тЖТ                yield result

   683тЖТ   683тЖТ    def _try_method(self, host, method, model, messages, tools, timeout, think=None):
   684тЖТ   684тЖТ        """Intenta una combinacion de host/metodo/modelo. Retorna None si falla.
   685тЖТ   685тЖТ
   686тЖТ   686тЖТ        Args:

   688тЖТ   688тЖТ                   Si None, se detecta automaticamente segun el modelo.
   689тЖТ   689тЖТ        """
   690тЖТ   690тЖТ        # Auto-detectar soporte de think nativo
   691тЖТ   691тЖТ        if think is None:
   692тЖТ   692тЖТ            try:
   693тЖТ   693тЖТ                from agent.deep_thinking import detect_native_thinking_support
   694тЖТ   694тЖТ                think = detect_native_thinking_support(model)
   695тЖТ   695тЖТ            except ImportError:

   697тЖТ   697тЖТ
   698тЖТ   698тЖТ        if method == 'client':
   699тЖТ   699тЖТ            return self._try_client(host, model, messages, tools, timeout, think=think)
   700тЖТ   700тЖТ        elif method == 'http':

   831тЖТ   831тЖТ        """Obtiene el embedding de un texto usando Ollama. Con cache LRU.

   836тЖТ   836тЖТ

   889тЖТ   889тЖТ    @staticmethod
   890тЖТ   890тЖТ    def cosine_similarity(vec1, vec2):
   891тЖТ   891тЖТ        """Calcula similitud coseno entre dos vectores.
   892тЖТ   892тЖТ        Usa numpy si esta disponible (10-50x mas rapido en batch).
   893тЖТ   893тЖТ        """
   894тЖТ   894тЖТ        if not vec1 or not vec2 or len(vec1) != len(vec2):
   895тЖТ   895тЖТ            return 0.0
   896тЖТ   896тЖТ        try:
   897тЖТ   897тЖТ            import numpy as np
   898тЖТ   898тЖТ            a = np.array(vec1, dtype=np.float32)
   899тЖТ   899тЖТ            b = np.array(vec2, dtype=np.float32)
   900тЖТ   900тЖТ            norm_a = np.linalg.norm(a)
   901тЖТ   901тЖТ            norm_b = np.linalg.norm(b)
   902тЖТ   902тЖТ            if norm_a == 0 or norm_b == 0:
   903тЖТ   903тЖТ                return 0.0
   904тЖТ   904тЖТ            return float(np.dot(a, b) / (norm_a * norm_b))
   905тЖТ   905тЖТ        except ImportError:
   906тЖТ   906тЖТ            # Fallback: Python puro
   907тЖТ   907тЖТ            dot_product = sum(a * b for a, b in zip(vec1, vec2))
   908тЖТ   908тЖТ            norm1 = sum(a * a for a in vec1) ** 0.5
   909тЖТ   909тЖТ            norm2 = sum(b * b for b in vec2) ** 0.5
   910тЖТ   910тЖТ            if norm1 == 0 or norm2 == 0:
   911тЖТ   911тЖТ                return 0.0
   912тЖТ   912тЖТ            return dot_product / (norm1 * norm2)

   914тЖТ   914тЖТ    @staticmethod
   915тЖТ   915тЖТ    def cosine_similarity_batch(query_vec, vectors_dict):
   916тЖТ   916тЖТ        """Calcula similitud coseno de un vector contra muchos vectores de una vez.
   917тЖТ   917тЖТ        
   918тЖТ   918тЖТ        Args:
   919тЖТ   919тЖТ            query_vec: Lista[float] - vector de consulta
   920тЖТ   920тЖТ            vectors_dict: Dict[id, Lista[float]] - diccionario de vectores
   921тЖТ   921тЖТ            
   922тЖТ   922тЖТ        Returns:
   923тЖТ   923тЖТ            Dict[id, float] - similitud de cada vector vs query
   924тЖТ   924тЖТ            
   925тЖТ   925тЖТ        Usa numpy para calculo vectorizado (10-50x mas rapido que bucle Python).
   926тЖТ   926тЖТ        Fallback a Python puro si numpy no esta disponible.
   927тЖТ   927тЖТ        """
   928тЖТ   928тЖТ        if not query_vec or not vectors_dict:
   929тЖТ   929тЖТ            return {}
   930тЖТ   930тЖТ
   931тЖТ   931тЖТ        ids = list(vectors_dict.keys())
   932тЖТ   932тЖТ        vecs = list(vectors_dict.values())
   933тЖТ   933тЖТ
   934тЖТ   934тЖТ        # Filtrar vectores de tamano incompatible
   935тЖТ   935тЖТ        valid_pairs = [(eid, v) for eid, v in zip(ids, vecs)
   936тЖТ   936тЖТ                       if v and len(v) == len(query_vec)]
   937тЖТ   937тЖТ        if not valid_pairs:
   938тЖТ   938тЖТ            return {}
   939тЖТ   939тЖТ
   940тЖТ   940тЖТ        valid_ids, valid_vecs = zip(*valid_pairs)
   941тЖТ   941тЖТ
   942тЖТ   942тЖТ        try:
   943тЖТ   943тЖТ            import numpy as np
   944тЖТ   944тЖТ            query = np.array(query_vec, dtype=np.float32)
   945тЖТ   945тЖТ            matrix = np.array(valid_vecs, dtype=np.float32)  # shape: (N, D)
   946тЖТ   946тЖТ
   947тЖТ   947тЖТ            # Normalizar query
   948тЖТ   948тЖТ            query_norm = np.linalg.norm(query)
   949тЖТ   949тЖТ            if query_norm == 0:
   950тЖТ   950тЖТ                return {eid: 0.0 for eid in valid_ids}
   951тЖТ   951тЖТ            query_normalized = query / query_norm
   952тЖТ   952тЖТ
   953тЖТ   953тЖТ            # Normalizar filas de la matriz
   954тЖТ   954тЖТ            row_norms = np.linalg.norm(matrix, axis=1, keepdims=True)
   955тЖТ   955тЖТ            row_norms = np.where(row_norms == 0, 1.0, row_norms)  # Evitar div/0
   956тЖТ   956тЖТ            matrix_normalized = matrix / row_norms
   957тЖТ   957тЖТ
   958тЖТ   958тЖТ            # Producto punto vectorizado: (D,) @ (N, D).T -> (N,)
   959тЖТ   959тЖТ            similarities = matrix_normalized @ query_normalized
   960тЖТ   960тЖТ
   961тЖТ   961тЖТ            return {eid: float(sim) for eid, sim in zip(valid_ids, similarities)}
   962тЖТ   962тЖТ
   963тЖТ   963тЖТ        except ImportError:
   964тЖТ   964тЖТ            # Fallback: Python puro (bucle)
   965тЖТ   965тЖТ            results = {}
   966тЖТ   966тЖТ            for eid, vec in valid_pairs:
   967тЖТ   967тЖТ                dot_product = sum(a * b for a, b in zip(query_vec, vec))
   968тЖТ   968тЖТ                norm1 = sum(a * a for a in query_vec) ** 0.5
   969тЖТ   969тЖТ                norm2 = sum(b * b for b in vec) ** 0.5
   970тЖТ   970тЖТ                if norm1 > 0 and norm2 > 0:
   971тЖТ   971тЖТ                    results[eid] = dot_product / (norm1 * norm2)
   972тЖТ   972тЖТ                else:
   973тЖТ   973тЖТ                    results[eid] = 0.0
   974тЖТ   974тЖТ            return results
