    61→function formatBytes(bytes: number): string {
    62→  if (bytes === 0) return "0 B";
    63→  const k = 1024;
    64→  const sizes = ["B", "KB", "MB", "GB", "TB"];
    65→  const i = Math.floor(Math.log(bytes) / Math.log(k));
    66→  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + " " + sizes[i];
    67→}
    68→
    69→function formatUptime(seconds: number): string {
    70→  const h = Math.floor(seconds / 3600);
    71→  const m = Math.floor((seconds % 3600) / 60);
    72→  const s = Math.floor(seconds % 60);
    73→  return `${h.toString().padStart(2, "0")}:${m.toString().padStart(2, "0")}:${s.toString().padStart(2, "0")}`;
    74→}

   164→// ─── Chat Message Component ──────────────────────────────────────────────────
   165→
   166→function ChatMessage({
   167→  msg,
   168→  isLastAssistant,
   169→  isLoading,
   170→  expandedThinking,
   171→  toggleThinking,
   172→}: {
   173→  msg: Message;
   174→  isLastAssistant: boolean;
   175→  isLoading: boolean;
   176→  expandedThinking: Record<string, boolean>;
   177→  toggleThinking: (id: string) => void;
   178→}) {
   179→  if (msg.role === "user") {
   180→    return (
   181→      <div className="flex justify-end animate-fade-in">
   182→        <div className="max-w-[70%] px-3 py-2 border border-[rgba(255,255,255,0.06)] text-[13px] leading-[1.7] text-[#e0e0e0]">
   183→          {msg.content}
   184→        </div>
   185→      </div>
   186→    );
   187→  }
   188→
   189→  const isStreaming = isLastAssistant && isLoading;
   190→  const cleanedThinking = msg.thinking ? cleanThinking(msg.thinking) : "";
   191→
   192→  return (
   193→    <div className="animate-fade-in">
   194→      <div className="max-w-[85%]">
   195→        <div className="border-l border-[rgba(0,212,255,0.2)] pl-3">
   196→          {/* Thinking section */}
   197→          {cleanedThinking && (
   198→            <div className="mb-2.5">
   199→              <button
   200→                onClick={() => toggleThinking(msg.id)}
   201→                className="flex items-center gap-1 text-[9px] text-[#555555] hover:text-[#888888] transition-colors duration-150"
   202→              >
   203→                {expandedThinking[msg.id] ? (
   204→                  <ChevronDown size={10} />
   205→                ) : (
   206→                  <ChevronRight size={10} />
   207→                )}
   208→                <span>thinking</span>
   209→                <span className="text-[#333333] ml-1">
   210→                  ({cleanedThinking.length} chars)
   211→                </span>
   212→              </button>
   213→              {expandedThinking[msg.id] && (
   214→                <div className="mt-1.5 text-[10px] leading-[1.6] text-[#3a3a3a] max-h-40 overflow-y-auto pr-2 whitespace-pre-wrap border-l border-[rgba(255,255,255,0.04)] pl-2">
   215→                  {cleanedThinking}
   216→                </div>
   217→              )}
   218→            </div>
   219→          )}
   220→
   221→          {/* Tool calls */}
   222→          {msg.toolCalls && msg.toolCalls.length > 0 && (
   223→            <div className="flex flex-wrap gap-1.5 mb-2">
   224→              {msg.toolCalls.map((tool, i) => (
   225→                <span
   226→                  key={i}
   227→                  className="text-[9px] px-1.5 py-0.5 border border-[rgba(0,212,255,0.15)] text-[rgba(0,212,255,0.5)]"
   228→                >
   229→                  [{tool}]
   230→                </span>
   231→              ))}
   232→            </div>
   233→          )}
   234→
   235→          {/* Content */}
   236→          <div className="text-[13px] leading-[1.7] text-[#e0e0e0] whitespace-pre-wrap">
   237→            {msg.content}
   238→            {/* Streaming cursor */}
   239→            {isStreaming && !msg.content && (
   240→              <span className="inline-block w-[6px] h-[14px] bg-[rgba(0,212,255,0.5)] cursor-blink ml-0.5" />
   241→            )}
   242→            {isStreaming && msg.content && (
   243→              <span className="inline-block w-[5px] h-[13px] bg-[rgba(0,212,255,0.5)] cursor-blink ml-0.5 align-middle" />
   244→            )}
   245→          </div>
   246→
   247→          {/* Response metadata */}
   248→          {msg.responseTime && !isStreaming && (
   249→            <div className="mt-2 text-[9px] text-[#333333] flex items-center gap-2">
   250→              <span>{msg.responseTime}ms</span>
   251→              {msg.tokenCount ? (
   252→                <>
   253→                  <span className="text-[#222222]">·</span>
   254→                  <span>~{msg.tokenCount} tokens</span>
   255→                </>
   256→              ) : null}
   257→            </div>
   258→          )}
   259→        </div>
   260→      </div>
   261→    </div>
   262→  );
   263→}

   267→export default function ZAIInterface() {
   268→  // State
   269→  const [messages, setMessages] = useState<Message[]>([]);
   270→  const [input, setInput] = useState("");
   271→  const [isLoading, setIsLoading] = useState(false);
   272→  const [selectedModel, setSelectedModel] = useState<string>("qwen3:4b");
   273→  const [status, setStatus] = useState<SystemStatus>({
   274→    connected: false,
   275→    models: [],
   276→    modelCount: 0,
   277→  });
   278→  const [uptime, setUptime] = useState(0);
   279→  const [sessionStats, setSessionStats] = useState<SessionStats>({
   280→    messageCount: 0,
   281→    toolsUsed: 0,
   282→    avgResponseTime: 0,
   283→    totalTokens: 0,
   284→  });
   285→  const [sidebarOpen, setSidebarOpen] = useState(true);
   286→  const [expandedThinking, setExpandedThinking] = useState<Record<string, boolean>>({});
   287→  const [isMobile, setIsMobile] = useState(false);
   288→
   289→  // Refs
   290→  const chatEndRef = useRef<HTMLDivElement>(null);
   291→  const inputRef = useRef<HTMLTextAreaElement>(null);
   292→  const responseTimesRef = useRef<number[]>([]);
   293→
   294→  // ─── Detect Mobile ──────────────────────────────────────────────────────
   295→
   296→  useEffect(() => {
   297→    const checkMobile = () => setIsMobile(window.innerWidth < 768);
   298→    checkMobile();
   299→    window.addEventListener("resize", checkMobile);
   300→    return () => window.removeEventListener("resize", checkMobile);
   301→  }, []);
   302→
   303→  // Close sidebar by default on mobile
   304→  useEffect(() => {
   305→    if (isMobile) setSidebarOpen(false);
   306→  }, [isMobile]);
   307→
   308→  // ─── Fetch Status ────────────────────────────────────────────────────────
   309→
   310→  const fetchStatus = useCallback(async () => {
   311→    try {
   312→      const res = await fetch("/api/status");
   313→      const data = await res.json();
   314→      setStatus(data);
   315→      if (data.connected && data.models.length > 0) {
   316→        setSelectedModel((prev) => {
   317→          if (data.models.find((m: OllamaModel) => m.name === prev)) return prev;
   318→          return data.models[0].name;
   319→        });
   320→      }
   321→    } catch {
   322→      setStatus({ connected: false, models: [], modelCount: 0, error: "Failed to fetch" });
   323→    }
   324→  }, []);
   325→
   326→  // ─── Effects ─────────────────────────────────────────────────────────────
   327→
   328→  useEffect(() => {
   329→    fetchStatus();
   330→    const interval = setInterval(fetchStatus, 10000);
   331→    return () => clearInterval(interval);
   332→  }, [fetchStatus]);
   333→
   334→  useEffect(() => {
   335→    const timer = setInterval(() => setUptime((prev) => prev + 1), 1000);
   336→    return () => clearInterval(timer);
   337→  }, []);
   338→
   339→  useEffect(() => {
   340→    chatEndRef.current?.scrollIntoView({ behavior: "smooth" });
   341→  }, [messages]);
   342→
   343→  // ─── Auto-select first model ────────────────────────────────────────────
   344→
   345→  const lastAssistantId = useMemo(() => {
   346→    for (let i = messages.length - 1; i >= 0; i--) {
   347→      if (messages[i].role === "assistant") return messages[i].id;
   348→    }
   349→    return null;
   350→  }, [messages]);
   351→
   352→  // ─── Send Message ───────────────────────────────────────────────────────
   353→
   354→  const sendMessage = async () => {
   355→    if (!input.trim() || isLoading) return;
   356→
   357→    const userMessage: Message = {
   358→      id: generateId(),
   359→      role: "user",
   360→      content: input.trim(),
   361→      timestamp: Date.now(),
   362→    };
   363→
   364→    const assistantId = generateId();
   365→    const assistantMessage: Message = {
   366→      id: assistantId,
   367→      role: "assistant",
   368→      content: "",
   369→      timestamp: Date.now(),
   370→      isStreaming: true,
   371→    };
   372→
   373→    setMessages((prev) => [...prev, userMessage, assistantMessage]);
   374→    setInput("");
   375→    setIsLoading(true);
   376→
   377→    // Reset textarea height
   378→    if (inputRef.current) {
   379→      inputRef.current.style.height = "auto";
   380→    }
   381→
   382→    const startTime = Date.now();
   383→
   384→    const ollamaMessages = [...messages, userMessage].map((m) => ({
   385→      role: m.role,
   386→      content: m.content,
   387→    }));
   388→
   389→    try {
   390→      const res = await fetch("/api/chat", {
   391→        method: "POST",
   392→        headers: { "Content-Type": "application/json" },
   393→        body: JSON.stringify({ messages: ollamaMessages, model: selectedModel }),
   394→      });
   395→
   396→      if (!res.ok) {
   397→        const errorData = await res.json().catch(() => ({ error: "Unknown error" }));
   398→        setMessages((prev) =>
   399→          prev.map((m) =>
   400→            m.id === assistantId
   401→              ? { ...m, content: `Error: ${errorData.error || "Unknown error"}`, isStreaming: false, responseTime: Date.now() - startTime }
   402→              : m
   403→          )
   404→        );
   405→        setIsLoading(false);
   406→        return;
   407→      }
   408→
   409→      const reader = res.body?.getReader();
   410→      if (!reader) {
   411→        setIsLoading(false);
   412→        return;
   413→      }
   414→
   415→      const decoder = new TextDecoder();
   416→      let fullContent = "";
   417→      let thinkingContent = "";
   418→      let isThinking = false;
   419→      let toolCalls: string[] = [];
   420→      let tokenCount = 0;
   421→
   422→      while (true) {
   423→        const { done, value } = await reader.read();
   424→        if (done) break;
   425→
   426→        const chunk = decoder.decode(value, { stream: true });
   427→        const lines = chunk.split("\n").filter((line) => line.startsWith("data: "));
   428→
   429→        for (const line of lines) {
   430→          const data = line.slice(6);
   431→          if (data === "[DONE]") continue;
   432→
   433→          try {
   434→            const parsed = JSON.parse(data);
   435→            if (parsed.error) {
   436→              fullContent += `\nError: ${parsed.error}`;
   437→              break;
   438→            }
   439→
   440→            const content = parsed.message?.content || "";
   441→            tokenCount++;
   442→
   443→            if (content.includes("<think")) {
   444→              isThinking = true;
   445→            }
   446→
   447→            if (isThinking) {
   448→              thinkingContent += content;
   449→              if (content.includes("</think")) {
   450→                isThinking = false;
   451→              }
   452→            } else {
   453→              fullContent += content;
   454→            }
   455→
   456→            // Detect tool calls
   457→            const toolMatches = content.match(/\[([a-z_]+)\]/g);
   458→            if (toolMatches) {
   459→              toolCalls = [...new Set([...toolCalls, ...toolMatches.map((t) => t.slice(1, -1))])];
   460→            }
   461→          } catch {
   462→            // Skip malformed JSON
   463→          }
   464→        }
   465→
   466→        setMessages((prev) =>
   467→          prev.map((m) =>
   468→            m.id === assistantId
   469→              ? {
   470→                  ...m,
   471→                  content: fullContent,
   472→                  thinking: thinkingContent || undefined,
   473→                  toolCalls: toolCalls.length > 0 ? toolCalls : undefined,
   474→                  tokenCount,
   475→                }
   476→              : m
   477→          )
   478→        );
   479→      }
   480→
   481→      const responseTime = Date.now() - startTime;
   482→      responseTimesRef.current.push(responseTime);
   483→
   484→      setMessages((prev) =>
   485→        prev.map((m) =>
   486→          m.id === assistantId ? { ...m, responseTime, isStreaming: false } : m
   487→        )
   488→      );
   489→
   490→      setSessionStats((prev) => ({
   491→        messageCount: prev.messageCount + 1,
   492→        toolsUsed: prev.toolsUsed + toolCalls.length,
   493→        avgResponseTime:
   494→          responseTimesRef.current.length > 0
   495→            ? Math.round(responseTimesRef.current.reduce((a, b) => a + b, 0) / responseTimesRef.current.length)
   496→            : 0,
   497→        totalTokens: prev.totalTokens + tokenCount,
   498→      }));
   499→    } catch (error) {
   500→      const errMsg = error instanceof Error ? error.message : "Unknown error";
   501→      setMessages((prev) =>
   502→        prev.map((m) =>
   503→          m.id === assistantId
   504→            ? { ...m, content: `Connection error: ${errMsg}`, isStreaming: false, responseTime: Date.now() - startTime }
   505→            : m
   506→        )
   507→      );
   508→    } finally {
   509→      setIsLoading(false);
   510→      inputRef.current?.focus();
   511→    }
   512→  };
   513→
   514→  // ─── Handle Key Press ───────────────────────────────────────────────────
   515→
   516→  const handleKeyDown = (e: React.KeyboardEvent) => {
   517→    if (e.key === "Enter" && !e.shiftKey) {
   518→      e.preventDefault();
   519→      sendMessage();
   520→    }
   521→  };
   522→
   523→  // ─── Toggle Thinking ────────────────────────────────────────────────────
   524→
   525→  const toggleThinking = (id: string) => {
   526→    setExpandedThinking((prev) => ({ ...prev, [id]: !prev[id] }));
   527→  };
   528→
   529→  // ─── Clear Chat ─────────────────────────────────────────────────────────
   530→
   531→  const clearChat = () => {
   532→    setMessages([]);
   533→    setSessionStats({ messageCount: 0, toolsUsed: 0, avgResponseTime: 0, totalTokens: 0 });
   534→    responseTimesRef.current = [];
   535→  };
   536→
   537→  // ─── Render ─────────────────────────────────────────────────────────────
   538→
   539→  return (
   540→    <div className="h-screen flex flex-col bg-[#000000] text-[#e0e0e0] overflow-hidden" style={{ fontFamily: "var(--font-geist-mono), 'JetBrains Mono', 'Fira Code', ui-monospace, monospace" }}>
   541→      {/* ─── Top Bar ─────────────────────────────────────────────────────── */}
   542→      <header className="flex items-center justify-between h-9 px-4 border-b border-[rgba(255,255,255,0.06)] shrink-0 select-none">
   543→        <div className="flex items-center gap-3">
   544→          <span className="text-[11px] tracking-[0.3em] text-[#e0e0e0] font-bold">ZAI</span>
   545→          <span className="text-[10px] text-[#222222]">│</span>
   546→          <span className="text-[10px] text-[#555555]">{selectedModel}</span>
   547→          <div className="flex items-center gap-1.5 ml-1">
   548→            <Circle
   549→              size={5}
   550→              className={status.connected ? "fill-[#00ff88] text-[#00ff88]" : "fill-[#ff3333] text-[#ff3333]"}
   551→            />
   552→            <span className="text-[9px] text-[#555555] tracking-wider">
   553→              {status.connected ? "CONNECTED" : "OFFLINE"}
   554→            </span>
   555→          </div>
   556→        </div>
   557→        <div className="flex items-center gap-2">
   558→          {messages.length > 0 && (
   559→            <button
   560→              onClick={clearChat}
   561→              className="text-[9px] text-[#444444] hover:text-[#666666] transition-colors px-2 py-0.5 border border-[rgba(255,255,255,0.04)] hover:border-[rgba(255,255,255,0.08)]"
   562→            >
   563→              CLEAR
   564→            </button>
   565→          )}
   566→          <button
   567→            onClick={() => setSidebarOpen(!sidebarOpen)}
   568→            className="p-1.5 text-[#555555] hover:text-[#e0e0e0] transition-colors duration-150"
   569→            aria-label={sidebarOpen ? "Close sidebar" : "Open sidebar"}
   570→          >
   571→            {sidebarOpen ? <PanelRightClose size={14} /> : <PanelRightOpen size={14} />}
   572→          </button>
   573→        </div>
   574→      </header>
   575→
   576→      {/* ─── Main Content ────────────────────────────────────────────────── */}
   577→      <div className="flex flex-1 overflow-hidden relative">
   578→        {/* ─── Chat Area ────────────────────────────────────────────────── */}
   579→        <main className="flex-1 flex flex-col min-w-0">
   580→          {/* Messages */}
