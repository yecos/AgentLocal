   721тЖТ   721тЖТ    def _execute_tool_calls(self, tool_calls, messages):
   722тЖТ   722тЖТ        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   723тЖТ   723тЖТ        from concurrent.futures import ThreadPoolExecutor, as_completed
   724тЖТ   724тЖТ
   725тЖТ   725тЖТ        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   726тЖТ   726тЖТ        filtered_calls = []
   727тЖТ   727тЖТ        for tc in tool_calls:
   728тЖТ   728тЖТ            tool_name = tc["name"]
   729тЖТ   729тЖТ            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   730тЖТ   730тЖТ            self._total_tool_calls += 1
   731тЖТ   731тЖТ
   732тЖТ   732тЖТ            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   733тЖТ   733тЖТ                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   734тЖТ   734тЖТ                break
   735тЖТ   735тЖТ
   736тЖТ   736тЖТ            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   737тЖТ   737тЖТ                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   738тЖТ   738тЖТ                continue
   739тЖТ   739тЖТ
   740тЖТ   740тЖТ            filtered_calls.append(tc)
   741тЖТ   741тЖТ
   742тЖТ   742тЖТ        tool_calls = filtered_calls
   743тЖТ   743тЖТ
   744тЖТ   744тЖТ        # Validar parametros de todos los tool calls primero
   745тЖТ   745тЖТ        for tc in tool_calls:
   746тЖТ   746тЖТ            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   747тЖТ   747тЖТ
   748тЖТ   748тЖТ        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   749тЖТ   749тЖТ        if len(tool_calls) == 1:
   750тЖТ   750тЖТ            return [self._execute_single_tool(tool_calls[0], messages)]
   751тЖТ   751тЖТ
   752тЖТ   752тЖТ        # Si hay multiples tools, ejecutar en paralelo
   753тЖТ   753тЖТ        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   754тЖТ   754тЖТ        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   755тЖТ   755тЖТ        
   756тЖТ   756тЖТ        if has_sequential:
   757тЖТ   757тЖТ            # Ejecutar secuencialmente para evitar race conditions
   758тЖТ   758тЖТ            results = []
   759тЖТ   759тЖТ            for tc in tool_calls:
   760тЖТ   760тЖТ                results.append(self._execute_single_tool(tc, messages))
   761тЖТ   761тЖТ            return results
   762тЖТ   762тЖТ
   763тЖТ   763тЖТ        # Ejecucion paralela para tools de solo lectura
   764тЖТ   764тЖТ        results = [None] * len(tool_calls)
   765тЖТ   765тЖТ        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   766тЖТ   766тЖТ            futures = {}
   767тЖТ   767тЖТ            for i, tc in enumerate(tool_calls):
   768тЖТ   768тЖТ                future = executor.submit(self._execute_single_tool, tc, messages)
   769тЖТ   769тЖТ                futures[future] = i
   770тЖТ   770тЖТ            
   771тЖТ   771тЖТ            for future in as_completed(futures):
   772тЖТ   772тЖТ                idx = futures[future]
   773тЖТ   773тЖТ                try:
   774тЖТ   774тЖТ                    results[idx] = future.result(timeout=30)
   775тЖТ   775тЖТ                except Exception as e:
   776тЖТ   776тЖТ                    results[idx] = f"ERROR: {e}"
   777тЖТ   777тЖТ
   778тЖТ   778тЖТ        return results

   780тЖТ   780тЖТ    def _execute_single_tool(self, tc, messages, max_retries=1):
   781тЖТ   781тЖТ        """Ejecuta un solo tool call con retry automatico."""
   782тЖТ   782тЖТ        import time as _time

   784тЖТ   784тЖТ        tool_params = tc["params"]

   797тЖТ   797тЖТ

   800тЖТ   800тЖТ

   802тЖТ   802тЖТ            get_metrics().record_error("tool:" + tool_name)

   812тЖТ   812тЖТ        if "PELIGROSO" in tool_result:
   813тЖТ   813тЖТ            self._log(f"Comando peligroso bloqueado: {tool_name}", "warning")
   814тЖТ   814тЖТ        if len(tool_result) > 50 and "ERROR" not in tool_result:
   815тЖТ   815тЖТ            self.memory.remember(
   816тЖТ   816тЖТ                f"Resultado de {tool_name}: {tool_result[:300]}",
   817тЖТ   817тЖТ                metadata={"type": "tool_result", "tool": tool_name}
   818тЖТ   818тЖТ            )
   819тЖТ   819тЖТ
   820тЖТ   820тЖТ        return tool_result
