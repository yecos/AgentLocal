   697тЖТ   697тЖТ    def _clean_json_leak(self, text):
   698тЖТ   698тЖТ        """Limpia texto que tiene restos de formato JSON para mostrar al usuario.
   699тЖТ   699тЖТ        v3: Mas agresivo - tambien limpia JSON parcial al inicio/final del texto.
   700тЖТ   700тЖТ        """
   701тЖТ   701тЖТ        # Si el texto es JSON completo, extraer solo el contenido util
   702тЖТ   702тЖТ        parsed = self._parse_json(text)
   703тЖТ   703тЖТ        if parsed:
   704тЖТ   704тЖТ            # Prioridad: respuesta_final > pensamiento > texto original
   705тЖТ   705тЖТ            if parsed.get("respuesta_final", "").strip():
   706тЖТ   706тЖТ                return parsed["respuesta_final"].strip()
   707тЖТ   707тЖТ            if parsed.get("pensamiento", "").strip():
   708тЖТ   708тЖТ                return parsed["pensamiento"].strip()
   709тЖТ   709тЖТ        # Si no es JSON completo, intentar limpiar restos
   710тЖТ   710тЖТ        # Caso: texto que empieza con JSON parcial
   711тЖТ   711тЖТ        import re as _re
   712тЖТ   712тЖТ        # Remover JSON parcial al inicio: {"pensamiento": "..." ... restos
   713тЖТ   713тЖТ        cleaned = _re.sub(r'^\s*\{[^}]*$', '', text).strip()
   714тЖТ   714тЖТ        # Remover JSON parcial al final
   715тЖТ   715тЖТ        cleaned = _re.sub(r'\{[^}]*$\s*$', '', cleaned).strip()
   716тЖТ   716тЖТ        # Si despues de limpiar quedo vacio, devolver texto original
   717тЖТ   717тЖТ        return cleaned if cleaned else text

   722тЖТ   722тЖТ    def _execute_tool_calls(self, tool_calls, messages):
   723тЖТ   723тЖТ        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   724тЖТ   724тЖТ        from concurrent.futures import ThreadPoolExecutor, as_completed
   725тЖТ   725тЖТ
   726тЖТ   726тЖТ        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   727тЖТ   727тЖТ        filtered_calls = []
   728тЖТ   728тЖТ        for tc in tool_calls:
   729тЖТ   729тЖТ            tool_name = tc["name"]
   730тЖТ   730тЖТ            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   731тЖТ   731тЖТ            self._total_tool_calls += 1
   732тЖТ   732тЖТ
   733тЖТ   733тЖТ            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   734тЖТ   734тЖТ                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   735тЖТ   735тЖТ                break
   736тЖТ   736тЖТ
   737тЖТ   737тЖТ            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   738тЖТ   738тЖТ                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   739тЖТ   739тЖТ                continue
   740тЖТ   740тЖТ
   741тЖТ   741тЖТ            filtered_calls.append(tc)
   742тЖТ   742тЖТ
   743тЖТ   743тЖТ        tool_calls = filtered_calls
   744тЖТ   744тЖТ
   745тЖТ   745тЖТ        # Validar parametros de todos los tool calls primero
   746тЖТ   746тЖТ        for tc in tool_calls:
   747тЖТ   747тЖТ            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   748тЖТ   748тЖТ
   749тЖТ   749тЖТ        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   750тЖТ   750тЖТ        if len(tool_calls) == 1:
   751тЖТ   751тЖТ            return [self._execute_single_tool(tool_calls[0], messages)]
   752тЖТ   752тЖТ
   753тЖТ   753тЖТ        # Si hay multiples tools, ejecutar en paralelo
   754тЖТ   754тЖТ        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   755тЖТ   755тЖТ        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   756тЖТ   756тЖТ        
   757тЖТ   757тЖТ        if has_sequential:
   758тЖТ   758тЖТ            # Ejecutar secuencialmente para evitar race conditions
   759тЖТ   759тЖТ            results = []
   760тЖТ   760тЖТ            for tc in tool_calls:
   761тЖТ   761тЖТ                results.append(self._execute_single_tool(tc, messages))
   762тЖТ   762тЖТ            return results
   763тЖТ   763тЖТ
   764тЖТ   764тЖТ        # Ejecucion paralela para tools de solo lectura
   765тЖТ   765тЖТ        results = [None] * len(tool_calls)
   766тЖТ   766тЖТ        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   767тЖТ   767тЖТ            futures = {}
   768тЖТ   768тЖТ            for i, tc in enumerate(tool_calls):
   769тЖТ   769тЖТ                future = executor.submit(self._execute_single_tool, tc, messages)
   770тЖТ   770тЖТ                futures[future] = i
   771тЖТ   771тЖТ            
   772тЖТ   772тЖТ            for future in as_completed(futures):
   773тЖТ   773тЖТ                idx = futures[future]
   774тЖТ   774тЖТ                try:
   775тЖТ   775тЖТ                    results[idx] = future.result(timeout=30)
   776тЖТ   776тЖТ                except Exception as e:
   777тЖТ   777тЖТ                    results[idx] = f"ERROR: {e}"
   778тЖТ   778тЖТ
   779тЖТ   779тЖТ        return results

   781тЖТ   781тЖТ    def _execute_single_tool(self, tc, messages, max_retries=1):
   782тЖТ   782тЖТ        """Ejecuta un solo tool call con retry automatico."""
   783тЖТ   783тЖТ        import time as _time
   784тЖТ   784тЖТ        tool_name = tc["name"]
   785тЖТ   785тЖТ        tool_params = tc["params"]
   786тЖТ   786тЖТ
   787тЖТ   787тЖТ        self._log(f"Ejecutando: {tool_name}({tool_params})", "execution")
   788тЖТ   788тЖТ
   789тЖТ   789тЖТ        _tool_start = _time.monotonic()
   790тЖТ   790тЖТ        tool_result = self._execute_tool(tool_name, tool_params)
   791тЖТ   791тЖТ
   792тЖТ   792тЖТ        # Retry si fallo y el tool es reintentable
   793тЖТ   793тЖТ        if "ERROR" in tool_result and max_retries > 0:
   794тЖТ   794тЖТ            retryable_errors = ["TIMEOUT", "ConnectionError", "empty", "vacia"]
   795тЖТ   795тЖТ            if any(err in tool_result for err in retryable_errors):
   796тЖТ   796тЖТ                self._log(f"Reintentando {tool_name} (error transitorio)...", "execution")
   797тЖТ   797тЖТ                tool_result = self._execute_tool(tool_name, tool_params)
   798тЖТ   798тЖТ
   799тЖТ   799тЖТ        _tool_elapsed_ms = (_time.monotonic() - _tool_start) * 1000.0
   800тЖТ   800тЖТ        get_metrics().record_tool_call(tool_name, _tool_elapsed_ms)
   801тЖТ   801тЖТ
   802тЖТ   802тЖТ        if "ERROR" in tool_result:
   803тЖТ   803тЖТ            get_metrics().record_error("tool:" + tool_name)
   804тЖТ   804тЖТ
   805тЖТ   805тЖТ        self._log(f"Resultado: {tool_result[:150]}...", "observation")
   806тЖТ   806тЖТ
   807тЖТ   807тЖТ        # Alimentar memoria de trabajo
   808тЖТ   808тЖТ        self.memory.add_step(f"{tool_name}({tool_params})", tool_result[:200])
   809тЖТ   809тЖТ        if "ERROR" in tool_result:
   810тЖТ   810тЖТ            self.memory.set_error(f"{tool_name}: {tool_result[:100]}")
   811тЖТ   811тЖТ        if "PELIGROSO" in tool_result:
   812тЖТ   812тЖТ            self._log(f"Comando peligroso bloqueado: {tool_name}", "warning")
   813тЖТ   813тЖТ        if len(tool_result) > 50 and "ERROR" not in tool_result:
   814тЖТ   814тЖТ            self.memory.remember(
   815тЖТ   815тЖТ                f"Resultado de {tool_name}: {tool_result[:300]}",
   816тЖТ   816тЖТ                metadata={"type": "tool_result", "tool": tool_name}
   817тЖТ   817тЖТ            )
   818тЖТ   818тЖТ
   819тЖТ   819тЖТ        return tool_result

  1026тЖТ  1026тЖТ        results = []
