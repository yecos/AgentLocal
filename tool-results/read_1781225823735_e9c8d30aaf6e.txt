   549тЖТ   549тЖТ    # EJECUCION DE TOOLS (paralelo + retry)
   550тЖТ   550тЖТ    # ----------------------------------------------------------
   551тЖТ   551тЖТ    def _execute_tool_calls(self, tool_calls, messages):
   552тЖТ   552тЖТ        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   553тЖТ   553тЖТ        from concurrent.futures import ThreadPoolExecutor, as_completed
   554тЖТ   554тЖТ
   555тЖТ   555тЖТ        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   556тЖТ   556тЖТ        filtered_calls = []
   557тЖТ   557тЖТ        for tc in tool_calls:
   558тЖТ   558тЖТ            tool_name = tc["name"]
   559тЖТ   559тЖТ            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   560тЖТ   560тЖТ            self._total_tool_calls += 1
   561тЖТ   561тЖТ
   562тЖТ   562тЖТ            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   563тЖТ   563тЖТ                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   564тЖТ   564тЖТ                break
   565тЖТ   565тЖТ
   566тЖТ   566тЖТ            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   567тЖТ   567тЖТ                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   568тЖТ   568тЖТ                continue
   569тЖТ   569тЖТ
   570тЖТ   570тЖТ            filtered_calls.append(tc)
   571тЖТ   571тЖТ
   572тЖТ   572тЖТ        tool_calls = filtered_calls
   573тЖТ   573тЖТ
   574тЖТ   574тЖТ        # Validar parametros de todos los tool calls primero
   575тЖТ   575тЖТ        for tc in tool_calls:
   576тЖТ   576тЖТ            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   577тЖТ   577тЖТ
   578тЖТ   578тЖТ        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   579тЖТ   579тЖТ        if len(tool_calls) == 1:
   580тЖТ   580тЖТ            return [self._execute_single_tool(tool_calls[0], messages)]
   581тЖТ   581тЖТ
   582тЖТ   582тЖТ        # Si hay multiples tools, ejecutar en paralelo
   583тЖТ   583тЖТ        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   584тЖТ   584тЖТ        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   585тЖТ   585тЖТ        
   586тЖТ   586тЖТ        if has_sequential:
   587тЖТ   587тЖТ            # Ejecutar secuencialmente para evitar race conditions
   588тЖТ   588тЖТ            results = []
   589тЖТ   589тЖТ            for tc in tool_calls:
   590тЖТ   590тЖТ                results.append(self._execute_single_tool(tc, messages))
   591тЖТ   591тЖТ            return results
   592тЖТ   592тЖТ
   593тЖТ   593тЖТ        # Ejecucion paralela para tools de solo lectura
   594тЖТ   594тЖТ        results = [None] * len(tool_calls)
   595тЖТ   595тЖТ        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   596тЖТ   596тЖТ            futures = {}
   597тЖТ   597тЖТ            for i, tc in enumerate(tool_calls):
   598тЖТ   598тЖТ                future = executor.submit(self._execute_single_tool, tc, messages)
   599тЖТ   599тЖТ                futures[future] = i
   600тЖТ   600тЖТ            
   601тЖТ   601тЖТ            for future in as_completed(futures):
   602тЖТ   602тЖТ                idx = futures[future]
   603тЖТ   603тЖТ                try:
   604тЖТ   604тЖТ                    results[idx] = future.result(timeout=30)
   605тЖТ   605тЖТ                except Exception as e:
   606тЖТ   606тЖТ                    results[idx] = f"ERROR: {e}"
   607тЖТ   607тЖТ
   608тЖТ   608тЖТ        return results
   609тЖТ   609тЖТ
   610тЖТ   610тЖТ    def _execute_single_tool(self, tc, messages, max_retries=1):
   611тЖТ   611тЖТ        """Ejecuta un solo tool call con retry automatico."""
   612тЖТ   612тЖТ        import time as _time
   613тЖТ   613тЖТ        tool_name = tc["name"]
   614тЖТ   614тЖТ        tool_params = tc["params"]
   615тЖТ   615тЖТ
   616тЖТ   616тЖТ        self._log(f"Ejecutando: {tool_name}({tool_params})", "execution")
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ        _tool_start = _time.monotonic()
   619тЖТ   619тЖТ        tool_result = self._execute_tool(tool_name, tool_params)
   620тЖТ   620тЖТ
   621тЖТ   621тЖТ        # Retry si fallo y el tool es reintentable
   622тЖТ   622тЖТ        if "ERROR" in tool_result and max_retries > 0:
   623тЖТ   623тЖТ            retryable_errors = ["TIMEOUT", "ConnectionError", "empty", "vacia"]
   624тЖТ   624тЖТ            if any(err in tool_result for err in retryable_errors):
   625тЖТ   625тЖТ                self._log(f"Reintentando {tool_name} (error transitorio)...", "execution")
   626тЖТ   626тЖТ                tool_result = self._execute_tool(tool_name, tool_params)
   627тЖТ   627тЖТ
   628тЖТ   628тЖТ        _tool_elapsed_ms = (_time.monotonic() - _tool_start) * 1000.0
   629тЖТ   629тЖТ        get_metrics().record_tool_call(tool_name, _tool_elapsed_ms)
   630тЖТ   630тЖТ
   631тЖТ   631тЖТ        if "ERROR" in tool_result:
   632тЖТ   632тЖТ            get_metrics().record_error("tool:" + tool_name)
   633тЖТ   633тЖТ
   634тЖТ   634тЖТ        self._log(f"Resultado: {tool_result[:150]}...", "observation")
   635тЖТ   635тЖТ
   636тЖТ   636тЖТ        # Alimentar memoria de trabajo
   637тЖТ   637тЖТ        self.memory.add_step(f"{tool_name}({tool_params})", tool_result[:200])
   638тЖТ   638тЖТ        if "ERROR" in tool_result:
   639тЖТ   639тЖТ            self.memory.set_error(f"{tool_name}: {tool_result[:100]}")

   802тЖТ   802тЖТ    def _parse_json(self, text):
   803тЖТ   803тЖТ        try:
   804тЖТ   804тЖТ            return json.loads(text)
   805тЖТ   805тЖТ        except Exception:
   806тЖТ   806тЖТ            pass
   807тЖТ   807тЖТ        patterns = [
   808тЖТ   808тЖТ            r'```json\s*(.*?)\s*```',
   809тЖТ   809тЖТ            r'```\s*(.*?)\s*```',
   810тЖТ   810тЖТ        ]
   811тЖТ   811тЖТ        for pattern in patterns:
   812тЖТ   812тЖТ            match = re.search(pattern, text, re.DOTALL)
   813тЖТ   813тЖТ            if match:
   814тЖТ   814тЖТ                try:
   815тЖТ   815тЖТ                    return json.loads(match.group(1))
   816тЖТ   816тЖТ                except Exception:
   817тЖТ   817тЖТ                    continue
   818тЖТ   818тЖТ        # Intentar parsear multiples JSONs en el texto
   819тЖТ   819тЖТ        # Cada JSON es una accion separada del ReAct
   820тЖТ   820тЖТ        jsons = self._extract_all_jsons(text)
   821тЖТ   821тЖТ        if len(jsons) == 1:
   822тЖТ   822тЖТ            return jsons[0]
   823тЖТ   823тЖТ        elif len(jsons) > 1:
   824тЖТ   824тЖТ            # Multiples JSONs = multiples tool calls
   825тЖТ   825тЖТ            # Combinar en una lista de tool calls
   826тЖТ   826тЖТ            tool_calls = []
   827тЖТ   827тЖТ            for j in jsons:
   828тЖТ   828тЖТ                accion = j.get("accion", "").strip()
   829тЖТ   829тЖТ                if accion:
   830тЖТ   830тЖТ                    tool_calls.append({"name": accion, "params": j.get("params", {})})
   831тЖТ   831тЖТ            if tool_calls:
   832тЖТ   832тЖТ                # Retornar formato especial para multiples tool calls
   833тЖТ   833тЖТ                return {"_multi_tool_calls": True, "tool_calls": tool_calls}
   834тЖТ   834тЖТ            return jsons[0]  # Fallback al primer JSON
   835тЖТ   835тЖТ        # Ultimo intento: buscar un solo objeto JSON
   836тЖТ   836тЖТ        match = re.search(r'(\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\})', text, re.DOTALL)
   837тЖТ   837тЖТ        if match:
   838тЖТ   838тЖТ            try:
   839тЖТ   839тЖТ                return json.loads(match.group(1))
   840тЖТ   840тЖТ            except Exception:
   841тЖТ   841тЖТ                pass
   842тЖТ   842тЖТ        return None
   843тЖТ   843тЖТ
   844тЖТ   844тЖТ    def _extract_all_jsons(self, text):
   845тЖТ   845тЖТ        """Extrae todos los objetos JSON validos de un texto."""
   846тЖТ   846тЖТ        results = []
   847тЖТ   847тЖТ        # Buscar secuencias que parezcan JSON
   848тЖТ   848тЖТ        i = 0
   849тЖТ   849тЖТ        while i < len(text):
   850тЖТ   850тЖТ            # Encontrar el proximo '{'
   851тЖТ   851тЖТ            start = text.find('{', i)
   852тЖТ   852тЖТ            if start == -1:
   853тЖТ   853тЖТ                break
   854тЖТ   854тЖТ            # Intentar parsear desde este punto
   855тЖТ   855тЖТ            depth = 0
   856тЖТ   856тЖТ            end = start
   857тЖТ   857тЖТ            for j in range(start, len(text)):
   858тЖТ   858тЖТ                if text[j] == '{':
   859тЖТ   859тЖТ                    depth += 1
   860тЖТ   860тЖТ                elif text[j] == '}':
   861тЖТ   861тЖТ                    depth -= 1
   862тЖТ   862тЖТ                if depth == 0:
   863тЖТ   863тЖТ                    end = j + 1
   864тЖТ   864тЖТ                    break
   865тЖТ   865тЖТ            if end > start:
   866тЖТ   866тЖТ                try:
   867тЖТ   867тЖТ                    parsed = json.loads(text[start:end])
   868тЖТ   868тЖТ                    if isinstance(parsed, dict):
   869тЖТ   869тЖТ                        results.append(parsed)
   870тЖТ   870тЖТ                except Exception:
   871тЖТ   871тЖТ                    pass
   872тЖТ   872тЖТ                i = end
   873тЖТ   873тЖТ            else:
   874тЖТ   874тЖТ                i = start + 1
   875тЖТ   875тЖТ        return results
