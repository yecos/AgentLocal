   369→   369→    # ----------------------------------------------------------
   370→   370→    # STREAMING LLM CON TOOL CALLING (token a token)
   371→   371→    # ----------------------------------------------------------
   372→   372→    def _stream_llm_with_tools(self, messages):
   373→   373→        """
   374→   374→        Genera respuesta del LLM con streaming REAL.
   375→   375→        Yields: {"type": "token"|"tool_calls"|"done", "data": ...}
   376→   376→        """
   377→   377→        import time as _time
   378→   378→        _llm_start = _time.monotonic()
   379→   379→        full_content = ""
   380→   380→        full_tool_calls = []
   381→   381→
   382→   382→        try:
   383→   383→            # Intentar streaming HTTP directo (mas rapido, sin overhead de lib)
   384→   384→            for chunk in ollama.generate_stream(messages, tools=TOOL_SCHEMAS):
   385→   385→                if isinstance(chunk, str):
   386→   386→                    # Token de texto - emitir inmediatamente
   387→   387→                    # PERO: si parece JSON de tool call, no emitirlo al usuario
   388→   388→                    full_content += chunk
   389→   389→                    # Detectar si el contenido es JSON de herramienta
   390→   390→                    if self._looks_like_tool_json(full_content):
   391→   391→                        continue  # Acumular sin emitir
   392→   392→                    # Si antes se acumulaba JSON pero ya no, limpiar
   393→   393→                    yield {"type": "token", "data": chunk}
   394→   394→                elif isinstance(chunk, dict):
   395→   395→                    # Resultado final con tool_calls
   396→   396→                    msg = chunk.get("message", chunk)
   397→   397→                    tool_calls = msg.get("tool_calls", [])
   398→   398→                    content = msg.get("content", "")
   399→   399→
   400→   400→                    if content and not full_content:
   401→   401→                        # Contenido que no se streameo (fallback)
   402→   402→                        full_content = content
   403→   403→                        yield {"type": "token", "data": content}
   404→   404→
   405→   405→                    if tool_calls:
   406→   406→                        # Parsear tool calls
   407→   407→                        parsed_calls = []
   408→   408→                        for tc in tool_calls:
   409→   409→                            tool_name = tc.get("function", {}).get("name", "")
   410→   410→                            tool_params = tc.get("function", {}).get("arguments", {})
   411→   411→                            self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
   412→   412→                            parsed_calls.append({"name": tool_name, "params": tool_params})
   413→   413→                        yield {"type": "tool_calls", "data": parsed_calls}
   414→   414→                        yield {"type": "done", "data": False}
   415→   415→                        return

   417→   417→            # Si llegamos aqui sin tool_calls, verificar si el contenido es JSON de herramientas
   418→   418→            if full_content:
   419→   419→                # Si el contenido acumulado era JSON de tool calls (no se emitio),
   420→   420→                # parsearlo y ejecutar como tool calls

   443→   443→
   444→   444→        except Exception as e:

   446→   446→            get_metrics().record_error("llm_stream")
   447→   447→
   448→   448→        # Fallback: modo no-streaming
   449→   449→        try:
   450→   450→            response = ollama.generate(messages, tools=TOOL_SCHEMAS)
   451→   451→            # generate() already records llm_call via @timed, so no duplicate
   452→   452→            if isinstance(response, str):
   453→   453→                yield {"type": "token", "data": response}
   454→   454→                yield {"type": "done", "data": True}
   455→   455→                return
   456→   456→
   457→   457→            message = response.get("message", response)
   458→   458→            tool_calls = message.get("tool_calls", [])
   459→   459→            content = message.get("content", "")

   463→   463→                for tc in tool_calls:
   464→   464→                    tool_name = tc.get("function", {}).get("name", "")
   465→   465→                    tool_params = tc.get("function", {}).get("arguments", {})
   466→   466→                    self._log(f"Tool call: {tool_name}({tool_params})", "thinking")
   467→   467→                    parsed_calls.append({"name": tool_name, "params": tool_params})
   468→   468→                yield {"type": "tool_calls", "data": parsed_calls}
   469→   469→                yield {"type": "done", "data": False}
   470→   470→                return
   471→   471→
   472→   472→            if content:
   473→   473→                yield {"type": "token", "data": content}
   474→   474→                yield {"type": "done", "data": True}
   475→   475→                return
   476→   476→
   477→   477→        except Exception as e:
   478→   478→            self._log(f"Error en fallback: {e}", "error")
   479→   479→
   480→   480→        yield {"type": "done", "data": False}

   533→   533→    def _react_with_json(self, messages, iteration):
   534→   534→        """ReAct usando JSON parsing (fallback)."""
   535→   535→        # generate_chat() already records llm_call via @timed decorator
   536→   536→        if not any("HERRAMIENTAS DISPONIBLES" in str(m.get("content", "")) for m in messages):
   537→   537→            system_msg_idx = next((i for i, m in enumerate(messages) if m["role"] == "system"), -1)
   538→   538→            if system_msg_idx >= 0:
   539→   539→                messages[system_msg_idx]["content"] += JSON_TOOLS_PROMPT

   541→   541→        try:
   542→   542→            response = ollama.generate_chat(messages)
   543→   543→            if not response:
   544→   544→                return ("error", "El LLM no respondio")
   545→   545→
   546→   546→            parsed = self._parse_json(response)
   547→   547→            if not parsed:
   548→   548→                # No es JSON - probablemente respuesta directa del modelo
   549→   549→                # Limpiar posibles restos de formato JSON
   550→   550→                clean = self._clean_json_leak(response)
   551→   551→                return ("respond", clean)
   552→   552→
   553→   553→            # 1. Si hay _multi_tool_calls, ejecutar todas las herramientas
   554→   554→            if parsed.get("_multi_tool_calls") and parsed.get("tool_calls"):
   555→   555→                return ("tool_calls", parsed["tool_calls"])
   556→   556→
   557→   557→            # 2. Si hay respuesta_final con contenido, usarla
   558→   558→            respuesta_final = parsed.get("respuesta_final", "").strip()
   559→   559→            if respuesta_final:
   560→   560→                return ("respond", respuesta_final)
   561→   561→
   562→   562→            accion = parsed.get("accion", "").strip()
   563→   563→            params = parsed.get("params", {})
   564→   564→            pensamiento = parsed.get("pensamiento", "").strip()
   565→   565→
   566→   566→            if pensamiento:
   567→   567→                self._log(f"Pensamiento: {pensamiento}", "thinking")
   568→   568→
   569→   569→            # 3. Si hay accion valida, ejecutar herramienta
   570→   570→            if accion and accion in TOOL_FUNCTIONS:
   571→   571→                # Si tambien hay pensamiento, inyectarlo como contexto
   572→   572→                if pensamiento:
   573→   573→                    messages.append({
   574→   574→                        "role": "assistant",
   575→   575→                        "content": f"[Pensamiento: {pensamiento}] Ejecutando {accion}..."
   576→   576→                    })

   578→   578→

   580→   580→            #    el modelo esta respondiendo en campo equivocado - usar pensamiento

   582→   582→                self._log("Modelo respondio en 'pensamiento' en vez de 'respuesta_final'", "info")
   583→   583→                return ("respond", pensamiento)
   584→   584→
   585→   585→            # 4. Fallback: devolver texto limpio (no JSON crudo)
   586→   586→            clean = self._clean_json_leak(response)
   587→   587→            return ("respond", clean)
   588→   588→
   589→   589→        except Exception as e:
   590→   590→            return ("error", str(e))

   592→   592→    def _clean_json_leak(self, text):
   593→   593→        """Limpia texto que tiene restos de formato JSON para mostrar al usuario.
   594→   594→        v3: Mas agresivo - tambien limpia JSON parcial al inicio/final del texto.
   595→   595→        """
   596→   596→        # Si el texto es JSON completo, extraer solo el contenido util
   597→   597→        parsed = self._parse_json(text)
   598→   598→        if parsed:
   599→   599→            # Prioridad: respuesta_final > pensamiento > texto original
   600→   600→            if parsed.get("respuesta_final", "").strip():
   601→   601→                return parsed["respuesta_final"].strip()
   602→   602→            if parsed.get("pensamiento", "").strip():
   603→   603→                return parsed["pensamiento"].strip()
   604→   604→        # Si no es JSON completo, intentar limpiar restos
   605→   605→        # Caso: texto que empieza con JSON parcial
   606→   606→        import re as _re
   607→   607→        # Remover JSON parcial al inicio: {"pensamiento": "..." ... restos
   608→   608→        cleaned = _re.sub(r'^\s*\{[^}]*$', '', text).strip()
   609→   609→        # Remover JSON parcial al final
   610→   610→        cleaned = _re.sub(r'\{[^}]*$\s*$', '', cleaned).strip()
   611→   611→        # Si despues de limpiar quedo vacio, devolver texto original
   612→   612→        return cleaned if cleaned else text

   617→   617→    def _execute_tool_calls(self, tool_calls, messages):
   618→   618→        """Ejecuta multiples tool calls con rate limiting y paralelismo."""
   619→   619→        from concurrent.futures import ThreadPoolExecutor, as_completed
   620→   620→
   621→   621→        # Rate limiting: filtrar herramientas que se llamaron demasiadas veces
   622→   622→        filtered_calls = []
   623→   623→        for tc in tool_calls:
   624→   624→            tool_name = tc["name"]
   625→   625→            self._tool_call_counts[tool_name] = self._tool_call_counts.get(tool_name, 0) + 1
   626→   626→            self._total_tool_calls += 1
   627→   627→
   628→   628→            if self._total_tool_calls > self.MAX_TOTAL_TOOL_CALLS:
   629→   629→                self._log(f"Rate limit: max total tool calls alcanzado ({self.MAX_TOTAL_TOOL_CALLS})", "warning")
   630→   630→                break
   631→   631→
   632→   632→            if self._tool_call_counts[tool_name] > self.MAX_SAME_TOOL_CALLS:
   633→   633→                self._log(f"Rate limit: {tool_name} llamada {self._tool_call_counts[tool_name]} veces (max {self.MAX_SAME_TOOL_CALLS})", "warning")
   634→   634→                continue
   635→   635→
   636→   636→            filtered_calls.append(tc)
   637→   637→
   638→   638→        tool_calls = filtered_calls
   639→   639→
   640→   640→        # Validar parametros de todos los tool calls primero
   641→   641→        for tc in tool_calls:
   642→   642→            tc["params"] = self._validate_tool_params(tc["name"], tc["params"])
   643→   643→
   644→   644→        # Si hay solo 1 tool, ejecutar directamente (sin overhead de threads)
   645→   645→        if len(tool_calls) == 1:
   646→   646→            return [self._execute_single_tool(tool_calls[0], messages)]
   647→   647→
   648→   648→        # Si hay multiples tools, ejecutar en paralelo
   649→   649→        # PERO no paralelizar si alguno es "ejecutar_comando" (puede tener side effects)
   650→   650→        has_sequential = any(tc["name"] == "ejecutar_comando" for tc in tool_calls)
   651→   651→        
   652→   652→        if has_sequential:
   653→   653→            # Ejecutar secuencialmente para evitar race conditions
   654→   654→            results = []
   655→   655→            for tc in tool_calls:
   656→   656→                results.append(self._execute_single_tool(tc, messages))
   657→   657→            return results
   658→   658→
   659→   659→        # Ejecucion paralela para tools de solo lectura
   660→   660→        results = [None] * len(tool_calls)
   661→   661→        with ThreadPoolExecutor(max_workers=min(len(tool_calls), 3)) as executor:
   662→   662→            futures = {}
   663→   663→            for i, tc in enumerate(tool_calls):
   664→   664→                future = executor.submit(self._execute_single_tool, tc, messages)
   665→   665→                futures[future] = i
   666→   666→            
   667→   667→            for future in as_completed(futures):
   668→   668→                idx = futures[future]
   669→   669→                try:
   670→   670→                    results[idx] = future.result(timeout=30)
   671→   671→                except Exception as e:
   672→   672→                    results[idx] = f"ERROR: {e}"
   673→   673→
   674→   674→        return results
   675→   675→
   676→   676→    def _execute_single_tool(self, tc, messages, max_retries=1):
   677→   677→        """Ejecuta un solo tool call con retry automatico."""
   678→   678→        import time as _time
   679→   679→        tool_name = tc["name"]
   680→   680→        tool_params = tc["params"]
   681→   681→
   682→   682→        self._log(f"Ejecutando: {tool_name}({tool_params})", "execution")
   683→   683→
   684→   684→        _tool_start = _time.monotonic()
   685→   685→        tool_result = self._execute_tool(tool_name, tool_params)
   686→   686→
   687→   687→        # Retry si fallo y el tool es reintentable
   688→   688→        if "ERROR" in tool_result and max_retries > 0:
   689→   689→            retryable_errors = ["TIMEOUT", "ConnectionError", "empty", "vacia"]
   690→   690→            if any(err in tool_result for err in retryable_errors):
   691→   691→                self._log(f"Reintentando {tool_name} (error transitorio)...", "execution")
   692→   692→                tool_result = self._execute_tool(tool_name, tool_params)
   693→   693→
   694→   694→        _tool_elapsed_ms = (_time.monotonic() - _tool_start) * 1000.0
   695→   695→        get_metrics().record_tool_call(tool_name, _tool_elapsed_ms)
   696→   696→
   697→   697→        if "ERROR" in tool_result:
   698→   698→            get_metrics().record_error("tool:" + tool_name)
   699→   699→
   700→   700→        self._log(f"Resultado: {tool_result[:150]}...", "observation")
   701→   701→
   702→   702→        # Alimentar memoria de trabajo
   703→   703→        self.memory.add_step(f"{tool_name}({tool_params})", tool_result[:200])
   704→   704→        if "ERROR" in tool_result:
   705→   705→            self.memory.set_error(f"{tool_name}: {tool_result[:100]}")
   706→   706→        if "PELIGROSO" in tool_result:
   707→   707→            self._log(f"Comando peligroso bloqueado: {tool_name}", "warning")
   708→   708→        if len(tool_result) > 50 and "ERROR" not in tool_result:
   709→   709→            self.memory.remember(
   710→   710→                f"Resultado de {tool_name}: {tool_result[:300]}",
   711→   711→                metadata={"type": "tool_result", "tool": tool_name}
   712→   712→            )
   713→   713→
   714→   714→        return tool_result

   716→   716→    def _validate_tool_params(self, tool_name, params):
   717→   717→        """Valida y limpia parametros de un tool call antes de ejecutar."""

   724→   724→            if isinstance(key, str) and len(key) < 50:

   868→   868→    def _parse_json(self, text):
   869→   869→        try:
   870→   870→            return json.loads(text)
   871→   871→        except Exception:
   872→   872→            pass
   873→   873→        patterns = [
   874→   874→            r'```json\s*(.*?)\s*```',
   875→   875→            r'```\s*(.*?)\s*```',
   876→   876→        ]
   877→   877→        for pattern in patterns:
   878→   878→            match = re.search(pattern, text, re.DOTALL)
   879→   879→            if match:
   880→   880→                try:
   881→   881→                    return json.loads(match.group(1))
   882→   882→                except Exception:
   883→   883→                    continue
   884→   884→        # Intentar parsear multiples JSONs en el texto
   885→   885→        # Cada JSON es una accion separada del ReAct
   886→   886→        jsons = self._extract_all_jsons(text)
   887→   887→        if len(jsons) == 1:
   888→   888→            return jsons[0]
   889→   889→        elif len(jsons) > 1:
   890→   890→            # Multiples JSONs = multiples tool calls
   891→   891→            # Combinar en una lista de tool calls
   892→   892→            tool_calls = []
   893→   893→            for j in jsons:
   894→   894→                accion = j.get("accion", "").strip()
   895→   895→                if accion:
   896→   896→                    tool_calls.append({"name": accion, "params": j.get("params", {})})
   897→   897→            if tool_calls:
   898→   898→                # Retornar formato especial para multiples tool calls
   899→   899→                return {"_multi_tool_calls": True, "tool_calls": tool_calls}
   900→   900→            return jsons[0]  # Fallback al primer JSON
   901→   901→        # Ultimo intento: buscar un solo objeto JSON
   902→   902→        match = re.search(r'(\{[^{}]*(?:\{[^{}]*\}[^{}]*)*\})', text, re.DOTALL)
   903→   903→        if match:
   904→   904→            try:
   905→   905→                return json.loads(match.group(1))
   906→   906→            except Exception:
   907→   907→                pass
   908→   908→        return None

   910→   910→    def _extract_all_jsons(self, text):
   911→   911→        """Extrae todos los objetos JSON validos de un texto."""
   912→   912→        results = []
   913→   913→        # Buscar secuencias que parezcan JSON
   914→   914→        i = 0
   915→   915→        while i < len(text):
   916→   916→            # Encontrar el proximo '{'
   917→   917→            start = text.find('{', i)
   918→   918→            if start == -1:
   919→   919→                break
   920→   920→            # Intentar parsear desde este punto
   921→   921→            depth = 0
   922→   922→            end = start
   923→   923→            for j in range(start, len(text)):
   924→   924→                if text[j] == '{':
   925→   925→                    depth += 1
   926→   926→                elif text[j] == '}':
   927→   927→                    depth -= 1
   928→   928→                if depth == 0:
   929→   929→                    end = j + 1
   930→   930→                    break
   931→   931→            if end > start:
   932→   932→                try:
   933→   933→                    parsed = json.loads(text[start:end])
   934→   934→                    if isinstance(parsed, dict):
   935→   935→                        results.append(parsed)
   936→   936→                except Exception:
   937→   937→                    pass
   938→   938→                i = end
   939→   939→            else:
   940→   940→                i = start + 1
   941→   941→        return results

   943→   943→    def _looks_like_tool_json(self, text):
   944→   944→        """Detecta si el texto parece ser JSON de tool call (no respuesta al usuario).
   945→   945→        v3: Mas robusto - detecta JSON que empieza despues de whitespace o texto corto.
   946→   946→        """
   947→   947→        text = text.strip()
   948→   948→        if not text:
   949→   949→            return False
   950→   950→        # Si el texto empieza con { probablemente es JSON de tool call
   951→   951→        if text.startswith('{'):
   952→   952→            return True
   953→   953→        # Si despues de whitespace/newline hay {, tambien es JSON
   954→   954→        # Esto captura casos donde el modelo genera: "\n{" o "  {"
   955→   955→        stripped = text.lstrip()
   956→   956→        if stripped.startswith('{'):
   957→   957→            return True
   958→   958→        # Si contiene patrones de tool calls en cualquier parte
   959→   959→        tool_indicators = ['"accion"', '"pensamiento"', '"params"', '"respuesta_final"',
   960→   960→                          '"abrir_', '"ejecutar_', '"leer_', '"escribir_', '"buscar_"']
   961→   961→        for indicator in tool_indicators:
   962→   962→            if indicator in text:
   963→   963→                return True
   964→   964→        # Si el texto es corto y parece inicio de JSON
   965→   965→        if len(text) < 5 and text in ['{', '"', '\n', '\r']:
   966→   966→            return True
   967→   967→        return False
