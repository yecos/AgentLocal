    30→def leer_pdf(ruta: str, pagina_inicio: int = None, pagina_fin: int = None) -> str:
    31→    """Lee el contenido de un archivo PDF. Extrae texto, tablas e info de paginas.
    32→
    33→    Args:
    34→        ruta: Ruta del archivo PDF
    35→        pagina_inicio: Pagina inicial (1-indexed, opcional)
    36→        pagina_fin: Pagina final (inclusive, opcional)
    37→    """
    38→    validation = validate_path(ruta)
    39→    if validation != ruta:
    40→        return validation
    41→
    42→    ruta = _resolve_path(ruta)
    43→    if not ruta:
    44→        return f"Archivo no encontrado: {ruta}"
    45→
    46→    if not ruta.lower().endswith('.pdf'):
    47→        return "ERROR: El archivo no es un PDF."
    48→
    49→    # Intentar con PyMuPDF (fitz)
    50→    result = _leer_pdf_fitz(ruta, pagina_inicio, pagina_fin)
    51→    if result is not None:
    52→        return result
    53→
    54→    # Intentar con pdfplumber
    55→    result = _leer_pdf_plumber(ruta, pagina_inicio, pagina_fin)
    56→    if result is not None:
    57→        return result
    58→
    59→    # Fallback: comando pdftotext
    60→    result = _leer_pdf_comando(ruta, pagina_inicio, pagina_fin)
    61→    if result is not None:
    62→        return result
    63→
    64→    return ("ERROR: No se pudo leer el PDF. Instala una libreria:\n"
    65→            "  pip install PyMuPDF   (recomendado)\n"
    66→            "  pip install pdfplumber (alternativa)\n"
    67→            "  O instala pdftotext: sudo apt install poppler-utils")
    68→
    69→
    70→def _leer_pdf_fitz(ruta, pagina_inicio, pagina_fin):
    71→    """Lee PDF con PyMuPDF (fitz)."""
    72→    try:
    73→        import fitz  # PyMuPDF
    74→        doc = fitz.open(ruta)
    75→
    76→        total_pages = len(doc)
    77→        start = max(0, (pagina_inicio or 1) - 1)
    78→        end = min(total_pages, pagina_fin or total_pages)
    79→
    80→        parts = [f"PDF: {os.path.basename(ruta)} ({total_pages} paginas)\n"]
    81→
    82→        for i in range(start, end):
    83→            page = doc[i]
    84→            text = page.get_text("text")
    85→            if text.strip():
    86→                parts.append(f"--- Pagina {i + 1} ---")
    87→                parts.append(text)
    88→
    89→            # Extraer tablas si hay
    90→            try:
    91→                tables = page.find_tables()
    92→                if tables and tables.tables:
    93→                    for ti, table in enumerate(tables.tables):
    94→                        table_data = table.extract()
    95→                        if table_data:
    96→                            parts.append(f"  Tabla {ti + 1}:")
    97→                            for row in table_data[:20]:
    98→                                parts.append("  | " + " | ".join(str(c or "") for c in row) + " |")
    99→            except Exception:
   100→                pass  # Tabla extraction no critica
   101→
   102→        doc.close()
   103→
   104→        content = "\n".join(parts)
   105→        if len(content) > MAX_FILE_READ:
   106→            content = content[:MAX_FILE_READ] + f"\n... [truncado, {total_pages} paginas total]"
   107→
   108→        return content
   109→
   110→    except ImportError:
   111→        return None
   112→    except Exception as e:
   113→        logger.debug(f"PyMuPDF fallo: {e}")
   114→        return None
   115→

   117→def _leer_pdf_plumber(ruta, pagina_inicio, pagina_fin):
   118→    """Lee PDF con pdfplumber (alternativa)."""
   119→    try:
   120→        import pdfplumber
   121→
   122→        with pdfplumber.open(ruta) as pdf:
   123→            total_pages = len(pdf.pages)
   124→            start = max(0, (pagina_inicio or 1) - 1)
   125→            end = min(total_pages, pagina_fin or total_pages)
   126→
   127→            parts = [f"PDF: {os.path.basename(ruta)} ({total_pages} paginas)\n"]
   128→
   129→            for i in range(start, end):
   130→                page = pdf.pages[i]
   131→                text = page.extract_text()
   132→                if text and text.strip():
   133→                    parts.append(f"--- Pagina {i + 1} ---")
   134→                    parts.append(text)
   135→
   136→                # Extraer tablas
   137→                try:
   138→                    tables = page.extract_tables()
   139→                    for ti, table in enumerate(tables[:3]):
   140→                        parts.append(f"  Tabla {ti + 1}:")
   141→                        for row in table[:20]:
   142→                            parts.append("  | " + " | ".join(str(c or "") for c in row) + " |")
   143→                except Exception:
   144→                    pass
   145→
   146→            content = "\n".join(parts)
   147→            if len(content) > MAX_FILE_READ:
   148→                content = content[:MAX_FILE_READ] + f"\n... [truncado, {total_pages} paginas total]"
   149→            return content
   150→
   151→    except ImportError:
   152→        return None
   153→    except Exception as e:
   154→        logger.debug(f"pdfplumber fallo: {e}")
   155→        return None

   158→def _leer_pdf_comando(ruta, pagina_inicio, pagina_fin):
   159→    """Fallback: lee PDF con comando pdftotext."""
   160→    import subprocess
   161→    try:
   162→        result = subprocess.run(
   163→            ["pdftotext", "-layout", ruta, "-"],
   164→            capture_output=True, text=True, timeout=30
   165→        )
   166→        if result.returncode == 0 and result.stdout.strip():
   167→            content = f"PDF: {os.path.basename(ruta)}\n\n{result.stdout}"
   168→            if len(content) > MAX_FILE_READ:
   169→                content = content[:MAX_FILE_READ] + "\n... [truncado]"
   170→            return content
   171→    except (FileNotFoundError, subprocess.TimeoutExpired):
   172→        pass
   173→    except Exception as e:
   174→        logger.debug(f"pdftotext fallo: {e}")
   175→    return None

   183→    """Lee el contenido de un archivo Word (.docx). Extrae texto, tablas y estilos.
   184→
   185→    Args:
   186→        ruta: Ruta del archivo .docx
   187→    """
   188→    validation = validate_path(ruta)
   189→    if validation != ruta:
   190→        return validation
   191→
   192→    ruta = _resolve_path(ruta)
   193→    if not ruta:
   194→        return f"Archivo no encontrado: {ruta}"
   195→
   196→    if not ruta.lower().endswith('.docx'):
   197→        return "ERROR: El archivo no es un .docx (para .doc viejo usar comando antiword)."
   198→
   199→    # Intentar con python-docx
   200→    result = _leer_docx_python(ruta)
   201→    if result is not None:
   202→        return result

   207→
   208→def _leer_docx_python(ruta):
   209→    """Lee DOCX con python-docx."""
   210→    try:
   211→        from docx import Document
   212→
   213→        doc = Document(ruta)
   214→        parts = [f"DOCX: {os.path.basename(ruta)}\n"]
   215→
   216→        # Extraer parrafos
   217→        for para in doc.paragraphs:
   218→            text = para.text.strip()
   219→            if text:
   220→                # Indicar estilo si es heading
   221→                style_name = para.style.name if para.style else ""
   222→                if "Heading" in style_name or "heading" in style_name:
   223→                    level = "".join(c for c in style_name if c.isdigit()) or "1"
   224→                    parts.append(f"{'#' * int(level)} {text}")
   225→                else:
   226→                    parts.append(text)
   227→
   228→        # Extraer tablas
   229→        for ti, table in enumerate(doc.tables):
   230→            parts.append(f"\nTabla {ti + 1}:")
   231→            for row in table.rows[:20]:
   232→                cells = [cell.text.strip() for cell in row.cells]
   233→                parts.append("| " + " | ".join(cells) + " |")
   234→
   235→        content = "\n".join(parts)
   236→        if len(content) > MAX_FILE_READ:
   237→            content = content[:MAX_FILE_READ] + "\n... [truncado]"
   238→        return content
   239→
   240→    except ImportError:
   241→        return None
   242→    except Exception as e:
   243→        logger.debug(f"python-docx fallo: {e}")
   244→        return f"ERROR leyendo DOCX: {e}"
   245→

   260→            capture_output=True, text=True, timeout=30
   261→        )
   262→        if result.returncode == 0 and result.stdout.strip():

   264→            if len(content) > MAX_FILE_READ:
   265→                content = content[:MAX_FILE_READ] + "\n... [truncado]"

   268→        pass
   269→

   271→            "  pip install python-docx   (recomendado)")

   279→    """Lee el contenido de un archivo Excel (.xlsx). Extrae datos de hojas y tablas.
   280→
   281→    Args:
   282→        ruta: Ruta del archivo .xlsx
   283→        hoja: Nombre de la hoja (opcional, por defecto la primera)
   284→        max_filas: Maximo de filas a leer por hoja (default 50)
   285→    """
   286→    validation = validate_path(ruta)
   287→    if validation != ruta:
   288→        return validation
   289→
   290→    ruta = _resolve_path(ruta)
   291→    if not ruta:
   292→        return f"Archivo no encontrado: {ruta}"
   293→
   294→    if not ruta.lower().endswith(('.xlsx', '.xls')):
   295→        return "ERROR: El archivo no es un Excel (.xlsx/.xls)."
   296→
   297→    # Intentar con openpyxl
   298→    result = _leer_xlsx_openpyxl(ruta, hoja, max_filas)
   299→    if result is not None:
   300→        return result
   301→
   302→    # Fallback: comando
   303→    return _leer_xlsx_comando(ruta, hoja, max_filas)

   305→
   306→def _leer_xlsx_openpyxl(ruta, hoja, max_filas):
   307→    """Lee XLSX con openpyxl."""
   308→    try:
   309→        from openpyxl import load_workbook
   310→
   311→        wb = load_workbook(ruta, read_only=True, data_only=True)
   312→        parts = [f"XLSX: {os.path.basename(ruta)} ({len(wb.sheetnames)} hojas)\n"]
   313→        parts.append(f"Hojas: {', '.join(wb.sheetnames)}\n")
   314→
   315→        # Determinar hojas a leer
   316→        if hoja:
   317→            if hoja in wb.sheetnames:
   318→                sheets = [wb[hoja]]
   319→            else:
   320→                return f"Hoja '{hoja}' no encontrada. Disponibles: {', '.join(wb.sheetnames)}"
   321→        else:
   322→            sheets = [wb[name] for name in wb.sheetnames[:3]]  # Max 3 hojas
   323→
   324→        for ws in sheets:
   325→            parts.append(f"=== Hoja: {ws.title} ===")
   326→
   327→            row_count = 0
   328→            for row in ws.iter_rows(values_only=True):
   329→                if row_count >= max_filas:
   330→                    parts.append(f"... [truncado, {max_filas} filas maximo]")
   331→                    break
   332→                cells = [str(c) if c is not None else "" for c in row]
   333→                if any(cells):  # Skip filas vacias
   334→                    parts.append("| " + " | ".join(cells) + " |")
   335→                    row_count += 1
   336→
   337→            parts.append(f"({row_count} filas leidas)\n")
   338→
   339→        wb.close()
   340→
   341→        content = "\n".join(parts)
   342→        if len(content) > MAX_FILE_READ:
   343→            content = content[:MAX_FILE_READ] + "\n... [truncado]"
   344→        return content
   345→
   346→    except ImportError:
   347→        return None
   348→    except Exception as e:
   349→        logger.debug(f"openpyxl fallo: {e}")
   350→        return f"ERROR leyendo XLSX: {e}"
   351→

   358→            f"python3 -c \""

   384→    """Lee el contenido de una presentacion PowerPoint (.pptx). Extrae texto de diapositivas.
   385→
   386→    Args:
   387→        ruta: Ruta del archivo .pptx
   388→    """
   389→    validation = validate_path(ruta)
   390→    if validation != ruta:
   391→        return validation
   392→
   393→    ruta = _resolve_path(ruta)
   394→    if not ruta:
   395→        return f"Archivo no encontrado: {ruta}"
   396→
   397→    if not ruta.lower().endswith('.pptx'):
   398→        return "ERROR: El archivo no es un .pptx."

   408→
   409→def _leer_pptx_python(ruta):
   410→    """Lee PPTX con python-pptx."""
   411→    try:
   412→        from pptx import Presentation
   413→
   414→        prs = Presentation(ruta)
   415→        parts = [f"PPTX: {os.path.basename(ruta)} ({len(prs.slides)} diapositivas)\n"]
   416→
   417→        for i, slide in enumerate(prs.slides):
   418→            parts.append(f"=== Diapositiva {i + 1} ===")
   419→
   420→            for shape in slide.shapes:
   421→                if shape.has_text_frame:
   422→                    for para in shape.text_frame.paragraphs:
   423→                        text = para.text.strip()
   424→                        if text:
   425→                            parts.append(text)
   426→
   427→                # Tablas en diapositivas
   428→                if shape.has_table:
   429→                    table = shape.table
   430→                    for row in table.rows:
   431→                        cells = [cell.text.strip() for cell in row.cells]
   432→                        parts.append("| " + " | ".join(cells) + " |")
   433→
   434→            parts.append("")  # Separador entre diapositivas
   435→
   436→        content = "\n".join(parts)
   437→        if len(content) > MAX_FILE_READ:
   438→            content = content[:MAX_FILE_READ] + "\n... [truncado]"
   439→        return content
   440→
   441→    except ImportError:
   442→        return None
   443→    except Exception as e:
   444→        logger.debug(f"python-pptx fallo: {e}")
   445→        return f"ERROR leyendo PPTX: {e}"

   474→# ============================================================
   475→
   476→def leer_csv(ruta: str, separador: str = ",", max_filas: int = 50) -> str:
   477→    """Lee un archivo CSV de forma estructurada. Detecta delimitador y codificacion.
   478→
   479→    Args:
   480→        ruta: Ruta del archivo CSV
   481→        separador: Separador de columnas (default: coma)
   482→        max_filas: Maximo de filas a leer (default 50)
   483→    """
   484→    validation = validate_path(ruta)
   485→    if validation != ruta:
   486→        return validation
   487→
   488→    ruta = _resolve_path(ruta)
   489→    if not ruta:
   490→        return f"Archivo no encontrado: {ruta}"
   491→
   492→    try:
   493→        # Intentar detectar codificacion
   494→        encodings = ['utf-8', 'latin-1', 'cp1252', 'utf-16']
   495→        content = None
   496→
   497→        for enc in encodings:
   498→            try:
   499→                with open(ruta, 'r', encoding=enc) as f:
   500→                    content = f.read()
   501→                break
   502→            except (UnicodeDecodeError, UnicodeError):
   503→                continue
   504→
   505→        if content is None:
   506→            return "ERROR: No se pudo detectar la codificacion del CSV."
   507→
   508→        # Parsear CSV
   509→        reader = csv.reader(io.StringIO(content), delimiter=separador)
   510→        rows = list(reader)
   511→
   512→        if not rows:
   513→            return "CSV vacio."
   514→
   515→        headers = rows[0] if rows else []
   516→        data_rows = rows[1:max_filas + 1]
   517→
   518→        parts = [f"CSV: {os.path.basename(ruta)} ({len(rows) - 1} filas, {len(headers)} columnas)\n"]
   519→        parts.append(f"Columnas: {', '.join(headers)}\n")
   520→
   521→        # Formato tabla
   522→        for row in data_rows:
   523→            # Rellenar filas cortas
   524→            while len(row) < len(headers):
   525→                row.append("")
   526→            parts.append("| " + " | ".join(row[:len(headers)]) + " |")
   527→
   528→        if len(rows) - 1 > max_filas:
   529→            parts.append(f"\n... [truncado, mostrando {max_filas} de {len(rows) - 1} filas]")
   530→
   531→        result = "\n".join(parts)
   532→        if len(result) > MAX_FILE_READ:
   533→            result = result[:MAX_FILE_READ] + "\n... [truncado]"
   534→        return result
   535→
   536→    except Exception as e:
   537→        return f"ERROR leyendo CSV: {e}"

   541→# LECTURA DE ARCHIVOS COMPRIMIDOS

   543→
   544→def leer_archivo_comprimido(ruta: str, archivo_interno: str = None) -> str:
   545→    """Lista o extrae contenido de archivos ZIP, TAR, GZ, RAR.
   546→
   547→    Args:
   548→        ruta: Ruta del archivo comprimido
   549→        archivo_interno: Archivo interno a extraer (opcional, si no se especifica lista contenido)
   550→    """
   551→    validation = validate_path(ruta)
   552→    if validation != ruta:
   553→        return validation
   554→
   555→    ruta = _resolve_path(ruta)
   556→    if not ruta:
   557→        return f"Archivo no encontrado: {ruta}"
   558→
   559→    ext = ruta.lower()
   560→
   561→    if ext.endswith('.zip'):
   562→        return _leer_zip(ruta, archivo_interno)
   563→    elif ext.endswith(('.tar', '.tar.gz', '.tgz', '.tar.bz2', '.tbz2')):
   564→        return _leer_tar(ruta, archivo_interno)
   565→    else:
   566→        return "ERROR: Formato no soportado. Usar .zip, .tar, .tar.gz, .tar.bz2"
   567→

   569→def _leer_zip(ruta, archivo_interno):
   570→    """Lee contenido de archivo ZIP."""
   571→    try:
   572→        import zipfile
   573→        with zipfile.ZipFile(ruta, 'r') as zf:
   574→            names = zf.namelist()
   575→
   576→            if archivo_interno:
   577→                # Extraer archivo interno especifico
   578→                if archivo_interno in names:
   579→                    with zf.open(archivo_interno) as f:
   580→                        content = f.read()
   581→                        try:
   582→                            text = content.decode('utf-8')
   583→                        except UnicodeDecodeError:
   584→                            text = content.decode('latin-1')
   585→
   586→                        if len(text) > MAX_FILE_READ:
   587→                            text = text[:MAX_FILE_READ] + "\n... [truncado]"
   588→                        return f"Contenido de {archivo_interno}:\n\n{text}"
   589→                else:
   590→                    return f"Archivo '{archivo_interno}' no encontrado en ZIP. Disponibles:\n" + "\n".join(f"  {n}" for n in names[:30])
   591→            else:
   592→                # Listar contenido
   593→                parts = [f"ZIP: {os.path.basename(ruta)} ({len(names)} archivos)\n"]
   594→                for name in names[:50]:
   595→                    info = zf.getinfo(name)
   596→                    size = info.file_size
   597→                    parts.append(f"  {'[DIR]' if name.endswith('/') else '[FILE]'} {name} ({size:,} bytes)")
   598→
   599→                if len(names) > 50:
   600→                    parts.append(f"\n... y {len(names) - 50} archivos mas")
   601→
   602→                return "\n".join(parts)
   603→
   604→    except Exception as e:
   605→        return f"ERROR leyendo ZIP: {e}"

   608→def _leer_tar(ruta, archivo_interno):
   609→    """Lee contenido de archivo TAR."""
   610→    try:
   611→        import tarfile
   612→        with tarfile.open(ruta, 'r:*') as tf:
   613→            members = tf.getmembers()
   614→
   615→            if archivo_interno:
   616→                for member in members:
   617→                    if member.name == archivo_interno:
   618→                        f = tf.extractfile(member)
   619→                        if f:
   620→                            content = f.read()
   621→                            try:
   622→                                text = content.decode('utf-8')
   623→                            except UnicodeDecodeError:
   624→                                text = content.decode('latin-1')
   625→                            if len(text) > MAX_FILE_READ:
   626→                                text = text[:MAX_FILE_READ] + "\n... [truncado]"
   627→                            return f"Contenido de {archivo_interno}:\n\n{text}"
   628→                return f"Archivo '{archivo_interno}' no encontrado en TAR."
   629→
   630→            # Listar contenido
   631→            parts = [f"TAR: {os.path.basename(ruta)} ({len(members)} archivos)\n"]
   632→            for m in members[:50]:
   633→                type_str = "[DIR]" if m.isdir() else "[FILE]"
   634→                parts.append(f"  {type_str} {m.name} ({m.size:,} bytes)")
   635→
   636→            if len(members) > 50:
   637→                parts.append(f"\n... y {len(members) - 50} archivos mas")
   638→
   639→            return "\n".join(parts)
   640→
   641→    except Exception as e:
   642→        return f"ERROR leyendo TAR: {e}"

   649→def consultar_sqlite(ruta: str, consulta: str = None, tabla: str = None, max_filas: int = 50) -> str:
   650→    """Consulta una base de datos SQLite. Puede listar tablas o ejecutar consultas SELECT.
   651→
   652→    Args:
   653→        ruta: Ruta del archivo .db o .sqlite
   654→        consulta: Consulta SQL SELECT (opcional)
   655→        tabla: Nombre de tabla para ver contenido (opcional)
   656→        max_filas: Maximo de filas a retornar (default 50)
   657→    """
   658→    validation = validate_path(ruta)
   659→    if validation != ruta:
   660→        return validation
   661→
   662→    ruta = _resolve_path(ruta)
   663→    if not ruta:
   664→        return f"Archivo no encontrado: {ruta}"
   665→
   666→    try:
   667→        import sqlite3
   668→        conn = sqlite3.connect(ruta)
   669→        cursor = conn.cursor()
   670→
   671→        if consulta:
   672→            # Seguridad: solo permitir SELECT
   673→            consulta_upper = consulta.strip().upper()
   674→            if not consulta_upper.startswith("SELECT") and not consulta_upper.startswith("PRAGMA"):
   675→                conn.close()
   676→                return "ERROR: Solo se permiten consultas SELECT o PRAGMA por seguridad."
   677→
   678→            # Bloquear comandos peligrosos
   679→            dangerous = ["DROP", "DELETE", "UPDATE", "INSERT", "ALTER", "CREATE", "ATTACH", "DETACH"]
   680→            for word in dangerous:
   681→                if word in consulta_upper.split():
   682→                    conn.close()
   683→                    return f"ERROR: Comando {word} no permitido por seguridad."
   684→
   685→            cursor.execute(consulta)
   686→            rows = cursor.fetchall()
   687→            columns = [desc[0] for desc in cursor.description] if cursor.description else []
   688→
   689→            parts = [f"Consulta: {consulta}"]
   690→            if columns:
   691→                parts.append("| " + " | ".join(columns) + " |")
   692→            for row in rows[:max_filas]:
   693→                parts.append("| " + " | ".join(str(c) for c in row) + " |")
   694→
   695→            if len(rows) > max_filas:
   696→                parts.append(f"\n... [truncado, {len(rows)} filas total]")
   697→
   698→            parts.append(f"\n({len(rows)} filas)")
   699→            conn.close()
   700→            return "\n".join(parts)
   701→
   702→        elif tabla:
   703→            # Ver contenido de una tabla
   704→            cursor.execute(f'SELECT * FROM "{tabla}" LIMIT {max_filas}')
   705→            rows = cursor.fetchall()
   706→            columns = [desc[0] for desc in cursor.description]
   707→
   708→            parts = [f"Tabla: {tabla} ({len(rows)} filas mostradas)\n"]
   709→            parts.append("| " + " | ".join(columns) + " |")
   710→            for row in rows:
   711→                parts.append("| " + " | ".join(str(c) for c in row) + " |")
   712→
   713→            conn.close()
   714→            return "\n".join(parts)
   715→
   716→        else:
   717→            # Listar tablas
   718→            cursor.execute("SELECT name FROM sqlite_master WHERE type='table' ORDER BY name")
   719→            tables = cursor.fetchall()
   720→
   721→            parts = [f"SQLite: {os.path.basename(ruta)} ({len(tables)} tablas)\n"]
   722→
   723→            for (table_name,) in tables:
   724→                cursor.execute(f'SELECT COUNT(*) FROM "{table_name}"')
   725→                count = cursor.fetchone()[0]
   726→                parts.append(f"  {table_name} ({count:,} filas)")
   727→
   728→                # Mostrar schema
   729→                cursor.execute(f'PRAGMA table_info("{table_name}")')
   730→                cols = cursor.fetchall()
   731→                col_info = ", ".join(f"{c[1]}({c[2]})" for c in cols[:8])
   732→                parts.append(f"    Columnas: {col_info}")
   733→
   734→            conn.close()
   735→            return "\n".join(parts)
   736→
   737→    except ImportError:
   738→        return "ERROR: sqlite3 no disponible (viene con Python estandar, esto no deberia pasar)."
   739→    except Exception as e:
   740→        return f"ERROR consultando SQLite: {e}"
   741→

   747→def leer_epub(ruta: str, max_capitulos: int = 10) -> str:
   748→    """Lee el contenido de un libro electronico ePub. Extrae texto de capitulos.
   749→
   750→    Args:
   751→        ruta: Ruta del archivo .epub
   752→        max_capitulos: Maximo de capitulos a leer (default 10)
   753→    """
   754→    validation = validate_path(ruta)
   755→    if validation != ruta:
   756→        return validation
   757→
   758→    ruta = _resolve_path(ruta)
   759→    if not ruta:
   760→        return f"Archivo no encontrado: {ruta}"
   761→
   762→    # Intentar con ebooklib
   763→    try:
   764→        import ebooklib
   765→        from ebooklib import epub
   766→        from html.parser import HTMLParser
   767→
   768→        class _TextExtractor(HTMLParser):
   769→            def __init__(self):
   770→                super().__init__()
   771→                self.text = []
   772→                self.skip = False
   773→
   774→            def handle_starttag(self, tag, attrs):
   775→                if tag in ('script', 'style'):
   776→                    self.skip = True
   777→
   778→            def handle_endtag(self, tag):
   779→                if tag in ('script', 'style'):
   780→                    self.skip = False
   781→                elif tag in ('p', 'div', 'h1', 'h2', 'h3', 'br'):
   782→                    self.text.append('\n')
   783→
   784→            def handle_data(self, data):
   785→                if not self.skip:
   786→                    self.text.append(data.strip())
   787→
   788→        book = epub.read_epub(ruta)
   789→        parts = [f"ePub: {os.path.basename(ruta)}\n"]
   790→
   791→        # Metadata
   792→        title = book.get_metadata('DC', 'title')
   793→        author = book.get_metadata('DC', 'author')
   794→        if title:
   795→            parts.append(f"Titulo: {title[0][0]}")
   796→        if author:
   797→            parts.append(f"Autor: {author[0][0]}")
   798→
   799→        # Extraer texto de documentos
   800→        chapters = 0
   801→        for item in book.get_items_of_type(ebooklib.ITEM_DOCUMENT):
   802→            if chapters >= max_capitulos:
   803→                break
   804→            content = item.get_content().decode('utf-8', errors='replace')
   805→            extractor = _TextExtractor()
   806→            extractor.feed(content)
   807→            text = " ".join(extractor.text).strip()
   808→            if text and len(text) > 50:
   809→                parts.append(f"\n--- Capitulo {chapters + 1} ---")
   810→                parts.append(text[:2000])
   811→                chapters += 1
   812→
   813→        content = "\n".join(parts)
   814→        if len(content) > MAX_FILE_READ:
   815→            content = content[:MAX_FILE_READ] + "\n... [truncado]"
   816→        return content
   817→
   818→    except ImportError:
   819→        # Fallback: tratar como ZIP y buscar HTML
   820→        return _leer_epub_zip(ruta, max_capitulos)
   821→    except Exception as e:
   822→        return f"ERROR leyendo ePub: {e}"
   823→
   824→
   825→def _leer_epub_zip(ruta, max_capitulos):
   826→    """Fallback: lee ePub como ZIP (formato basico)."""
   827→    try:
   828→        import zipfile
   829→        from html.parser import HTMLParser
   830→
   831→        class _SimpleText(HTMLParser):
   832→            def __init__(self):
   833→                super().__init__()
   834→                self.text = []
   835→            def handle_data(self, data):
   836→                self.text.append(data.strip())
   837→
   838→        with zipfile.ZipFile(ruta, 'r') as zf:
   839→            html_files = [n for n in zf.namelist() if n.endswith(('.html', '.xhtml'))]
   840→
   841→            parts = [f"ePub (basico): {os.path.basename(ruta)} ({len(html_files)} secciones)\n"]
   842→
   843→            for i, name in enumerate(html_files[:max_capitulos]):
   844→                content = zf.open(name).read().decode('utf-8', errors='replace')
   845→                extractor = _SimpleText()
   846→                extractor.feed(content)
   847→                text = " ".join(t for t in extractor.text if t)
   848→                if text:
   849→                    parts.append(f"--- Seccion {i + 1} ---")
   850→                    parts.append(text[:1500])
   851→
   852→            content = "\n".join(parts)
   853→            if len(content) > MAX_FILE_READ:
   854→                content = content[:MAX_FILE_READ] + "\n... [truncado]"
   855→            return content
   856→
   857→    except Exception as e:
   858→        return f"ERROR leyendo ePub: {e}\nInstala: pip install ebooklib"

   862→# LECTOR UNIVERSAL (auto-detectar formato)

   864→
   865→def leer_documento(ruta: str, **kwargs) -> str:
   866→    """Lee cualquier documento detectando automaticamente el formato.
   867→    Soporta: PDF, DOCX, XLSX, PPTX, CSV, ZIP, TAR, SQLite, ePub, y texto plano.
   868→
   869→    Args:
   870→        ruta: Ruta del archivo a leer
   871→        **kwargs: Argumentos adicionales segun el tipo (hoja, consulta, etc.)
   872→    """
   873→    validation = validate_path(ruta)
   874→    if validation != ruta:
   875→        return validation
   876→
   877→    ruta_resolved = _resolve_path(ruta)
   878→    if not ruta_resolved:
   879→        return f"Archivo no encontrado: {ruta}"
   880→
   881→    ext = ruta_resolved.lower()
   882→
   883→    if ext.endswith('.pdf'):
   884→        return leer_pdf(ruta_resolved, **{k: v for k, v in kwargs.items() if k in ('pagina_inicio', 'pagina_fin')})
   885→    elif ext.endswith('.docx'):
   886→        return leer_docx(ruta_resolved)
   887→    elif ext.endswith(('.xlsx', '.xls')):
   888→        return leer_xlsx(ruta_resolved, **{k: v for k, v in kwargs.items() if k in ('hoja', 'max_filas')})
   889→    elif ext.endswith('.pptx'):
   890→        return leer_pptx(ruta_resolved)
   891→    elif ext.endswith('.csv'):
   892→        return leer_csv(ruta_resolved, **{k: v for k, v in kwargs.items() if k in ('separador', 'max_filas')})
   893→    elif ext.endswith(('.zip', '.tar', '.tar.gz', '.tgz', '.tar.bz2', '.tbz2')):
   894→        return leer_archivo_comprimido(
   895→            ruta_resolved,
   896→            **{k: v for k, v in kwargs.items() if k in ('archivo_interno',)}
   897→        )
   898→    elif ext.endswith(('.db', '.sqlite', '.sqlite3')):
   899→        return consultar_sqlite(
   900→            ruta_resolved,
   901→            **{k: v for k, v in kwargs.items() if k in ('consulta', 'tabla', 'max_filas')}
   902→        )
   903→    elif ext.endswith('.epub'):
   904→        return leer_epub(ruta_resolved)
   905→    else:
   906→        # Fallback: intentar como texto
   907→        try:
   908→            with open(ruta_resolved, 'r', encoding='utf-8', errors='replace') as f:
   909→                content = f.read()
   910→            if len(content) > MAX_FILE_READ:
   911→                content = content[:MAX_FILE_READ] + "\n... [truncado]"
   912→            return content
   913→        except Exception as e:
   914→            return f"ERROR: No se pudo leer {ruta_resolved}: {e}"

   920→
   921→def _resolve_path(ruta):
   922→    """Resuelve la ruta del archivo, buscando en ubicaciones alternativas."""
   923→    if os.path.isabs(ruta) and os.path.exists(ruta):
   924→        return ruta
   925→
   926→    # Ruta relativa
   927→    if os.path.exists(ruta):
   928→        return ruta
   929→
   930→    # Buscar en REPOS_DIR
   931→    alt = os.path.join(REPOS_DIR, ruta)
   932→    if os.path.exists(alt):
   933→        return alt
   934→
   935→    # Buscar en subdirectorios
   936→    try:
   937→        for d in os.listdir(REPOS_DIR):
   938→            alt2 = os.path.join(REPOS_DIR, d, ruta)
   939→            if os.path.exists(alt2):
   940→                return alt2
   941→    except OSError:
   942→        pass
   943→
   944→    # Buscar en home
   945→    home = os.path.expanduser("~")
   946→    alt3 = os.path.join(home, ruta)
   947→    if os.path.exists(alt3):
   948→        return alt3
   949→
   950→    return None
