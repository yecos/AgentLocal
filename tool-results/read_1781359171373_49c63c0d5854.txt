   488→                "accion_stash": {"type": "string", "description": "Para stash: save, list, pop, drop"},
   489→                "staged": {"type": "boolean", "description": "Para diff: mostrar solo staged (default: false)"},
   490→                "count": {"type": "integer", "description": "Para log: numero de commits (default: 10)"}
   491→            },
   492→            "required": ["operacion"]
   493→        }
   494→    }
   495→})
   496→def git_operacion(operacion: str, ruta_repo: str = "", mensaje: str = "",
   497→                  branch: str = "", archivos: str = "",
   498→                  accion_branch: str = "list", accion_stash: str = "save",
   499→                  staged: bool = False, count: int = 10) -> str:
   500→    """Ejecuta operaciones Git de forma estructurada."""
   501→    from . import git_tool as gt
   502→
   503→    path = ruta_repo or None
   504→    op = operacion.lower().strip()
   505→
   506→    if op == "status":
   507→        result = gt.git_status(path)
   508→    elif op == "diff":
   509→        result = gt.git_diff(path, staged=staged)
   510→    elif op == "add":
   511→        files = [f.strip() for f in archivos.split(",")] if archivos and archivos != "all" else None
   512→        result = gt.git_add(path, files=files, all_changes=archivos == "all" or not archivos)
   513→    elif op == "commit":
   514→        result = gt.git_commit(path, message=mensaje or None, add_all=False)
   515→    elif op == "branch":
   516→        result = gt.git_branch(path, action=accion_branch, branch_name=branch or None)
   517→    elif op == "log":
   518→        result = gt.git_log(path, count=count)
   519→    elif op == "push":
   520→        result = gt.git_push(path, branch=branch or None)
   521→    elif op == "pull":
   522→        result = gt.git_pull(path, branch=branch or None)
   523→    elif op == "stash":
   524→        result = gt.git_stash(path, action=accion_stash, message=mensaje)
   525→    elif op == "init":
   526→        result = gt.git_init(path)
   527→    else:
   528→        return f"Operacion no soportada: {op}. Usa: status, diff, add, commit, branch, log, push, pull, stash, init"
   529→
   530→    if result.get("success"):
   531→        # Formatear resultado para el agente
   532→        return json.dumps(result, ensure_ascii=False, indent=2, default=str)[:2000]
   533→    return f"ERROR: {result.get('error', 'Error desconocido')}"
