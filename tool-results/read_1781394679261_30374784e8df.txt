   257тЖТ   257тЖТdef _build_graph(tipo, datos):
   258тЖТ   258тЖТ    """Construye un grafo networkx desde los datos JSON."""
   259тЖТ   259тЖТ    import networkx as nx
   260тЖТ   260тЖТ
   261тЖТ   261тЖТ    G = nx.DiGraph() if tipo in ("tree", "org", "flowchart", "class", "sequence") else nx.Graph()
   262тЖТ   262тЖТ
   263тЖТ   263тЖТ    # Nodos
   264тЖТ   264тЖТ    nodes = datos.get("nodes", datos.get("nodos", []))
   265тЖТ   265тЖТ    for node in nodes:
   266тЖТ   266тЖТ        if isinstance(node, str):
   267тЖТ   267тЖТ            G.add_node(node, label=node)
   268тЖТ   268тЖТ        elif isinstance(node, dict):
   269тЖТ   269тЖТ            nid = node.get("id", node.get("nombre", ""))
   270тЖТ   270тЖТ            label = node.get("label", node.get("nombre", node.get("name", nid)))
   271тЖТ   271тЖТ            G.add_node(nid, label=label, **{k: v for k, v in node.items()
   272тЖТ   272тЖТ                                            if k not in ("id", "label", "nombre", "name")})
   273тЖТ   273тЖТ
   274тЖТ   274тЖТ    # Aristas
   275тЖТ   275тЖТ    edges = datos.get("edges", datos.get("aristas", datos.get("conexiones", [])))
   276тЖТ   276тЖТ    for edge in edges:
   277тЖТ   277тЖТ        if isinstance(edge, (list, tuple)) and len(edge) >= 2:
   278тЖТ   278тЖТ            G.add_edge(edge[0], edge[1], label=str(edge[2]) if len(edge) > 2 else "")
   279тЖТ   279тЖТ        elif isinstance(edge, dict):
   280тЖТ   280тЖТ            src = edge.get("from", edge.get("source", edge.get("desde", "")))
   281тЖТ   281тЖТ            dst = edge.get("to", edge.get("target", edge.get("hasta", "")))
   282тЖТ   282тЖТ            label = edge.get("label", edge.get("etiqueta", ""))
   283тЖТ   283тЖТ            G.add_edge(src, dst, label=label)
   284тЖТ   284тЖТ
   285тЖТ   285тЖТ    # Si no hay nodos explicitos, generar desde aristas
   286тЖТ   286тЖТ    if not nodes and not edges:
   287тЖТ   287тЖТ        # Intentar formato simple: jerarquia
   288тЖТ   288тЖТ        hierarchy = datos.get("hierarchy", datos.get("jerarquia", {}))
   289тЖТ   289тЖТ        if hierarchy:
   290тЖТ   290тЖТ            _add_hierarchy(G, hierarchy, root=None)
   291тЖТ   291тЖТ
   292тЖТ   292тЖТ    return G
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ
   295тЖТ   295тЖТdef _add_hierarchy(G, data, root=None):
   296тЖТ   296тЖТ    """Agrega nodos y aristas desde una estructura jerarquica."""
   297тЖТ   297тЖТ    if isinstance(data, dict):
   298тЖТ   298тЖТ        for key, children in data.items():
   299тЖТ   299тЖТ            G.add_node(key, label=key)
   300тЖТ   300тЖТ            if root is not None:
   301тЖТ   301тЖТ                G.add_edge(root, key)
   302тЖТ   302тЖТ            if isinstance(children, dict):
   303тЖТ   303тЖТ                _add_hierarchy(G, children, root=key)
   304тЖТ   304тЖТ            elif isinstance(children, list):
   305тЖТ   305тЖТ                for child in children:
   306тЖТ   306тЖТ                    if isinstance(child, str):
   307тЖТ   307тЖТ                        G.add_node(child, label=child)
   308тЖТ   308тЖТ                        G.add_edge(key, child)
   309тЖТ   309тЖТ                    elif isinstance(child, dict):
   310тЖТ   310тЖТ                        _add_hierarchy(G, child, root=key)
   311тЖТ   311тЖТ    elif isinstance(data, list):
   312тЖТ   312тЖТ        for item in data:
   313тЖТ   313тЖТ            if isinstance(item, str):
   314тЖТ   314тЖТ                G.add_node(item, label=item)
   315тЖТ   315тЖТ                if root is not None:
   316тЖТ   316тЖТ                    G.add_edge(root, item)

   660тЖТ   660тЖТdef _mermaid_flowchart(datos, titulo=""):

   662тЖТ   662тЖТ    direction = datos.get("direction", datos.get("direccion", "TD"))

   674тЖТ   674тЖТ            safe_id = _mermaid_id(node)
   675тЖТ   675тЖТ            lines.append(f"    {safe_id}[\"{node}\"]")
   676тЖТ   676тЖТ        elif isinstance(node, dict):
   677тЖТ   677тЖТ            nid = _mermaid_id(node.get("id", node.get("nombre", "")))

   683тЖТ   683тЖТ                lines.append(f"    {nid}(\"{label}\")")

   685тЖТ   685тЖТ                lines.append(f"    {nid}((\"{label}\"))")

   687тЖТ   687тЖТ                lines.append(f"    {nid}([\"{label}\"])")
   688тЖТ   688тЖТ            else:
   689тЖТ   689тЖТ                lines.append(f"    {nid}[\"{label}\"]")
   690тЖТ   690тЖТ
   691тЖТ   691тЖТ    # Aristas
   692тЖТ   692тЖТ    for edge in edges:
   693тЖТ   693тЖТ        if isinstance(edge, (list, tuple)) and len(edge) >= 2:
   694тЖТ   694тЖТ            src = _mermaid_id(str(edge[0]))
   695тЖТ   695тЖТ            dst = _mermaid_id(str(edge[1]))
   696тЖТ   696тЖТ            label = str(edge[2]) if len(edge) > 2 else ""
   697тЖТ   697тЖТ            arrow = "-->" + f"|{label}|" if label else "-->"
   698тЖТ   698тЖТ            lines.append(f"    {src} {arrow} {dst}")
   699тЖТ   699тЖТ        elif isinstance(edge, dict):
   700тЖТ   700тЖТ            src = _mermaid_id(edge.get("from", edge.get("source", "")))
   701тЖТ   701тЖТ            dst = _mermaid_id(edge.get("to", edge.get("target", "")))
   702тЖТ   702тЖТ            label = edge.get("label", edge.get("etiqueta", ""))
   703тЖТ   703тЖТ            arrow = "-->" + f"|{label}|" if label else "-->"
   704тЖТ   704тЖТ            lines.append(f"    {src} {arrow} {dst}")
   705тЖТ   705тЖТ
   706тЖТ   706тЖТ    # Jerarquia
   707тЖТ   707тЖТ    hierarchy = datos.get("hierarchy", datos.get("jerarquia", {}))
   708тЖТ   708тЖТ    if hierarchy and not nodes and not edges:

   711тЖТ   711тЖТ    return "\n".join(lines)

   895тЖТ   895тЖТ                for child in children:
